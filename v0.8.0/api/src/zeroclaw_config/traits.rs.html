17/// Runtime type classification for config property values.
18#[derive(Debug, Clone, Copy, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
19#[serde(rename_all = "snake_case")]
20pub enum PropKind {
21    String,
22    Bool,
23    Integer,
24    Float,
25    /// An enum or other serde-serializable type (parsed as TOML string).
26    Enum,
27    /// A `Vec<String>` field; set via comma-separated input.
28    StringArray,
29    /// A `Vec<T>` field where `T` is a serializable struct (e.g. `Vec<McpServerConfig>`,
30    /// `Vec<PeripheralBoardConfig>`). Round-tripped on the wire as a JSON array of
31    /// objects; the dashboard renders a per-row sub-form using the JSON Schema
32    /// from `OPTIONS /api/config` to discover the element type's field shape.
33    /// Schema v3 / #5947 will migrate the load-bearing ones (mcp.servers etc.)
34    /// to `HashMap<String, T>` keyed tables; until then this kind covers them.
35    ObjectArray,
36    /// A struct-shaped scalar field (e.g. `Option<ModelPricing>`). Round-tripped
37    /// on the wire as a JSON object; the dashboard renders a sub-form for the
38    /// inner fields using the JSON Schema from `OPTIONS /api/config`. Distinct
39    /// from `String`, which inserts the raw value as a TOML string and breaks
40    /// the serde round-trip for typed structs.
41    Object,
42}

44/// Maps Rust types to PropKind at compile time.
45/// Scalars have explicit impls; the blanket impl catches everything
46/// else as `PropKind::Enum`.
47pub trait HasPropKind {
48    const PROP_KIND: PropKind;
49
50    /// Terminal field names whose values must be redacted when this type is
51    /// displayed as an object/object-array prop. Most prop kinds have no
52    /// nested secret surface; Configurable object-array element types can
53    /// override this by delegating to their generated `secret_field_terminals`.
54    fn display_secret_terminals() -> Vec<&'static str> {
55        Vec::new()
56    }
57}

352/// Mask and restore secret fields on config structs.
353///
354/// Automatically implemented by `#[derive(Configurable)]` for any struct that
355/// has fields annotated with `#[secret]` or `#[nested]`. A blanket impl covers
356/// `HashMap<String, T: MaskSecrets>` so the trait propagates through alias maps
357/// without any per-type boilerplate.
358pub trait MaskSecrets {
359    fn mask_secrets(&mut self);
360    fn restore_secrets_from(&mut self, current: &Self);
361}
362
363impl<T: MaskSecrets> MaskSecrets for std::collections::HashMap<String, T> {
364    fn mask_secrets(&mut self) {
365        for v in self.values_mut() {
366            v.mask_secrets();
367        }
368    }
369    fn restore_secrets_from(&mut self, current: &Self) {
370        for (k, v) in self.iter_mut() {
371            if let Some(cur) = current.get(k) {
372                v.restore_secrets_from(cur);
373            }
374        }
375    }
376}
377
378impl<T: MaskSecrets> MaskSecrets for Vec<T> {
379    fn mask_secrets(&mut self) {
380        for v in self.iter_mut() {
381            v.mask_secrets();
382        }
383    }
384    fn restore_secrets_from(&mut self, current: &Self) {
385        for (v, cur) in self.iter_mut().zip(current.iter()) {
386            v.restore_secrets_from(cur);
387        }
388    }
389}

397/// Per-field secret operations the `Configurable` derive emits for every
398/// `#[secret]` field. Generalizes mask / restore / encrypt / decrypt / is_set
399/// across the supported shapes â€” `String`, `Option<String>`, `Vec<String>`,
400/// `HashMap<String, String>`, and `Option<HashMap<String, String>>` â€” so adding
401/// a new shape is a single trait impl rather than a fourth branch in the macro.
402///
403/// `encrypt_in_place` and `decrypt_in_place` are idempotent: encrypting an
404/// already-`enc2:`-prefixed value or decrypting a plaintext value is a no-op,
405/// detected via [`crate::security::SecretStore::is_encrypted`]. The `field`
406/// argument is the dotted config-path (e.g. `mcp.servers`); the impls suffix
407/// per-element coordinates (`[<idx>]` for `Vec`, `.<key>` for `HashMap`) so
408/// error messages point at the exact failed entry.
409pub trait SecretField {
410    /// Replace each non-empty inner string with [`MASKED_SECRET`].
411    fn mask(&mut self);
412
413    /// Restore inner strings that currently equal [`MASKED_SECRET`] from the
414    /// matching position in `current`. The dashboard write path relies on this
415    /// so re-posting an already-displayed masked value doesn't overwrite the
416    /// real secret in config.
417    fn restore_from(&mut self, current: &Self);
418
419    /// Encrypt every non-empty, not-already-encrypted inner string.
420    fn encrypt_in_place(
421        &mut self,
422        store: &crate::security::SecretStore,
423        field: &str,
424    ) -> anyhow::Result<()>;
425
426    /// Inverse of [`Self::encrypt_in_place`].
427    fn decrypt_in_place(
428        &mut self,
429        store: &crate::security::SecretStore,
430        field: &str,
431    ) -> anyhow::Result<()>;
432
433    /// Whether the field carries at least one non-empty inner string. Reported
434    /// back through [`SecretFieldInfo::is_set`].
435    fn is_set(&self) -> bool;
436}
437
438impl SecretField for String {
439    fn mask(&mut self) {
440        if !self.is_empty() {
441            *self = MASKED_SECRET.to_string();
442        }
443    }
444
445    fn restore_from(&mut self, current: &Self) {
446        if is_masked_secret(self) {
447            self.clone_from(current);
448        }
449    }
450
451    fn encrypt_in_place(
452        &mut self,
453        store: &crate::security::SecretStore,
454        field: &str,
455    ) -> anyhow::Result<()> {
456        use anyhow::Context;
457        // `is_encrypted` also includes external secret references (`op://`):
458        // encryption must preserve them, while decryption resolves them for use.
459        if !self.is_empty() && !crate::security::SecretStore::is_encrypted(self) {
460            *self = store
461                .encrypt(self)
462                .with_context(|| format!("Failed to encrypt {field}"))?;
463        }
464        Ok(())
465    }
466
467    fn decrypt_in_place(
468        &mut self,
469        store: &crate::security::SecretStore,
470        field: &str,
471    ) -> anyhow::Result<()> {
472        use anyhow::Context;
473        if crate::security::SecretStore::is_encrypted(self) {
474            *self = store
475                .decrypt(self)
476                .with_context(|| format!("Failed to decrypt {field}"))?;
477        }
478        Ok(())
479    }
480
481    fn is_set(&self) -> bool {
482        !self.is_empty()
483    }
484}

667    /// Dotted section path, e.g. `providers.models`, `mcp.servers`.
668    pub path: &'static str,
669    /// Whether the section is a map or a list.
670    pub kind: MapKeyKind,
671    /// Rust type name of the value, e.g. `ModelProviderConfig`. For display only.
672    pub value_type: &'static str,
673    /// Doc comment on the field (flattened to one line). What the user sees
674    /// when picking which kind of thing to add.
675    pub description: &'static str,
676    /// For `Kind::List` Vec sections that declared `#[natural_key = "<f>"]`
677    /// (the per-element editor opt-in shipped with the `mcp.servers`
678    /// per-field editor), this is the snake_case field name on the inner
679    /// type that holds the alias used to address each element â€” e.g.
680    /// `"name"` for `[[mcp.servers]]`. The incremental TOML writer
681    /// (`apply_dirty_path` in `schema.rs`) reads this to walk a
682    /// `<section>.<alias>.<inner>` dirty path through an array of tables.
683    /// Without it, the writer flat-faceplants on `as_table()` at the
684    /// alias segment and silently drops the edit on the floor.
685    ///
686    /// `None` for HashMap sections (alias *is* the TOML key, no further
687    /// hint needed) and for plain `Vec<T>` sections without
688    /// `#[natural_key]` (legacy whole-array `ObjectArray` round-trip,
689    /// no per-element addressing).
690    pub natural_key: Option<&'static str>,
691}

693/// Serializable wire representation of a config field for API consumers
694/// (RPC dispatch, gateway, TUI). Single source of truth â€” replaces the
695/// gateway's local `ListEntry` and the RPC dispatch's ad-hoc JSON.
696///
697/// Built from [`PropFieldInfo`] via [`ConfigFieldEntry::from_prop_field`].
698#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
699pub struct ConfigFieldEntry {
700    pub path: String,
701    pub category: String,
702    pub kind: PropKind,
703    pub type_hint: String,
704    #[serde(skip_serializing_if = "Option::is_none")]
705    pub value: Option<serde_json::Value>,
706    pub populated: bool,
707    pub is_secret: bool,
708    #[serde(default)]
709    pub is_env_overridden: bool,
710    #[serde(default, skip_serializing_if = "Vec::is_empty")]
711    pub enum_variants: Vec<String>,
712    pub description: String,
713    #[serde(skip_serializing_if = "Option::is_none")]
714    pub section: Option<String>,
715    /// Tab grouping. `ConfigTab::None` = no tab grouping (flat display).
716    #[serde(default, skip_serializing_if = "ConfigTab::is_none")]
717    pub tab: ConfigTab,
718}
719
720impl ConfigFieldEntry {
721    /// Convert a [`PropFieldInfo`] (server-side introspection) into its wire
722    /// representation. Secrets are masked (value omitted). The caller supplies
723    /// `is_env_overridden` from `Config::prop_is_env_overridden`.
724    pub fn from_prop_field(info: PropFieldInfo, is_env_overridden: bool) -> Self {
725        let populated = info.display_value != crate::traits::UNSET_DISPLAY;
726        let is_sensitive = info.is_secret || info.derived_from_secret;
727        let value = if is_sensitive {
728            None
729        } else {
730            Some(serde_json::Value::String(info.display_value))
731        };
732        let enum_variants = info.enum_variants.map(|f| f()).unwrap_or_default();
733        let section = crate::sections::Section::from_key(info.name.split('.').next().unwrap_or(""))
734            .map(|s| s.as_str().to_string());
735
736        Self {
737            path: info.name,
738            category: info.category.to_string(),
739            kind: info.kind,
740            type_hint: info.type_hint.to_string(),
741            value,
742            populated,
743            is_secret: is_sensitive,
744            is_env_overridden,
745            enum_variants,
746            description: info.description.to_string(),
747            section,
748            tab: info.tab,
749        }
750    }
751}

753/// One row emitted by the `Configurable` derive's `nested_option_entries()`
754/// method â€” every `#[nested] Option<XConfig>` field on a struct shows up here
755/// with its `present` bit and the per-field `#[display_name = "..."]` /
756/// `#[description = "..."]` metadata. The integrations registry consumes
757/// this verbatim instead of carrying its own per-field hand-list.

759pub struct NestedOptionEntry {
760    /// snake_case field name on the parent struct (e.g. `"telegram"`,
761    /// `"voice_duplex"`).
762    pub field: &'static str,
763    /// `true` when the parent struct's field is `Some(_)`.
764    pub present: bool,
765    /// Display name from `#[display_name = "..."]`; falls back to a
766    /// title-cased rendering of the snake_case field name when the
767    /// attribute is absent.
768    pub display_name: &'static str,
769    /// One-line summary from `#[description = "..."]`. Empty when the
770    /// attribute is absent.
771    pub description: &'static str,
772}

774/// One row emitted by the `Configurable` derive's `integration_descriptor()`
775/// method on structs annotated with `#[integration(...)]`. Used for nested
776/// toggleable configs (e.g. `BrowserConfig`, `CronConfig`) where the
777/// integration is "active" iff a named bool field on the struct is `true`.
778#[derive(Debug, Clone, Copy)]
779pub struct IntegrationDescriptor {

787    /// Snapshot of the named status field at the moment this descriptor
