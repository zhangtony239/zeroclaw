1//! Quickstart preset tables and submission shape.
2//!
3//! Two preset tables — [`RISK_PRESETS`] and [`RUNTIME_PRESETS`] — give
4//! the Quickstart UI a fixed-shape menu of named, opinionated profile
5//! defaults the user can pick from. Each preset carries:
6//!
7//! - `preset_name`  — the alias key written to config when picked
8//!   (`risk-profiles.<preset_name>` / `runtime-profiles.<preset_name>`).
9//!   Never `default`. The preset is canonical: picking it again
10//!   overwrites the alias of the same name with the preset's struct
11//!   values.
12//! - `label` / `help` — the strings the UI renders.
13//! - `values` — a struct literal of [`RiskProfileConfig`] /
14//!   [`RuntimeProfileConfig`] field values. The Quickstart writes
15//!   these verbatim into the corresponding config table on apply.
16//!
17//! Adding or removing a preset is one row in the `risk_presets!` /
18//! `runtime_presets!` table below; every consumer dispatches off
19//! `&'static [RiskPreset]` / `&'static [RuntimePreset]` so drift is
20//! impossible.
21//!
22//! [`BuilderSubmission`] is the single payload shape both surfaces
23//! (web gateway HTTP route, zerocode RPC route) and the CLI build and
24//! hand to `zeroclaw-runtime`'s apply path. The runtime validates and
25//! writes atomically. There is exactly one type, one validator, one
26//! apply function — surface code never assembles config directly.
27

39/// One row in the Risk preset table. The Quickstart UI renders the
40/// `label`, the runtime writes `values` to
41/// `risk-profiles.<preset_name>` on apply.
42#[derive(Debug, Clone, Copy, serde::Serialize)]
43pub struct RiskPreset {
44    /// Alias key written to `risk-profiles.<preset_name>`. Doubles as
45    /// the stable wire identifier (`BuilderSubmission.risk_preset`).
46    pub preset_name: &'static str,
47    /// Short label rendered in the picker UI.
48    pub label: &'static str,
49    /// One-line help blurb rendered next to the label.
50    pub help: &'static str,
51    /// Factory that produces the [`RiskProfileConfig`] this preset
52    /// installs. A function (not a const value) because
53    /// `RiskProfileConfig` has owned `Vec<String>` fields that cannot
54    /// live in a `const`.
55    #[serde(skip)]
56    pub values: fn() -> RiskProfileConfig,
57}

277/// Model provider widget submission. The Quickstart UI surfaces only
278/// the "greatest hits" fields an agent literally cannot start
279/// without; everything else (retry policy, rate limits, custom
280/// headers) lives in the post-Quickstart config editor.
281///
282/// `provider_type` is the type key written to
283/// `providers.models.<provider_type>.<alias>`. The exact set of
284/// recognised type strings tracks the existing
285/// `providers::ProviderKind`; Quickstart validates the chosen value
286/// at apply time via the runtime entry point.
287#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
288#[cfg_attr(feature = "schema-export", derive(schemars::JsonSchema))]
289pub struct ModelProviderChoice {
290    /// Provider type identifier (`anthropic`, `openai`, `openrouter`,
291    /// `ollama`, etc.). Used as the type segment in the TOML path.
292    pub provider_type: String,
293    /// User-named alias. Defaults to `"default"` in the UI; users
294    /// override when stacking multiple aliases of the same provider
295    /// type (e.g. `anthropic-work`, `anthropic-personal`).
296    pub alias: String,
297    /// Model id written to `providers.models.<type>.<alias>.model` at
298    /// apply time.
299    pub model: String,
300    /// Round-trip of every field the daemon described in
301    /// `quickstart/fields`. Surfaces echo back exactly what was
302    /// emitted; the daemon writes each entry under `<prefix>.<key>`
303    /// using its own schema knowledge.
304    #[serde(default, skip_serializing_if = "std::collections::HashMap::is_empty")]
305    pub fields: std::collections::HashMap<String, String>,
306}

326    /// (Telegram, Discord). `None` for channels that don't.

368/// The complete Quickstart submission both surfaces hand to
369/// `zeroclaw-runtime::quickstart::apply` (and pre-validate via
370/// `validate_only`). Single source of truth; assembling config
371/// outside this type is a layering bug.
372///
373/// Every field's `*_preset` / choice value is the user's resolved
374/// selection — the runtime translates preset keys into struct
375/// values via [`risk_preset`] / [`runtime_preset`] and looks up
376/// existing aliases against the live config when the UI submitted
377/// "use existing" rather than a fresh choice.
378#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
379#[cfg_attr(feature = "schema-export", derive(schemars::JsonSchema))]
380pub struct BuilderSubmission {
381    /// Model provider step submission. Always a `Create new` shape
382    /// in this initial cut — `Use existing` is represented by
383    /// [`SelectorChoice::Existing`] in the wrapper enum below.
384    pub model_provider: SelectorChoice<ModelProviderChoice>,
385    /// Risk profile preset key from [`RISK_PRESETS`], or the alias
386    /// of an existing `risk-profiles.<alias>`.
387    pub risk_profile: SelectorChoice<String>,
388    /// Runtime profile preset key from [`RUNTIME_PRESETS`], or the
389    /// alias of an existing `runtime-profiles.<alias>`.
390    pub runtime_profile: SelectorChoice<String>,
391    /// Memory step. Either a fresh [`MemoryChoice`] or the alias of
392    /// an existing `storage.<type>.<alias>` entry.
393    pub memory: SelectorChoice<MemoryChoice>,
394    /// Channels step. 0..N entries. Each is either a freshly-built
395    /// [`ChannelQuickStart`] or the alias of an existing channel.
396    /// The agent's `channels` field is auto-bound to every entry in
397    /// this vec at apply time.
398    pub channels: Vec<SelectorChoice<ChannelQuickStart>>,
399    /// Peer groups to materialize. Each entry can reference either a
400    /// staged channel from `channels` (above) or an already-configured
401    /// channel ref. Empty list = no peer-group rows written.
402    #[serde(default)]
403    pub peer_groups: Vec<QuickstartPeerGroup>,
404    /// Agent identity (always create-new — there's no reuse path).
405    pub agent: AgentIdentity,
406}
407
408/// Peer-group entry staged in the Quickstart. Maps 1:1 to a
409/// `[peer-groups.<name>]` table written at apply time. The `channel`
410/// field carries a `<type>.<alias>` ref pointing at either a staged
411/// channel from the same submission or a pre-existing one.
412#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
413#[cfg_attr(feature = "schema-export", derive(schemars::JsonSchema))]
414pub struct QuickstartPeerGroup {
415    /// Map key written to `peer-groups.<name>`. Synthesized by surfaces
416    /// from the channel ref so no `match` table is involved.
417    pub name: String,
418    /// Channel ref (`<type>.<alias>`) the peer group authorizes.
419    pub channel: String,
420    /// External (non-agent) peer usernames the channel should accept.
421    #[serde(default)]
422    pub external_peers: Vec<String>,
423    /// Per-group blocklist applied to the resolved peer set.
424    #[serde(default)]
425    pub ignore: Vec<String>,
426}

428/// Dual-mode selector outcome. Every Quickstart selector lets the
429/// user either pick an existing configured alias or create a fresh
430/// one; this enum carries which path was taken so the runtime apply
431/// path can branch on `Existing` (record an alias ref only, no
432/// writes to that section) vs `Fresh` (write a new entry).
433#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
434#[cfg_attr(feature = "schema-export", derive(schemars::JsonSchema))]
435#[serde(rename_all = "snake_case", tag = "mode", content = "value")]
436pub enum SelectorChoice<T> {
437    /// Use an already-configured alias under the corresponding
438    /// section. Carries only the alias key — the runtime resolves
439    /// against the live config at apply time.
440    Existing(String),
441    /// Create a new entry from the carried value.
442    Fresh(T),
443}
