5/// For a `#[nested] HashMap<String, T>` field, parse a `get_prop`/`set_prop`
6/// path of the form `<my_prefix>.<field_name>.<hm_key>.<inner_suffix>` and
7/// return the HashMap key + the fully-qualified inner name that the value
8/// type's own `get_prop` / `set_prop` expects.
9///
10/// HashMap keys are user-controlled and may contain dots, URLs, or hostnames
11/// (for example `model_providers.custom:https://example.invalid/v1.api-key`).
12/// Inner values may themselves be deeply nested (`AliasedAgentConfig` has
13/// `agent.thinking.<...>` subpaths), so neither left-splitting nor
14/// right-splitting works in isolation. Match against the actual present
15/// keys and pick the longest prefix that is followed by `.` — this
16/// correctly handles dotted keys *and* deep inner paths in one parse.
17///
18/// `keys` is an iterator over the live HashMap's keys (typically
19/// `self.<field>.keys().map(String::as_str)` from the derive). Returns
20/// `None` when the path doesn't match, letting the derive's generated
21/// code fall through to the next nested field.
22pub fn route_hashmap_path<'a, 'k, I>(
23    name: &'a str,
24    my_prefix: &str,
25    field_name: &str,
26    inner_prefix: &str,
27    keys: I,
28) -> Option<(&'a str, String)>
29where
30    I: IntoIterator<Item = &'k str>,
31{
32    let key_prefix = if my_prefix.is_empty() {
33        field_name.to_string()
34    } else {
35        format!("{my_prefix}.{field_name}")
36    };
37    let rest = name.strip_prefix(&key_prefix)?.strip_prefix('.')?;
38    // Longest-match against present map keys. Dotted keys (URL-shaped
39    // custom provider entries) sort longer than their unprefixed siblings,
40    // so this also disambiguates `custom:https://x` vs. `custom`.
41    let mut best: Option<(usize, &'a str)> = None;
42    for k in keys {
43        if let Some(_suffix) = rest.strip_prefix(k).and_then(|s| s.strip_prefix('.'))
44            && best.is_none_or(|(len, _)| k.len() > len)
45        {
46            // Slice the original `rest` so we can keep the lifetime tied
47            // to `name` rather than to a transient `&str` from the keys
48            // iterator.
49            let hm_key = &rest[..k.len()];
50            best = Some((k.len(), hm_key));
51        }
52    }
53    let (key_len, hm_key) = best?;
54    let inner_suffix = &rest[key_len + 1..];
55    let inner_name = if inner_prefix.is_empty() {
56        inner_suffix.to_string()
57    } else {
58        format!("{inner_prefix}.{inner_suffix}")
59    };
60    Some((hm_key, inner_name))
61}

63/// For a `#[nested] HashMap<String, HashMap<String, T>>` field, parse a path
64/// `<my_prefix>.<field_name>.<outer_key>.<inner_key>.<inner_suffix>` and
65/// return (outer_key, inner_key, fully-qualified inner name for T::get_prop).
66///
67/// Returns `None` when the path doesn't match (wrong prefix or too few segments).
68pub fn route_double_hashmap_path<'a>(
69    name: &'a str,
70    my_prefix: &str,
71    field_name: &str,
72    inner_prefix: &str,
73) -> Option<(&'a str, &'a str, String)> {
74    let key_prefix = if my_prefix.is_empty() {
75        field_name.to_string()
76    } else {
77        format!("{my_prefix}.{field_name}")
78    };
79    let rest = name.strip_prefix(&key_prefix)?.strip_prefix('.')?;
80    let (outer_key, rest2) = rest.split_once('.')?;
81    let (inner_key, inner_suffix) = rest2.split_once('.')?;
82    let inner_name = if inner_prefix.is_empty() {
83        inner_suffix.to_string()
84    } else {
85        format!("{inner_prefix}.{inner_suffix}")
86    };
87    Some((outer_key, inner_key, inner_name))
88}

90/// For a `#[nested] Vec<T>` field whose element type `T` carries a
91/// natural-key field (e.g. `McpServerConfig::name`), parse a path of the
92/// form `<my_prefix>.<field_name>.<natural_key>.<inner_suffix>` and
93/// return `(matched_natural_key_index_in_vec, fully-qualified inner name
94/// for T::get_prop / set_prop)`.
95///
96/// `natural_keys` is an iterator over `(index, key)` pairs from the live
97/// `Vec<T>`: typically `self.<field>.iter().enumerate().map(|(i, e)|
98/// (i, e.<natural_key_field>.as_str()))` from the derive.
99///
100/// Matching is longest-key-wins (same as [`route_hashmap_path`]) so dotted
101/// natural keys disambiguate against their shorter siblings.
102///
103/// Ambiguity (two elements sharing the same natural key) is *not* resolved
104/// silently: when two or more indices match the same longest natural key,
105/// the returned [`VecRoute`] is [`VecRoute::Ambiguous`] carrying the key
106/// and the duplicate count. Callers (get_prop / set_prop / prop_is_secret
107/// dispatch sites) surface this as an error rather than mutating one of
108/// the duplicates by accident — the schema validator catches the broken
109/// state at save time, but until the operator fixes it, in-flight edits
110/// must refuse to route.
111///
112/// Returns [`VecRoute::Miss`] when the path doesn't match this field's
113/// prefix, letting the derive's generated code fall through to the next
114/// nested field (same fall-through contract as [`route_hashmap_path`]).
115pub fn route_vec_path<'a, 'k, I>(
116    name: &'a str,
117    my_prefix: &str,
118    field_name: &str,
119    inner_prefix: &str,
120    natural_keys: I,
121) -> VecRoute<'a>
122where
123    I: IntoIterator<Item = (usize, &'k str)>,
124{
125    let key_prefix = if my_prefix.is_empty() {
126        field_name.to_string()
127    } else {
128        format!("{my_prefix}.{field_name}")
129    };
130    let Some(rest) = name
131        .strip_prefix(&key_prefix)
132        .and_then(|s| s.strip_prefix('.'))
133    else {
134        return VecRoute::Miss;
135    };
136
137    // Longest-match against live natural keys, collecting every index that
138    // hits the same maximal key so duplicates surface as Ambiguous.
139    let mut best_len: Option<usize> = None;
140    let mut best_hits: Vec<usize> = Vec::new();
141    for (idx, key) in natural_keys {
142        if key.is_empty() {
143            // Empty natural-key slots can't be addressed by name; skip so
144            // they don't soak up every otherwise-unmatched path via the
145            // empty prefix.
146            continue;
147        }
148        if rest
149            .strip_prefix(key)
150            .and_then(|s| s.strip_prefix('.'))
151            .is_some()
152            || rest == key
153        {
154            match best_len {
155                Some(len) if key.len() < len => continue,
156                Some(len) if key.len() == len => best_hits.push(idx),
157                _ => {
158                    best_len = Some(key.len());
159                    best_hits.clear();
160                    best_hits.push(idx);
161                }
162            }
163        }
164    }
165    let Some(key_len) = best_len else {
166        return VecRoute::Miss;
167    };
168    let key_slice = &rest[..key_len];
169    if best_hits.len() > 1 {
170        return VecRoute::Ambiguous {
171            key: key_slice,
172            count: best_hits.len(),
173        };
174    }
175    let idx = best_hits[0];
176    // `rest == key` means the inner suffix is empty (i.e. the path
177    // addresses the element itself, not a property on it). Treat as a
178    // miss for routing-into-properties; callers asking about the whole
179    // element use map_keys / map_key_create instead.
180    if rest.len() == key_len {
181        return VecRoute::Miss;
182    }
183    let inner_suffix = &rest[key_len + 1..];
184    let inner_name = if inner_prefix.is_empty() {
185        inner_suffix.to_string()
186    } else {
187        format!("{inner_prefix}.{inner_suffix}")
188    };
189    VecRoute::Hit {
190        index: idx,
191        inner_name,
192    }
193}

195/// Outcome of routing a `<prefix>.<field>.<natural_key>.<suffix>` path
196/// into a `#[nested] Vec<T>` field. See [`route_vec_path`].
197#[derive(Debug, Clone, PartialEq, Eq)]
198pub enum VecRoute<'a> {
199    /// Path is not addressed at this field at all (wrong prefix, or
200    /// addresses the element itself rather than a sub-property). Caller
201    /// should fall through to the next nested field.
202    Miss,
203    /// Path resolved to a unique element. `index` is the element's
204    /// position in the underlying `Vec`; `inner_name` is the property
205    /// path to pass to `T::get_prop` / `T::set_prop`.
206    Hit { index: usize, inner_name: String },
207    /// Path's natural key matches two or more elements. Callers must
208    /// surface this as an error: editing either duplicate silently is
209    /// a correctness hazard. The schema's per-section validator
210    /// (`validate_mcp_config` for `mcp.servers`) is the source of truth
211    /// for the on-save check; this variant is the in-flight equivalent.
212    Ambiguous { key: &'a str, count: usize },
213}

215/// Return a comma-separated string of valid enum variant names for display in error messages.
216#[cfg(feature = "schema-export")]
217pub fn enum_variants<T: schemars::JsonSchema>() -> String {
218    #[cfg(feature = "schema-export")]
219    let schema = schemars::schema_for!(T);
220    let json = match serde_json::to_value(&schema) {
221        Ok(v) => v,
222        Err(_) => return "(unknown variants)".to_string(),
223    };
224
225    if let Some(variants) = json.get("enum").and_then(|v| v.as_array()) {
226        let names: Vec<&str> = variants.iter().filter_map(|v| v.as_str()).collect();
227        if !names.is_empty() {
228            return names.join(", ");
229        }
230    }
231
232    if let Some(one_of) = json.get("oneOf").and_then(|v| v.as_array()) {
233        let names: Vec<&str> = one_of
234            .iter()
235            .filter_map(|s| {
236                s.get("const").and_then(|v| v.as_str()).or_else(|| {
237                    s.get("enum")
238                        .and_then(|v| v.as_array())
239                        .and_then(|arr| arr.first())
240                        .and_then(|v| v.as_str())
241                })
242            })
243            .collect();
244        if !names.is_empty() {
245            return names.join(", ");
246        }
247    }
248
249    "(unknown variants)".to_string()
250}

252/// Build a `PropFieldInfo` by reading the display value from a serialized TOML table.
253#[allow(clippy::too_many_arguments)]
254pub fn make_prop_field(
255    table: Option<&toml::Table>,
256    name: &str,
257    serde_name: &str,
258    category: &'static str,
259    type_hint: &'static str,
260    kind: PropKind,
261    is_secret: bool,
262    enum_variants: Option<fn() -> Vec<String>>,
263    description: &'static str,
264    derived_from_secret: bool,
265    credential_class: Option<CredentialSurfaceClass>,
266    tab: ConfigTab,
267    display_secret_terminals: &[&str],
268) -> PropFieldInfo {
269    let display_value = if is_secret || derived_from_secret {
270        match table.and_then(|t| t.get(serde_name)) {
271            Some(toml::Value::String(s)) if !s.is_empty() => "****".to_string(),
272            Some(toml::Value::Array(arr)) if !arr.is_empty() => {
273                format!("[{}]", vec!["****"; arr.len()].join(", "))
274            }
275            _ => crate::traits::UNSET_DISPLAY.to_string(),
276        }
277    } else {
278        toml_value_to_display_for_kind(
279            table.and_then(|t| t.get(serde_name)),
280            kind,
281            display_secret_terminals,
282        )
283    };
284    PropFieldInfo {
285        name: name.to_string(),
286        category,
287        display_value,
288        type_hint,
289        kind,
290        is_secret,
291        enum_variants,
292        description,
293        derived_from_secret,
294        credential_class,
295        tab,
296    }
297}

299/// Get a property value via serde serialization.
300pub fn serde_get_prop<T: serde::Serialize>(
301    target: &T,
302    prefix: &str,
303    name: &str,
304    is_secret: bool,
305    kind: PropKind,
306    display_secret_terminals: &[&str],
307) -> anyhow::Result<String> {
308    if is_secret {
309        return Ok("**** (encrypted)".to_string());
310    }
311    let serde_name = prop_name_to_serde_field(prefix, name)?;
312    let table = toml::Value::try_from(target)?;
313    Ok(toml_value_to_display_for_kind(
314        table.as_table().and_then(|t| t.get(&serde_name)),
315        kind,
316        display_secret_terminals,
317    ))
318}

320/// Set a property value via serde roundtrip.
321pub fn serde_set_prop<T: serde::Serialize + serde::de::DeserializeOwned>(
322    target: &mut T,
323    prefix: &str,
324    name: &str,
325    value_str: &str,
326    kind: PropKind,
327    is_option: bool,
328) -> anyhow::Result<()> {
329    let serde_name = prop_name_to_serde_field(prefix, name)?;
330    let mut table: toml::Table = toml::from_str(&toml::to_string(target)?)?;
331    if (value_str.is_empty() || value_str == crate::traits::UNSET_DISPLAY || value_str == "****")
332        && is_option
333    {
334        table.remove(&serde_name);
335    } else {
336        table.insert(serde_name, parse_prop_value(value_str, kind)?);
337    }
338    *target = toml::from_str(&toml::to_string(&table)?)?;
339    Ok(())
340}

342fn toml_value_to_display(value: Option<&toml::Value>) -> String {
343    match value {
344        None => crate::traits::UNSET_DISPLAY.to_string(),
345        Some(toml::Value::String(s)) => s.clone(),
346        Some(v) => v.to_string(),
347    }
348}

350fn toml_value_to_display_for_kind(
351    value: Option<&toml::Value>,
352    kind: PropKind,
353    display_secret_terminals: &[&str],
354) -> String {
355    match kind {
356        PropKind::Object | PropKind::ObjectArray => match value {
357            None => crate::traits::UNSET_DISPLAY.to_string(),
358            Some(toml::Value::String(s)) => s.clone(),
359            Some(v) => {
360                let mut redacted = v.clone();
361                redact_toml_display_secrets(&mut redacted, display_secret_terminals);
362                redacted.to_string()
363            }
364        },
365        _ => toml_value_to_display(value),
366    }
367}

369pub fn object_array_json_display_value(
370    value: &impl serde::Serialize,
371    display_secret_terminals: &[&str],
372) -> String {
373    match serde_json::to_value(value) {
374        Ok(mut value) => {
375            redact_json_display_secrets(&mut value, display_secret_terminals);
376            serde_json::to_string(&value).unwrap_or_else(|_| "[]".to_string())
377        }
378        Err(_) => "[]".to_string(),
379    }

382fn redact_json_display_secrets(value: &mut serde_json::Value, display_secret_terminals: &[&str]) {
383    match value {
384        serde_json::Value::Array(items) => {
385            for item in items {
386                redact_json_display_secrets(item, display_secret_terminals);
387            }
388        }
389        serde_json::Value::Object(map) => {
390            for (key, nested) in map.iter_mut() {
391                if display_key_is_secret_terminal(key, display_secret_terminals) {
392                    mask_json_value(nested);
393                } else {
394                    redact_json_display_secrets(nested, display_secret_terminals);
395                }
396            }
397        }
398        _ => {}
399    }
400}

422fn mask_json_value(value: &mut serde_json::Value) {
423    match value {
424        serde_json::Value::Array(items) => {
425            for item in items {
426                mask_json_value(item);
427            }
428        }
429        serde_json::Value::Object(map) => {
430            for nested in map.values_mut() {
431                mask_json_value(nested);
432            }
433        }
434        serde_json::Value::Null => {}
435        _ => *value = serde_json::Value::String("****".to_string()),
436    }

455fn display_key_is_secret_terminal(key: &str, display_secret_terminals: &[&str]) -> bool {
456    let normalized = normalize_display_key(key);
457    display_secret_terminals
458        .iter()
459        .any(|terminal| normalize_display_key(terminal) == normalized)
460}

462fn normalize_display_key(key: &str) -> String {
463    key.replace('-', "_").to_ascii_lowercase()

466fn prop_name_to_serde_field(prefix: &str, name: &str) -> anyhow::Result<String> {
467    let suffix = if prefix.is_empty() {
468        name
469    } else {
470        name.strip_prefix(prefix)
471            .and_then(|s| s.strip_prefix('.'))
472            .ok_or_else(|| {
473                ::zeroclaw_log::record!(
474                    WARN,
475                    ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Reject)
476                        .with_outcome(::zeroclaw_log::EventOutcome::Failure)
477                        .with_attrs(::serde_json::json!({"prefix": prefix, "name": name})),
478                    "prop_name_to_serde_field: property name does not share the configured prefix"
479                );
480                anyhow::Error::msg(format!("Unknown property '{name}'"))
481            })?
482    };
483    let field_part = suffix.split('.').next().unwrap_or(suffix);
484    Ok(field_part.replace('-', "_"))
485}

487fn parse_prop_value(value_str: &str, kind: PropKind) -> anyhow::Result<toml::Value> {
488    let reject = |reason: &'static str, attrs: serde_json::Value| {
489        ::zeroclaw_log::record!(
490            WARN,
491            ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Reject)
492                .with_outcome(::zeroclaw_log::EventOutcome::Failure)
493                .with_attrs(attrs),
494            "parse_prop_value rejected input"
495        );
496        let _ = reason;
497    };
498    match kind {
499        PropKind::Bool => Ok(toml::Value::Boolean(value_str.parse().map_err(|_| {
500            reject(
501                "bool",
502                ::serde_json::json!({"kind": "bool", "got_len": value_str.len()}),
503            );
504            anyhow::Error::msg(format!(
505                "Invalid bool value '{value_str}', expected 'true' or 'false'"
506            ))
507        })?)),
508        PropKind::Integer => Ok(toml::Value::Integer(value_str.parse().map_err(|_| {
509            reject(
510                "integer",
511                ::serde_json::json!({"kind": "integer", "got_len": value_str.len()}),
512            );
513            anyhow::Error::msg(format!("Invalid integer value '{value_str}'"))
514        })?)),
515        PropKind::Float => Ok(toml::Value::Float(value_str.parse().map_err(|_| {
516            reject(
517                "float",
518                ::serde_json::json!({"kind": "float", "got_len": value_str.len()}),
519            );
520            anyhow::Error::msg(format!("Invalid float value '{value_str}'"))
521        })?)),
522        PropKind::String | PropKind::Enum => Ok(toml::Value::String(value_str.to_string())),
523        PropKind::StringArray => {
524            let trimmed = value_str.trim();
525            // Accept JSON/TOML array syntax: ["a", "b", "c"]
526            if trimmed.starts_with('[')
527                && let Ok(arr) = serde_json::from_str::<Vec<String>>(trimmed)
528            {
529                return Ok(toml::Value::Array(
530                    arr.into_iter()
531                        .filter(|s| !s.is_empty() && s != crate::traits::UNSET_DISPLAY)
532                        .map(toml::Value::String)
533                        .collect(),
534                ));
535            }
536            // Fall back to comma-separated input.
537            let items = value_str
538                .split(',')
539                .map(|s| toml::Value::String(s.trim().to_string()))
540                .filter(|v| {
541                    v.as_str()
542                        .is_some_and(|s| !s.is_empty() && s != crate::traits::UNSET_DISPLAY)
543                })
544                .collect();
545            Ok(toml::Value::Array(items))
546        }
547        // `Vec<T>` of structs: round-trip a JSON array of objects to a
548        // TOML array. JSON `null` (used by serde for `Option::None`) is
549        // dropped because TOML has no null - the absent key conveys the
550        // same meaning when the field deserializes back into `Option<T>`.
551        PropKind::ObjectArray => {
552            let v: serde_json::Value = serde_json::from_str(value_str).map_err(|e| {
553                reject(
554                    "object_array",
555                    ::serde_json::json!({"kind": "object_array", "error": format!("{}", e)}),
556                );
557                anyhow::Error::msg(format!("invalid JSON array of objects: {e}"))
558            })?;
559            json_to_toml(v).ok_or_else(|| {
560                reject(
561                    "object_array_nulls",
562                    ::serde_json::json!({"kind": "object_array", "reason": "all-null"}),
563                );
564                anyhow::Error::msg("JSON value contained only nulls, nothing to write")
565            })
566        }
567        // Struct-shaped scalar: parse the JSON object into a TOML table so
568        // the parent serde round-trip deserializes into the typed struct
569        // (e.g. `Option<ModelPricing>`). Inserting a raw String here would
570        // fail serde because the field is typed, not free-form text.
571        PropKind::Object => {
572            let v: serde_json::Value = serde_json::from_str(value_str).map_err(|e| {
573                reject(
574                    "object",
575                    ::serde_json::json!({"kind": "object", "error": format!("{}", e)}),
576                );
577                anyhow::Error::msg(format!("invalid JSON object: {e}"))
578            })?;
579            if !matches!(v, serde_json::Value::Object(_)) {
580                reject(
581                    "object_shape",
582                    ::serde_json::json!({"kind": "object", "got_shape": "non-object"}),
583                );
584                anyhow::bail!("Object field requires a JSON object; got {v}");
585            }
586            json_to_toml(v).ok_or_else(|| {
587                reject(
588                    "object_nulls",
589                    ::serde_json::json!({"kind": "object", "reason": "all-null"}),
590                );
591                anyhow::Error::msg("JSON object contained only nulls, nothing to write")
592            })
593        }
594    }
595}

597/// Walk a `serde_json::Value` into a `toml::Value`, dropping any `null`s
598/// (TOML has no null; absence of a key conveys `Option::None`).
599fn json_to_toml(v: serde_json::Value) -> Option<toml::Value> {
600    match v {
601        serde_json::Value::Null => None,
602        serde_json::Value::Bool(b) => Some(toml::Value::Boolean(b)),
603        serde_json::Value::String(s) => Some(toml::Value::String(s)),
604        serde_json::Value::Number(n) => {
605            if let Some(i) = n.as_i64() {
606                Some(toml::Value::Integer(i))
607            } else if let Some(u) = n.as_u64() {
608                // TOML integers are i64; clamp pathological u64 values.
609                Some(toml::Value::Integer(i64::try_from(u).unwrap_or(i64::MAX)))
610            } else {
611                n.as_f64().map(toml::Value::Float)
612            }
613        }
614        serde_json::Value::Array(items) => Some(toml::Value::Array(
615            items.into_iter().filter_map(json_to_toml).collect(),
616        )),
617        serde_json::Value::Object(map) => {
618            let mut table = toml::map::Map::new();
619            for (k, val) in map {
620                if let Some(tv) = json_to_toml(val) {
621                    table.insert(k, tv);
622                }
623            }
624            Some(toml::Value::Table(table))
625        }
626    }
627}

629/// Validate that an alias key is safe for use in TOML dotted paths, URLs,
630/// filesystem paths on Windows/macOS/Linux, and `ZEROCLAW_*` env-var grammar.
631///
632/// Allowed: lowercase ASCII alphanumeric plus single underscore, 1-63 chars.
633/// Must start AND end with alphanumeric. Adjacent underscores (`__`) are
634/// forbidden because they collide with the env-var grammar's path separator.
635///
636/// The env-var grammar uses `__` as path separator, which lets aliases keep
637/// single `_` literally (`prod_v2`, `staging_api`). Hyphens are forbidden
638/// because they are illegal in POSIX env-var identifiers; uppercase is
639/// forbidden so the bootstrap env-vars (`ZEROCLAW_WORKSPACE`,
640/// `ZEROCLAW_CONFIG_DIR`) stay disambiguated by case.
641pub fn validate_alias_key(key: &str) -> Result<(), String> {
642    if key.is_empty() {
643        return Err("alias must not be empty".to_string());
644    }
645    if key.len() > 63 {
646        return Err(format!(
647            "alias '{}' is too long ({} chars); maximum is 63",
648            key,
649            key.len()
650        ));
651    }
652    let first = key.chars().next().unwrap();
653    let last = key.chars().next_back().unwrap();
654    if !matches!(first, 'a'..='z' | '0'..='9') {
655        return Err(format!(
656            "alias '{key}' must start with a lowercase letter or digit"
657        ));
658    }
659    if !matches!(last, 'a'..='z' | '0'..='9') {
660        return Err(format!(
661            "alias '{key}' must end with a lowercase letter or digit"
662        ));
663    }
664    if key.contains("__") {
665        return Err(format!(
666            "alias '{key}' must not contain `__`; it is reserved as the env-var grammar's path separator"
667        ));
668    }
669    for ch in key.chars() {
670        if !matches!(ch, 'a'..='z' | '0'..='9' | '_') {
671            return Err(format!(
672                "alias '{}' contains invalid character {:?}; \
673                 only lowercase letters, digits, and single underscores are allowed (no hyphen, no uppercase)",
674                key, ch
675            ));
676        }
677    }
678    Ok(())
679}

681/// Resolve a CLI-typed config path to its canonical form.
682///
683/// Field segments derived from the schema are kebab-case; aliases are
684/// snake-only per [`validate_alias_key`]. For each known canonical
685/// path, segments are compared pairwise: equal verbatim, equal after
686/// swapping `-` → `_` when the canonical segment contains `-`, or
687/// equal after swapping `_` → `-` for the final field segment. The
688/// final-segment rule lets older CLI spelling like `api-key` resolve
689/// to schema-canonical `api_key` without rewriting map aliases such as
690/// `my_bot`. Returns `raw` unchanged when no canonical path matches.
691#[must_use]
692pub fn resolve_field_path(known_paths: &[String], raw: &str) -> String {
693    let raw_segs: Vec<&str> = raw.split('.').collect();
694    for known in known_paths {
695        let known_segs: Vec<&str> = known.split('.').collect();
696        if known_segs.len() != raw_segs.len() {
697            continue;
698        }
699        let final_index = known_segs.len().saturating_sub(1);
700        let all_match = known_segs
701            .iter()
702            .zip(raw_segs.iter())
703            .enumerate()
704            .all(|(idx, (k, r))| {
705                k == r
706                    || (k.contains('-') && k.replace('-', "_") == **r)
707                    || (idx == final_index && k.contains('_') && k.replace('_', "-") == **r)
708            });
709        if all_match {
710            return known.clone();
711        }
712    }
713    raw.to_string()
714}

716/// Inverse of the `Configurable` macro's internal `snake_to_kebab`.
717///
718/// Field paths emitted by `prop_fields()` are kebab-case (per the macro's
719/// snake→kebab transform of the underlying Rust idents). Surfaces that want
720/// to display the field under its serde-canonical snake_case spelling — for
721/// example `api_key` rather than `api-key` — use this to convert.
722///
723/// No-op for keys without `-`.
724pub fn kebab_to_snake(key: &str) -> String {
725    key.replace('-', "_")
726}
