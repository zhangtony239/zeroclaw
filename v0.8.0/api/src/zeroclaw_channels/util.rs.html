1/// Truncate a string to `max_chars` Unicode characters, appending "..." if truncated.
2pub fn truncate_with_ellipsis(s: &str, max_chars: usize) -> String {
3    match s.char_indices().nth(max_chars) {
4        Some((idx, _)) => {
5            let truncated = &s[..idx];
6            format!("{}...", truncated.trim_end())
7        }
8        None => s.to_string(),
9    }
10}

12/// Largest byte index `<= max_bytes` that is still a valid UTF-8 boundary.
13pub fn floor_char_boundary(s: &str, max_bytes: usize) -> usize {
14    if max_bytes >= s.len() {
15        return s.len();
16    }
17    let mut end = max_bytes;
18    while end > 0 && !s.is_char_boundary(end) {
19        end -= 1;
20    }
21    end
22}

26pub fn strip_tool_call_tags(message: &str) -> String {
27    const TOOL_CALL_OPEN_TAGS: [&str; 7] = [
28        "<function_calls>",
29        "<function_call>",
30        "<tool_call>",
31        "<toolcall>",
32        "<tool-call>",
33        "<tool>",
34        "<invoke>",
35    ];
36
37    fn find_first_tag<'a>(haystack: &str, tags: &'a [&'a str]) -> Option<(usize, &'a str)> {
38        tags.iter()
39            .filter_map(|tag| haystack.find(tag).map(|idx| (idx, *tag)))
40            .min_by_key(|(idx, _)| *idx)
41    }
42
43    fn matching_close_tag(open_tag: &str) -> Option<&'static str> {
44        match open_tag {
45            "<function_calls>" => Some("</function_calls>"),
46            "<function_call>" => Some("</function_call>"),
47            "<tool_call>" => Some("</tool_call>"),
48            "<toolcall>" => Some("</toolcall>"),
49            "<tool-call>" => Some("</tool-call>"),
50            "<tool>" => Some("</tool>"),
51            "<invoke>" => Some("</invoke>"),
52            _ => None,
53        }
54    }
55
56    fn extract_first_json_end(input: &str) -> Option<usize> {
57        let trimmed = input.trim_start();
58        let trim_offset = input.len().saturating_sub(trimmed.len());
59
60        for (byte_idx, ch) in trimmed.char_indices() {
61            if ch != '{' && ch != '[' {
62                continue;
63            }
64
65            let slice = &trimmed[byte_idx..];
66            let mut stream =
67                serde_json::Deserializer::from_str(slice).into_iter::<serde_json::Value>();
68            if let Some(Ok(_value)) = stream.next() {
69                let consumed = stream.byte_offset();
70                if consumed > 0 {
71                    return Some(trim_offset + byte_idx + consumed);
72                }
73            }
74        }
75
76        None
77    }
78
79    fn strip_leading_close_tags(mut input: &str) -> &str {
80        loop {
81            let trimmed = input.trim_start();
82            if !trimmed.starts_with("</") {
83                return trimmed;
84            }
85
86            let Some(close_end) = trimmed.find('>') else {
87                return "";
88            };
89            input = &trimmed[close_end + 1..];
90        }
91    }
92
93    // Does the tag structure run to the end of the message? A *real* truncated
94    // tool call is the model getting cut off, so the unterminated structure is
95    // the last thing in the message. If natural-language prose resumes after the
96    // tags, this is an inline *example* (the model is discussing tool calls), not
97    // a truncation — so we should keep it. Bias toward keeping: a little leaked
98    // XML beats eating the user's text.
99    fn tool_structure_runs_to_end(inner: &str) -> bool {
100        let mut rest = inner.trim_start();
101        // Consume a run of `<...>` tags (and whitespace between them).
102        while rest.starts_with('<') {
103            match rest.find('>') {
104                Some(gt) => rest = rest[gt + 1..].trim_start(),
105                // Cut off mid-tag (no closing '>') — a classic truncation.
106                None => return true,
107            }
108        }
109        let tail = rest.trim();
110        if tail.is_empty() {
111            // Tags ran cleanly to the end → truncation.
112            return true;
113        }
114        // Non-empty tail: prose ⇒ inline example (keep); otherwise it's a
115        // truncated tag/param value (drop).
116        !looks_like_prose(tail)
117    }
118
119    // Heuristic: does `text` read like resumed natural-language prose (as opposed
120    // to a cut-off parameter value)? True on an internal sentence boundary
121    // (". " / "! " / "? " + a letter) or a multi-word string that ends like a
122    // sentence. Deliberately lenient so ambiguous tails are kept, not dropped.
123    fn looks_like_prose(text: &str) -> bool {
124        let bytes = text.as_bytes();
125        for i in 0..bytes.len().saturating_sub(1) {
126            if matches!(bytes[i], b'.' | b'!' | b'?')
127                && matches!(bytes[i + 1], b' ' | b'\n' | b'\t')
128                && text[i + 1..]
129                    .trim_start()
130                    .chars()
131                    .next()
132                    .is_some_and(|c| c.is_alphabetic())
133            {
134                return true;
135            }
136        }
137        let trimmed = text.trim_end();
138        let ends_like_sentence = trimmed
139            .chars()
140            .last()
141            .is_some_and(|c| matches!(c, '.' | '!' | '?'))
142            && trimmed
143                .chars()
144                .rev()
145                .nth(1)
146                .is_some_and(|c| c.is_alphabetic());
147        ends_like_sentence && text.trim().contains(' ')
148    }
149
150    let mut kept_segments = Vec::new();
151    let mut remaining = message;
152
153    while let Some((start, open_tag)) = find_first_tag(remaining, &TOOL_CALL_OPEN_TAGS) {
154        let before = &remaining[..start];
155        if !before.is_empty() {
156            kept_segments.push(before.to_string());
157        }
158
159        let Some(close_tag) = matching_close_tag(open_tag) else {
160            break;
161        };
162        let after_open = &remaining[start + open_tag.len()..];
163
164        if let Some(close_idx) = after_open.find(close_tag) {
165            remaining = &after_open[close_idx + close_tag.len()..];
166            continue;
167        }
168
169        if let Some(consumed_end) = extract_first_json_end(after_open) {
170            remaining = strip_leading_close_tags(&after_open[consumed_end..]);
171            continue;
172        }
173
174        // Unterminated open tag with no parseable JSON body. Drop the broken
175        // tail ONLY when it looks like tool-call structure (`<invoke>` /
176        // `<parameter>` / `<tool*>` / `<function*>` / `{` / `[`) AND that
177        // structure runs to the end of the message — i.e. a real truncation
178        // where the model was cut off mid-call. If prose resumes after the
179        // structure, the model is showing an *example*, not making a call, so
180        // keep it verbatim (a little leaked XML beats eating the user's reply).
181        // Text that merely mentions a tag is likewise kept.
182        let inner = after_open.trim_start();
183        let inner_lower = inner.to_ascii_lowercase();
184        let looks_like_tool_structure = inner_lower.starts_with("<invoke")
185            || inner_lower.starts_with("<parameter")
186            || inner_lower.starts_with("<tool")
187            || inner_lower.starts_with("<function")
188            || inner.starts_with('{')
189            || inner.starts_with('[');
190        if looks_like_tool_structure && tool_structure_runs_to_end(inner) {
191            remaining = "";
192            break;
193        }
194
195        kept_segments.push(remaining[start..].to_string());
196        remaining = "";
197        break;
198    }
199
200    if !remaining.is_empty() {
201        kept_segments.push(remaining.to_string());
202    }
203
204    let mut result = kept_segments.concat();
205
206    // Clean up any resulting blank lines (but preserve paragraphs)
207    while result.contains("\n\n\n") {
208        result = result.replace("\n\n\n", "\n\n");
209    }
210
211    result.trim().to_string()
212}

214/// Recognized attachment marker kinds (e.g. `[IMAGE:/path]`, `[DOCUMENT:url]`).
215const ATTACHMENT_KINDS: &[&str] = &[
216    "IMAGE", "PHOTO", "DOCUMENT", "FILE", "VIDEO", "AUDIO", "VOICE",
217];
218
219/// Parse `[KIND:target]` attachment markers out of a message.
220/// Returns cleaned text (markers removed) and a vec of `(kind, target)` pairs.
221pub fn parse_attachment_markers(message: &str) -> (String, Vec<(String, String)>) {
222    let mut cleaned = String::with_capacity(message.len());
223    let mut attachments = Vec::new();
224    let mut cursor = 0usize;
225
226    while let Some(rel_start) = message[cursor..].find('[') {
227        let start = cursor + rel_start;
228        cleaned.push_str(&message[cursor..start]);
229
230        let Some(rel_end) = message[start..].find(']') else {
231            cleaned.push_str(&message[start..]);
232            cursor = message.len();
233            break;
234        };
235        let end = start + rel_end;
236        let marker_text = &message[start + 1..end];
237
238        let parsed = marker_text.split_once(':').and_then(|(kind, target)| {
239            let kind_upper = kind.trim().to_ascii_uppercase();
240            let target = target.trim();
241            if target.is_empty() || !ATTACHMENT_KINDS.contains(&kind_upper.as_str()) {
242                return None;
243            }
244            Some((kind_upper, target.to_string()))
245        });
246
247        if let Some(attachment) = parsed {
248            attachments.push(attachment);
249        } else {
250            cleaned.push_str(&message[start..=end]);
251        }
252
253        cursor = end + 1;
254    }
255
256    if cursor < message.len() {
257        cleaned.push_str(&message[cursor..]);
258    }
259
260    (cleaned.trim().to_string(), attachments)
261}

263/// Generate a short 6-character lowercase alphanumeric approval token.
264///
265/// Uses the full `[a-z0-9]` alphabet (36 options per position, 36^6 ≈ 2.2B
266/// combinations) — not UUID hex (which would give only 16^6 ≈ 16.7M and
267/// would materially weaken the WhatsApp no-per-sender-check design
268/// described in the PR #6010 security note).
269#[cfg(any(
270    feature = "channel-discord",
271    feature = "channel-signal",
272    feature = "channel-slack",
273    feature = "channel-whatsapp-cloud",
274    feature = "whatsapp-web",
275    test
276))]
277pub(crate) fn new_approval_token() -> String {
278    use rand::RngExt;
279    const CHARSET: &[u8] = b"abcdefghijklmnopqrstuvwxyz0123456789";
280    let mut rng = rand::rng();
281    (0..6)
282        .map(|_| CHARSET[rng.random_range(0..CHARSET.len())] as char)
283        .collect()
284}

286/// Parse an approval reply of the form `"TOKEN yes|no|always ..."`.
287///
288/// Returns `Some((token, response))` when the text begins with a 6-character
289/// alphanumeric token followed by a recognised action word. Returns `None`
290/// for any other input so normal messages are not intercepted.
291pub fn parse_approval_reply(
292    text: &str,
293) -> Option<(String, zeroclaw_api::channel::ChannelApprovalResponse)> {
294    use zeroclaw_api::channel::ChannelApprovalResponse;
295    let lower = text.trim().to_lowercase();
296    let mut parts = lower.splitn(2, ' ');
297    let token = parts.next()?.to_string();
298    if token.len() != 6 || !token.chars().all(|c| c.is_ascii_alphanumeric()) {
299        return None;
300    }
301    let action_word = parts.next()?.split_whitespace().next()?;
302    let response = match action_word {
303        "yes" | "y" | "approve" => ChannelApprovalResponse::Approve,
304        "no" | "n" | "deny" => ChannelApprovalResponse::Deny,
305        "always" => ChannelApprovalResponse::AlwaysApprove,
306        _ => return None,
307    };
308    Some((token, response))
309}

311/// Generate a conversation history key from a channel message.
312pub fn conversation_history_key(msg: &zeroclaw_api::channel::ChannelMessage) -> String {
313    match &msg.thread_ts {
314        Some(tid) => format!(
315            "{}_{}_{}_{}",
316            msg.channel, msg.reply_target, tid, msg.sender
317        ),
318        None => format!("{}_{}_{}", msg.channel, msg.reply_target, msg.sender),
319    }
320}
