271    /// Encode all registered metrics into Prometheus text exposition format.
272    pub fn encode(&self) -> String {
273        let encoder = TextEncoder::new();
274        let families = self.registry.gather();
275        let mut buf = Vec::new();
276        encoder.encode(&families, &mut buf).unwrap_or_default();
277        String::from_utf8(buf).unwrap_or_default()
278    }

280    /// Process-wide singleton handle. All call sites that obtain a Prometheus
281    /// observer through this function share the same `Registry` and the same
282    /// underlying counters, so events recorded by the channel orchestrator and
283    /// events recorded by the gateway accumulate into the same time series and
284    /// are visible on a single `/metrics` scrape.
285    ///
286    /// `PrometheusObserver::new()` still returns a fresh, isolated instance —
287    /// kept for tests so parallel test cases don't see each other's counts.
288    pub fn shared() -> Arc<Self> {
289        static SINGLETON: OnceLock<Arc<PrometheusObserver>> = OnceLock::new();
290        SINGLETON.get_or_init(|| Arc::new(Self::new())).clone()
291    }

294impl Observer for PrometheusObserver {
295    fn record_event(&self, event: &ObserverEvent) {
296        match event {
297            ObserverEvent::AgentStart {
298                model_provider,
299                model,
300                channel: _,
301                agent_alias: _,
302                turn_id: _,
303            } => {
304                self.agent_starts
305                    .with_label_values(&[model_provider, model])
306                    .inc();
307            }
308            ObserverEvent::AgentEnd {
309                model_provider,
310                model,
311                duration,
312                tokens_used,
313                cost_usd: _,
314                channel: _,
315                agent_alias: _,
316                turn_id: _,
317            } => {
318                // Agent duration is recorded via the histogram with model_provider/model labels
319                self.agent_duration
320                    .with_label_values(&[model_provider, model])
321                    .observe(duration.as_secs_f64());
322                if let Some(usage) = tokens_used {
323                    self.tokens_used.set(
324                        i64::try_from(usage.input_tokens.saturating_add(usage.output_tokens))
325                            .unwrap_or(i64::MAX),
326                    );
327                }
328            }
329            ObserverEvent::LlmResponse {
330                model_provider,
331                model,
332                success,
333                input_tokens,
334                output_tokens,
335                ..
336            } => {
337                let success_str = if *success { "true" } else { "false" };
338                self.llm_requests
339                    .with_label_values(&[model_provider.as_str(), model.as_str(), success_str])
340                    .inc();
341                if let Some(input) = input_tokens {
342                    self.tokens_input_total
343                        .with_label_values(&[model_provider.as_str(), model.as_str()])
344                        .inc_by(*input);
345                }
346                if let Some(output) = output_tokens {
347                    self.tokens_output_total
348                        .with_label_values(&[model_provider.as_str(), model.as_str()])
349                        .inc_by(*output);
350                }
351            }
352            ObserverEvent::ToolCallStart { .. }
353            | ObserverEvent::TurnComplete
354            | ObserverEvent::LlmRequest { .. }
355            | ObserverEvent::DeploymentStarted { .. }
356            | ObserverEvent::RecoveryCompleted { .. } => {}
357            ObserverEvent::ToolCall {
358                tool,
359                duration,
360                success,
361                ..
362            } => {
363                let success_str = if *success { "true" } else { "false" };
364                self.tool_calls
365                    .with_label_values(&[tool.as_str(), success_str])
366                    .inc();
367                self.tool_duration
368                    .with_label_values(&[tool.as_str()])
369                    .observe(duration.as_secs_f64());
370            }
371            ObserverEvent::ChannelMessage { channel, direction } => {
372                self.channel_messages
373                    .with_label_values(&[channel, direction])
374                    .inc();
375            }
376            ObserverEvent::HeartbeatTick => {
377                self.heartbeat_ticks.inc();
378            }
379            ObserverEvent::CacheHit {
380                cache_type,
381                tokens_saved,
382            } => {
383                self.cache_hits.with_label_values(&[cache_type]).inc();
384                self.cache_tokens_saved
385                    .with_label_values(&[cache_type])
386                    .inc_by(*tokens_saved);
387            }
388            ObserverEvent::CacheMiss { cache_type } => {
389                self.cache_misses.with_label_values(&[cache_type]).inc();
390            }
391            ObserverEvent::Error {
392                component,
393                message: _,
394            } => {
395                self.errors.with_label_values(&[component]).inc();
396            }
397            ObserverEvent::DeploymentCompleted { .. } => {
398                self.deployments_total.with_label_values(&["success"]).inc();
399                let s = self
400                    .deploy_success_count
401                    .fetch_add(1, std::sync::atomic::Ordering::Relaxed)
402                    + 1;
403                let f = self
404                    .deploy_failure_count
405                    .load(std::sync::atomic::Ordering::Relaxed);
406                let total = s + f;
407                if total > 0 {
408                    self.deployment_failure_rate.set(f as f64 / total as f64);
409                }
410            }
411            ObserverEvent::DeploymentFailed { .. } => {
412                self.deployments_total.with_label_values(&["failure"]).inc();
413                let f = self
414                    .deploy_failure_count
415                    .fetch_add(1, std::sync::atomic::Ordering::Relaxed)
416                    + 1;
417                let s = self
418                    .deploy_success_count
419                    .load(std::sync::atomic::Ordering::Relaxed);
420                let total = s + f;
421                if total > 0 {
422                    self.deployment_failure_rate.set(f as f64 / total as f64);
423                }
424            }
425        }
426    }
427
428    fn record_metric(&self, metric: &ObserverMetric) {
429        match metric {
430            ObserverMetric::RequestLatency(d) => {
431                self.request_latency.observe(d.as_secs_f64());
432            }
433            ObserverMetric::TokensUsed(t) => {
434                self.tokens_used.set(i64::try_from(*t).unwrap_or(i64::MAX));
435            }
436            ObserverMetric::ActiveSessions(s) => {
437                self.active_sessions
438                    .with_label_values(&[] as &[&str])
439                    .set(*s as f64);
440            }
441            ObserverMetric::QueueDepth(d) => {
442                self.queue_depth
443                    .with_label_values(&[] as &[&str])
444                    .set(*d as f64);
445            }
446            ObserverMetric::DeploymentLeadTime(d) => {
447                self.deployment_lead_time.observe(d.as_secs_f64());
448            }
449            ObserverMetric::RecoveryTime(d) => {
450                self.recovery_time.observe(d.as_secs_f64());
451                self.mttr.set(d.as_secs_f64());
452            }
453        }
454    }
455
456    fn name(&self) -> &str {
457        "prometheus"
458    }
459
460    fn as_any(&self) -> &dyn std::any::Any {
461        self
462    }
463}
