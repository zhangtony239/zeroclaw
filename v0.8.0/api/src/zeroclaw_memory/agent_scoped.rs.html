1//! Runtime memory wrapper bound to one agent.
2//!
3//! Each agent holds its own per-agent backend instance (selected at
4//! agent creation via `[agents.<alias>.memory.backend]`, immutable
5//! thereafter). The wrapper sits directly on top of that instance and:
6//!
7//! - Stamps the bound agent's UUID on every store via the inner
8//!   backend's `store_with_agent` trait method (real implementations
9//!   on every backend; the agent_id is never silently dropped at the
10//!   trait boundary).
11//! - Filters every recall through the inner backend's
12//!   `recall_for_agents` with the resolved allowlist (own UUID + the
13//!   `read_memory_from` allowlist from
14//!   `[agents.<alias>.workspace.read_memory_from]`).
15//! - Intersects caller-supplied per-call allowlists with the bound
16//!   allowlist so a caller can never widen scope past what the agent's
17//!   config permits.
18//!
19//! Cross-backend allowlist entries are rejected at config load. The
20//! wrapper only ever sees same-backend sibling UUIDs in its
21//! `allowed_agent_ids` set.
22

29/// A `Memory` impl that scopes every read and write to a bound agent's
30/// UUID + a resolved cross-agent allowlist.
31///
32/// Construct via [`AgentScopedMemory::new`] at agent-loop entry. The
33/// runtime holds one per agent. Non-generic over the inner backend
34/// (holds `Arc<dyn Memory>`) so the per-agent factory can hand back a
35/// single concrete type regardless of the agent's chosen backend kind.
36pub struct AgentScopedMemory {
37    /// The wrapped backend. `Arc<dyn Memory>` to slot into the existing
38    /// per-install plumbing while the runtime factory hands out one
39    /// instance per agent.
40    inner: Arc<dyn Memory>,
41    /// The bound agent's UUID (from `agents.id`). Stamped on every
42    /// write through this wrapper.
43    agent_id: String,
44    /// Set of agent UUIDs this wrapper recalls from. Always contains
45    /// [`Self::agent_id`] (an agent always sees its own rows); any
46    /// additional UUIDs come from the configured `read_memory_from`
47    /// allowlist resolved at construction.
48    allowed_agent_ids: HashSet<String>,
49}
50
51impl AgentScopedMemory {
52    /// Build a new agent-scoped wrapper around `inner`.
53    ///
54    /// `agent_id` is the bound agent's UUID (looked up from the
55    /// `agents` table by alias at construction time in the runtime
56    /// factory). `allowed_sibling_agent_ids` is the resolved
57    /// `read_memory_from` allowlist; the bound `agent_id` is added
58    /// automatically to the in-memory `allowed_agent_ids` set so
59    /// callers do not need to remember to include themselves.
60    #[must_use]
61    pub fn new(
62        inner: Arc<dyn Memory>,
63        agent_id: impl Into<String>,
64        allowed_sibling_agent_ids: impl IntoIterator<Item = String>,
65    ) -> Self {
66        let agent_id = agent_id.into();
67        let mut allowed_agent_ids: HashSet<String> =
68            allowed_sibling_agent_ids.into_iter().collect();
69        allowed_agent_ids.insert(agent_id.clone());
70        Self {
71            inner,
72            agent_id,
73            allowed_agent_ids,
74        }
75    }
76
77    /// Build a `Vec<&str>` of the allowlist for passing to the
78    /// `Memory::recall_for_agents` trait method, which takes a
79    /// borrowed slice. Stable iteration order is not required.
80    fn allowed_slice(&self) -> Vec<&str> {
81        self.allowed_agent_ids.iter().map(String::as_str).collect()
82    }
83}

99    async fn store(
100        &self,
101        key: &str,
102        content: &str,
103        category: MemoryCategory,
104        session_id: Option<&str>,
105    ) -> Result<()> {
106        // Every store routes through `store_with_agent` so the bound
107        // agent's UUID is persisted. Backends with native agent_id
108        // columns (Sqlite, Postgres, Lucid) write the column; Qdrant
109        // writes the payload field; Markdown attributes via the on-
110        // disk path; None drops it. Each backend's behavior is
111        // explicit at the trait boundary.
112        self.inner
113            .store_with_agent(
114                key,
115                content,
116                category,
117                session_id,
118                None,
119                None,
120                Some(&self.agent_id),
121            )
122            .await
123    }

147    async fn store_with_agent(
148        &self,
149        key: &str,
150        content: &str,
151        category: MemoryCategory,
152        session_id: Option<&str>,
153        namespace: Option<&str>,
154        importance: Option<f64>,
155        agent_id: Option<&str>,
156    ) -> Result<()> {
157        // The wrapper's whole purpose is to make every persisted row
158        // attributable to its bound agent. A caller passing an
159        // explicit `agent_id` that does not match is a bug; refuse
160        // loudly so the misuse is debuggable rather than silently
161        // misattributed.
162        if let Some(requested) = agent_id
163            && requested != self.agent_id
164        {
165            ::zeroclaw_log::record!(
166                WARN,
167                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Reject)
168                    .with_outcome(::zeroclaw_log::EventOutcome::Failure)
169                    .with_attrs(::serde_json::json!({
170                        "bound_agent": self.agent_id,
171                        "requested_agent": requested,
172                        "key": key,
173                    })),
174                "store_with_agent refused: foreign agent_id"
175            );
176            anyhow::bail!(
177                "AgentScopedMemory refuses store_with_agent for foreign agent_id; use a wrapper bound to the target agent"
178            );
179        }
180        self.inner
181            .store_with_agent(
182                key,
183                content,
184                category,
185                session_id,
186                namespace,
187                importance,
188                Some(&self.agent_id),
189            )
190            .await
191    }

193    async fn recall(
194        &self,
195        query: &str,
196        limit: usize,
197        session_id: Option<&str>,
198        since: Option<&str>,
199        until: Option<&str>,
200    ) -> Result<Vec<MemoryEntry>> {
201        let allowed = self.allowed_slice();
202        self.inner
203            .recall_for_agents(&allowed, query, limit, session_id, since, until)
204            .await
205    }
206
207    async fn recall_for_agents(
208        &self,
209        caller_allowed: &[&str],
210        query: &str,
211        limit: usize,
212        session_id: Option<&str>,
213        since: Option<&str>,
214        until: Option<&str>,
215    ) -> Result<Vec<MemoryEntry>> {
216        // Intersect the caller-supplied allowlist with the bound
217        // allowlist so a caller cannot widen scope past what the
218        // agent's config permits. Empty caller allowlist means "no
219        // extra restriction"; the bound allowlist still applies.
220        // A non-empty caller allowlist whose intersection with the
221        // bound allowlist is empty means "no rows match" — return
222        // early so the empty-allowlist sentinel ("no filter") on the
223        // inner backend does not silently widen scope.
224        if caller_allowed.is_empty() {
225            let bound: Vec<&str> = self.allowed_agent_ids.iter().map(String::as_str).collect();
226            return self
227                .inner
228                .recall_for_agents(&bound, query, limit, session_id, since, until)
229                .await;
230        }
231
232        let intersected: Vec<&str> = caller_allowed
233            .iter()
234            .copied()
235            .filter(|id| self.allowed_agent_ids.contains(*id))
236            .collect();
237        if intersected.is_empty() {
238            return Ok(Vec::new());
239        }
240        self.inner
241            .recall_for_agents(&intersected, query, limit, session_id, since, until)
242            .await
243    }

245    async fn get(&self, key: &str) -> Result<Option<MemoryEntry>> {
246        // Bound agent's row wins; fall back to allowlisted siblings.
247        // Each lookup is `inner.get_for_agent(key, agent_id)` so
248        // composite-uniqueness backends return the right row per agent
249        // (a single `inner.get(key)` could return any one of the
250        // colliding-key rows).
251        if let Some(own) = self.inner.get_for_agent(key, &self.agent_id).await? {
252            return Ok(Some(own));
253        }
254        for sibling in &self.allowed_agent_ids {
255            if sibling == &self.agent_id {
256                continue;
257            }
258            if let Some(hit) = self.inner.get_for_agent(key, sibling).await? {
259                return Ok(Some(hit));
260            }
261        }
262        Ok(None)
263    }
264
265    async fn get_for_agent(&self, key: &str, agent_id: &str) -> Result<Option<MemoryEntry>> {
266        if agent_id != self.agent_id && !self.allowed_agent_ids.iter().any(|a| a == agent_id) {
267            return Ok(None);
268        }
269        self.inner.get_for_agent(key, agent_id).await
270    }

272    async fn list(
273        &self,
274        category: Option<&MemoryCategory>,
275        session_id: Option<&str>,
276    ) -> Result<Vec<MemoryEntry>> {
277        // Inner.list returns rows across every agent on the install;
278        // post-filter by the bound + allowlisted set so a wrapper-using
279        // caller cannot inspect sibling rows it did not opt into via
280        // `read_memory_from`.
281        let entries = self.inner.list(category, session_id).await?;
282        Ok(entries
283            .into_iter()
284            .filter(|e| {
285                e.agent_id
286                    .as_deref()
287                    .is_some_and(|aid| self.allowed_agent_ids.contains(aid))
288            })
289            .collect())
290    }

292    async fn forget(&self, key: &str) -> Result<bool> {
293        // Only the bound agent's own row may be deleted. Sibling rows
294        // visible via `read_memory_from` are read-only by design — the
295        // allowlist grants recall, never delete. A composite delete on
296        // (key, agent_id) leaves sibling rows untouched and refuses
297        // cross-agent deletion by construction (no row matches).
298        //
299        // When the composite delete finds nothing and `inner.get(key)`
300        // (no agent filter) surfaces a row belonging to another agent,
301        // emit a structured refusal so the operator sees `key`,
302        // `row_agent`, and `bound_agent` as attribution-bound fields.
303        if self.inner.forget_for_agent(key, &self.agent_id).await? {
304            return Ok(true);
305        }
306        match self.inner.get(key).await? {
307            None => Ok(false),
308            Some(entry) => match entry.agent_id.as_deref() {
309                Some(other) => {
310                    ::zeroclaw_log::record!(
311                        WARN,
312                        ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Reject)
313                            .with_outcome(::zeroclaw_log::EventOutcome::Failure)
314                            .with_attrs(::serde_json::json!({
315                                "key": key,
316                                "row_agent": other,
317                                "bound_agent": self.agent_id,
318                            })),
319                        "forget refused: row attributed to a different agent"
320                    );
321                    anyhow::bail!(
322                        "AgentScopedMemory refuses to forget cross-agent row: key attributed to agent other than the bound agent"
323                    );
324                }
325                None => {
326                    ::zeroclaw_log::record!(
327                        WARN,
328                        ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Reject)
329                            .with_outcome(::zeroclaw_log::EventOutcome::Failure)
330                            .with_attrs(::serde_json::json!({
331                                "key": key,
332                                "bound_agent": self.agent_id,
333                            })),
334                        "forget refused: row has no agent attribution"
335                    );
336                    anyhow::bail!(
337                        "AgentScopedMemory refuses to forget unattributed row: legacy or backend without per-agent tracking; resolve via an admin Memory handle"
338                    );
339                }
340            },
341        }
342    }

344    async fn forget_for_agent(&self, key: &str, agent_id: &str) -> Result<bool> {
345        // Only the bound agent can delete its own row through the
346        // wrapper. Allowlist grants recall, never delete.
347        if agent_id != self.agent_id {
348            ::zeroclaw_log::record!(
349                WARN,
350                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Reject)
351                    .with_outcome(::zeroclaw_log::EventOutcome::Failure)
352                    .with_attrs(::serde_json::json!({
353                        "key": key,
354                        "row_agent": agent_id,
355                        "bound_agent": self.agent_id,
356                    })),
357                "forget_for_agent refused: cross-agent delete through wrapper"
358            );
359            anyhow::bail!(
360                "AgentScopedMemory refuses cross-agent forget_for_agent: bound agent and target agent differ"
361            );
362        }
363        self.inner.forget_for_agent(key, agent_id).await
364    }

366    async fn count(&self) -> Result<usize> {
367        // Scope to the bound + allowlisted agents so a wrapper-using
368        // caller does not see the install-wide row total.
369        let entries = self.inner.list(None, None).await?;
370        Ok(entries
371            .into_iter()
372            .filter(|e| {
373                e.agent_id
374                    .as_deref()
375                    .is_some_and(|aid| self.allowed_agent_ids.contains(aid))
376            })
377            .count())
378    }

399    async fn purge_session(&self, session_id: &str) -> Result<usize> {
400        // Bulk session deletes must be scoped by both session and bound
401        // agent at the backend boundary. Listing a session and deleting by
402        // `(key, agent_id)` can delete the bound agent's row from a
403        // different session when keys collide.
404        self.inner
405            .purge_session_for_agent(session_id, &self.agent_id)
406            .await
407    }

425    async fn recall_namespaced(
426        &self,
427        namespace: &str,
428        query: &str,
429        limit: usize,
430        session_id: Option<&str>,
431        since: Option<&str>,
432        until: Option<&str>,
433    ) -> Result<Vec<MemoryEntry>> {
434        // Recall through the agent-scoped recall path so the bound +
435        // allowlisted UUIDs filter at the SQL boundary, then
436        // post-filter for the namespace match. The default trait impl
437        // would route through `recall` which the wrapper has already
438        // overridden, but routing explicitly here keeps the read shape
439        // visible to anyone tracing the call chain.
440        let entries = self
441            .recall(query, limit * 2, session_id, since, until)
442            .await?;
443        Ok(entries
444            .into_iter()
445            .filter(|e| e.namespace == namespace)
446            .take(limit)
447            .collect())
448    }

450    async fn export(&self, filter: &ExportFilter) -> Result<Vec<MemoryEntry>> {
451        // Export is the GDPR data-portability path. An agent-scoped
452        // export sees only the bound + allowlisted agents' rows. The
453        // wrapper's `list` already does the per-agent filtering;
454        // delegate to it and apply the rest of the export filter
455        // post-fetch.
456        let entries = self
457            .list(filter.category.as_ref(), filter.session_id.as_deref())
458            .await?;
459        Ok(entries
460            .into_iter()
461            .filter(|e| {
462                if let Some(ref ns) = filter.namespace
463                    && e.namespace != *ns
464                {
465                    return false;
466                }
467                if let Some(ref since) = filter.since
468                    && e.timestamp.as_str() < since.as_str()
469                {
470                    return false;
471                }
472                if let Some(ref until) = filter.until
473                    && e.timestamp.as_str() > until.as_str()
474                {
475                    return false;
476                }
477                true
478            })
479            .collect())
480    }
