1//! `zeroclaw-spawn` — the sanctioned interface against `tokio::spawn` for
2//! the ZeroClaw workspace.
3//!
4//! Every call site that needs to fan out a background task must go
5//! through [`spawn!`] instead of reaching for `tokio::spawn` directly.
6//! Doing so buys two things, neither of which changes runtime behavior:
7//!
8//! 1. **Attribution propagation.** The spawned future is wrapped with
9//!    `Instrument::in_current_span()` so any `record!` it emits while
10//!    running inherits the caller's `attribution_span` (channel, agent,
11//!    session, cron job, …). Without this wrapper, tokio detaches the
12//!    task from the caller's span stack and records orphan.
13//!
14//! 2. **Lifecycle telemetry.** Two structured log events are emitted
15//!    through the standard `zeroclaw_log::record!` pipeline:
16//!    - `runtime.task.spawn` (`Action::Spawn`) at spawn time, attributed
17//!      to the caller's current span.
18//!    - `runtime.task.spawn` (`Action::Complete`) when the future
19//!      resolves, attributed to the spawned task's span (which is the
20//!      caller's by inheritance), carrying elapsed `duration_ms`.
21//!
22//!    Both records carry `zc_file` / `zc_line` of the call site via the
23//!    standard `record!` machinery, and a `task_site` attribute giving
24//!    the same as a single string for grep convenience.
25//!
26//! The returned [`tokio::task::JoinHandle`] is the unmodified handle
27//! from `tokio::spawn` — panics propagate as `JoinError::Panic`, cancel
28//! semantics are unchanged, and the future's output type is preserved.
29//! `zeroclaw-spawn` adds telemetry, not control flow.
30//!
31//! ## Layering
32//!
33//! `zeroclaw-spawn` depends on `zeroclaw-log`. Crates that already depend
34//! on `zeroclaw-log` (i.e. almost everything in the workspace) can add
35//! `zeroclaw-spawn` as a peer dependency without inverting the graph.
36//! The lowest-level crate (`zeroclaw-api`) intentionally does NOT depend
37//! on `zeroclaw-spawn` — its small handful of internal spawn needs go
38//! through the workspace-wide `disallowed_methods` exemption documented
39//! in `clippy.toml`.

55/// Stable event name for spawn-lifecycle records emitted by [`spawn!`].
56/// Exposed so dashboards / queries can match on a single string instead
57/// of recomputing it.
58pub const TASK_EVENT_NAME: &str = "runtime.task.spawn";

60/// Spawn a future onto the current tokio runtime with attribution
61/// propagation and lifecycle telemetry. Drop-in replacement for
62/// `tokio::spawn` — same signature, same `JoinHandle<T>`, same panic
63/// and cancel semantics.
64///
65/// ```ignore
66/// use zeroclaw_spawn::spawn;
67///
68/// let handle = spawn!(async move {
69///     do_work().await
70/// });
71/// let output = handle.await?;
72/// ```
73///
74/// On spawn the macro emits one `Action::Spawn` record attributed to
75/// the caller's current span. When the future resolves it emits one
76/// `Action::Complete` record with elapsed `duration_ms`, attributed to
77/// the spawned task's span (which is the caller's by inheritance via
78/// `in_current_span`). Neither record alters the future's output or the
79/// returned `JoinHandle`.
80#[macro_export]
81macro_rules! spawn {
82    ($body:expr) => {{
83        #[allow(unused_imports)]
84        use $crate::__private::tracing::Instrument as _;
85
86        // Capture the call-site once. `module_path!`, `file!`, `line!`
87        // expand at the spawn point, not inside the spawned task, so
88        // both lifecycle records carry the originating location.
89        const __ZC_TASK_MODULE: &'static str = module_path!();
90        const __ZC_TASK_FILE: &'static str = file!();
91        const __ZC_TASK_LINE: u32 = line!();
92
93        // Spawn-time record — fires synchronously on the caller's
94        // thread, attributed to whatever span the caller currently has
95        // entered.
96        $crate::__private::zeroclaw_log::record!(
97            INFO,
98            $crate::__private::zeroclaw_log::Event::new(
99                $crate::TASK_EVENT_NAME,
100                $crate::__private::zeroclaw_log::Action::Spawn,
101            )
102            .with_attrs($crate::__private::serde_json::json!({
103                "task_site": format!("{}:{}", __ZC_TASK_FILE, __ZC_TASK_LINE),
104                "task_module": __ZC_TASK_MODULE,
105            })),
106            "task spawned"
107        );
108
109        // Wrap the user's future so we can stamp a Complete record when
110        // it resolves. The wrapper is itself `.in_current_span()`d, so
111        // the completion record inherits the same attribution context
112        // the caller had at spawn time.
113        let __zc_task_started_at = $crate::__private::tokio::time::Instant::now();
114        let __zc_task_future = async move {
115            let __zc_task_output = { $body }.await;
116            let __zc_task_elapsed_ms = __zc_task_started_at.elapsed().as_millis() as u64;
117            $crate::__private::zeroclaw_log::record!(
118                INFO,
119                $crate::__private::zeroclaw_log::Event::new(
120                    $crate::TASK_EVENT_NAME,
121                    $crate::__private::zeroclaw_log::Action::Complete,
122                )
123                .with_outcome($crate::__private::zeroclaw_log::EventOutcome::Success)
124                .with_duration(__zc_task_elapsed_ms)
125                .with_attrs($crate::__private::serde_json::json!({
126                    "task_site": format!("{}:{}", __ZC_TASK_FILE, __ZC_TASK_LINE),
127                    "task_module": __ZC_TASK_MODULE,
128                })),
129                "task complete"
130            );
131            __zc_task_output
132        };
133
134        #[allow(clippy::disallowed_methods)]
135        let __zc_spawn_handle =
136            $crate::__private::tokio::spawn(__zc_task_future.in_current_span());
137        __zc_spawn_handle
138    }};
139}

141#[cfg(test)]
142mod tests {
143    use std::sync::{Arc, Mutex};
144    use tracing::{Subscriber, span};
145    use tracing_subscriber::layer::{Context, Layer, SubscriberExt};
146    use tracing_subscriber::registry::{LookupSpan, Registry};
147
148    /// Layer that records, for every event it sees, the names of every
149    /// span on the event's span stack at the moment of recording. Lets
150    /// us assert "yes, the spawned task's event saw the caller's span"
151    /// without depending on `zeroclaw-log` formatting.
152    #[derive(Clone, Default)]
153    struct SpanCapture {
154        events: Arc<Mutex<Vec<Vec<String>>>>,
155    }
156
157    impl<S> Layer<S> for SpanCapture
158    where
159        S: Subscriber + for<'a> LookupSpan<'a>,
160    {
161        fn on_event(&self, _event: &tracing::Event<'_>, ctx: Context<'_, S>) {
162            let mut stack = Vec::new();
163            if let Some(scope) = ctx.event_scope(_event) {
164                for span in scope.from_root() {
165                    stack.push(span.name().to_string());
166                }
167            }
168            self.events.lock().unwrap().push(stack);
169        }
170    }
171
172    #[tokio::test]
173    async fn spawn_returns_future_output() {
174        let handle = crate::spawn!(async { 42_u32 });
175        assert_eq!(handle.await.unwrap(), 42);
176    }
177
178    #[tokio::test]
179    async fn spawn_preserves_error_type() {
180        let handle = crate::spawn!(async { Err::<(), &'static str>("nope") });
181        assert_eq!(handle.await.unwrap(), Err("nope"));
182    }
183
184    #[tokio::test]
185    async fn spawn_runs_to_completion_with_await_point() {
186        let handle = crate::spawn!(async {
187            tokio::time::sleep(std::time::Duration::from_millis(1)).await;
188            "done"
189        });
190        assert_eq!(handle.await.unwrap(), "done");
191    }
192
193    /// Load-bearing test for the whole point of this crate: an event
194    /// emitted *inside* a spawned task must observe the caller's
195    /// `attribution_span` on its span stack. If this regresses, every
196    /// `record!` from inside a `spawn!` body silently re-attributes to
197    /// the tokio root and dashboards lose session/channel/agent
198    /// attribution — exactly the bug `.in_current_span()` exists to
199    /// prevent.
200    #[tokio::test]
201    async fn spawn_propagates_callers_span_into_task() {
202        let capture = SpanCapture::default();
203        let subscriber = Registry::default().with(capture.clone());
204        let _guard = tracing::subscriber::set_default(subscriber);
205
206        let outer = span!(tracing::Level::INFO, "attribution_span");
207        let _entered = outer.enter();
208
209        let handle = crate::spawn!(async {
210            // Yield once so the task actually re-enters its instrumented
211            // span on a different poll than the spawn site.
212            tokio::task::yield_now().await;
213            tracing::event!(tracing::Level::INFO, "inside_spawned_task");
214        });
215        handle.await.unwrap();
216
217        drop(_entered);
218
219        let events = capture.events.lock().unwrap();
220        // Find the event we emitted from inside the task and assert it
221        // saw `attribution_span` on its stack.
222        let saw_outer = events
223            .iter()
224            .any(|stack| stack.iter().any(|name| name == "attribution_span"));
225        assert!(
226            saw_outer,
227            "spawned task lost caller's span; captured stacks: {:?}",
228            *events
229        );
230    }
