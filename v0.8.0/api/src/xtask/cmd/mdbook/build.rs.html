8pub fn run(tag: Option<&str>) -> anyhow::Result<()> {
9    let root = repo_root();
10    require_tool("cargo", "https://rustup.rs")?;
11    ensure_cargo_tool("mdbook", "mdbook")?;
12    ensure_cargo_tool("mdbook-xgettext", "mdbook-i18n-helpers")?;
13    ensure_cargo_tool("mdbook-gettext", "mdbook-i18n-helpers")?;
14    ensure_cargo_tool("mdbook-mermaid", "mdbook-mermaid")?;
15
16    build_refs(&root)?;
17    build_api(&root)?;
18    build_locales(&root, tag)?;
19    crate::cmd::mdbook::linkcheck::check_internal_links(&root, tag.unwrap_or(DEFAULT_TAG))?;
20    assemble(&root, tag)?;
21    println!(
22        "==> Done. Open: {}",
23        book_dir(&root)
24            .join("book")
25            .join(tag.unwrap_or(DEFAULT_TAG))
26            .join("index.html")
27            .display()
28    );
29    Ok(())
30}

32pub fn build_locales(root: &std::path::Path, tag: Option<&str>) -> anyhow::Result<()> {
33    let book = book_dir(root);
34    let entries = locale_entries();
35    println!(
36        "==> Building mdBook for locales: {}",
37        entries
38            .iter()
39            .map(|e| e.code.as_str())
40            .collect::<Vec<_>>()
41            .join(" ")
42    );
43    inject_lang_switcher_locales(&book, &entries)?;
44    crate::cmd::mdbook::themes::run(root)?;
45    crate::cmd::mdbook::keymap::run(root)?;
46    crate::cmd::mdbook::hardware::run(root)?;
47    let mdbook = mdbook_program()?;
48    let preprocessor_env = peer_groups_preprocessor_env();
49    let tag_dir = tag.unwrap_or(DEFAULT_TAG);
50    for entry in &entries {
51        let dest = format!("book/{}/{}", tag_dir, entry.code);
52        let mut cmd = Command::new(&mdbook);
53        cmd.args(["build", "-d", &dest])
54            .env("MDBOOK_BOOK__LANGUAGE", &entry.code)
55            .current_dir(&book);
56        if let Some((key, value)) = &preprocessor_env {
57            cmd.env(key, value);
58        }
59        run_cmd(&mut cmd)?;
60    }
61    Ok(())
62}

64/// Render `theme/lang-switcher.js.tpl` into `theme/lang-switcher.js` with the
65/// `LOCALES` array filled from `locales.toml`. The `.js` output is gitignored
66/// (every locale add/remove rewrites it); the `.tpl` source is the tracked
67/// truth. Errors loudly when the template is missing — silently skipping
68/// would let mdBook fail later with a confusing "missing additional-js"
69/// message.
70pub fn inject_lang_switcher_locales(book: &Path, entries: &[LocaleEntry]) -> anyhow::Result<()> {
71    let tpl_path = book.join("theme/lang-switcher.js.tpl");
72    let js_path = book.join("theme/lang-switcher.js");
73    let src = std::fs::read_to_string(&tpl_path).map_err(|e| {
74        anyhow::Error::msg(format!(
75            "lang-switcher.js.tpl missing at {}: {e}. The template is the tracked source of \
76             truth for the locale switcher; do not delete it.",
77            tpl_path.display(),
78        ))
79    })?;
80    let locale_lines: String = entries
81        .iter()
82        .map(|e| format!("    {{ code: {:?}, label: {:?} }},", e.code, e.label))
83        .collect::<Vec<_>>()
84        .join("\n");
85    let new_block = format!("const LOCALES = [\n{locale_lines}\n  ];");
86
87    let start = src
88        .find("const LOCALES = [")
89        .ok_or_else(|| anyhow::Error::msg("lang-switcher.js.tpl: LOCALES array not found"))?;
90    let end = src[start..]
91        .find("];")
92        .ok_or_else(|| anyhow::Error::msg("lang-switcher.js.tpl: LOCALES closing ]; not found"))?;
93    let updated = format!("{}{}{}", &src[..start], new_block, &src[start + end + 2..]);
94    std::fs::write(&js_path, updated)?;
95    Ok(())
96}

103pub fn assemble(root: &std::path::Path, tag: Option<&str>) -> anyhow::Result<()> {
104    println!("==> Assembling site (rustdoc + locale redirect)");
105    let book = book_dir(root);
106    let tag_dir = tag.unwrap_or(DEFAULT_TAG);
107    let api_dest = book.join("book").join(tag_dir).join("api");
108    let _ = std::fs::remove_dir_all(&api_dest);
109    copy_dir_all(doc_dir(root), &api_dest)?;
110
111    const INDEX_HTML: &str = "<!doctype html>\n<meta charset=\"utf-8\">\n<meta http-equiv=\"refresh\" content=\"0; url=./en/\">\n<link rel=\"canonical\" href=\"./en/\">\n<title>ZeroClaw Docs</title>\n";
112    let out_dir = book.join("book").join(tag_dir);
113    std::fs::create_dir_all(&out_dir)?;
114    std::fs::write(out_dir.join("index.html"), INDEX_HTML)?;
115    // Write small metadata file with the version tag
116    let version_meta = format!("{}\n", tag.unwrap_or(DEFAULT_TAG));
117    std::fs::write(out_dir.join("_version.txt"), version_meta)?;
118
119    let version_dir = out_dir;
120    let shared_dir = book.join("book").join("_shared");
121    extract_shared_chrome(&version_dir, &shared_dir)?;
122    Ok(())
123}

125pub fn extract_shared_chrome(version_dir: &Path, shared_dir: &Path) -> anyhow::Result<()> {
126    println!("==> Extracting shared chrome layer");
127
128    let first_locale = locale_entries()
129        .into_iter()
130        .next()
131        .map(|e| e.code)
132        .unwrap_or_else(|| "en".to_string());
133    let src_dir = version_dir.join(&first_locale);
134    if !src_dir.exists() {
135        return Ok(());
136    }
137
138    // Map each hashed chrome file (path relative to the locale dir, e.g.
139    // `theme/custom-abc12345.css`) to its unhashed `_shared`-relative path
140    // (e.g. `theme/custom.css`). The `../` prefix is applied per HTML file
141    // below, because a page's correct depth to the version root — and thus to
142    // `_shared` at the gh-pages root — depends on how deep the page sits.
143    let mut replacements = Vec::new();
144    let prefixes = [
145        "css/chrome",
146        "theme/custom",
147        "theme/version-selector",
148        "theme/lang-switcher",
149        "favicon",
150        "theme/pc-themes",
151        "theme/pc-enhance",
152    ];
153
154    let walk_dir = |dir: &Path| -> Vec<std::path::PathBuf> {
155        let mut paths = Vec::new();
156        let mut stack = vec![dir.to_path_buf()];
157        while let Some(path) = stack.pop() {
158            if let Ok(entries) = std::fs::read_dir(&path) {
159                for entry in entries.flatten() {
160                    if let Ok(ty) = entry.file_type() {
161                        if ty.is_dir() {
162                            stack.push(entry.path());
163                        } else {
164                            paths.push(entry.path());
165                        }
166                    }
167                }
168            }
169        }
170        paths
171    };
172
173    for file in walk_dir(&src_dir) {
174        if let Ok(rel) = file.strip_prefix(&src_dir) {
175            let rel_str = rel.to_string_lossy().replace('\\', "/");
176            if !prefixes.iter().any(|p| rel_str.starts_with(p)) {
177                continue;
178            }
179            let file_name = file.file_name().unwrap().to_string_lossy();
180            if let Some(pos) = file_name.rfind('-')
181                && let Some(ext_pos) = file_name.rfind('.')
182                && pos < ext_pos
183            {
184                let hash = &file_name[pos + 1..ext_pos];
185                if hash.len() == 8 && hash.chars().all(|c| c.is_ascii_hexdigit()) {
186                    let unhashed_name = format!("{}{}", &file_name[..pos], &file_name[ext_pos..]);
187                    let dest_rel = rel.parent().unwrap().join(unhashed_name);
188                    let dest = shared_dir.join(&dest_rel);
189                    std::fs::create_dir_all(dest.parent().unwrap())?;
190                    std::fs::copy(&file, &dest)?;
191                    let dest_rel_str = dest_rel.to_string_lossy().replace('\\', "/");
192                    // Store (locale-relative hashed path, unhashed shared-relative path).
193                    replacements.push((rel_str.clone(), dest_rel_str));
194                }
195            }
196        }
197    }
198
199    for entry in locale_entries() {
200        let loc_dir = version_dir.join(&entry.code);
201        for file in walk_dir(&loc_dir) {
202            // Depth of this HTML file below the locale dir. mdBook emits chrome
203            // refs as `<../ × (locale_depth + page_depth)>theme/foo-HASH.css`
204            // for an HTML page `page_depth` levels under the locale dir; the
205            // matching `_shared` ref needs the same total `../` count plus one
206            // to clear the version dir up to the gh-pages root where `_shared`
207            // lives. Concretely: page directly in `<tag>/<locale>/` -> `../../`,
208            // one level deeper -> `../../../`, and so on.
209            let page_depth = file
210                .strip_prefix(&loc_dir)
211                .ok()
212                .map(|rel| rel.components().count().saturating_sub(1))
213                .unwrap_or(0);
214            let up = "../".repeat(page_depth + 2);
215
216            if file.extension().is_some_and(|e| e == "html")
217                && let Ok(mut content) = std::fs::read_to_string(&file)
218            {
219                let mut changed = false;
220                for (hashed_rel, shared_rel) in &replacements {
221                    // mdBook references the chrome file relative to the page, so
222                    // the on-disk ref is `<../ × page_depth><hashed_rel>`. Match
223                    // and replace at that exact depth with the `_shared` ref at
224                    // the correct depth — never a hardcoded prefix.
225                    let page_up = "../".repeat(page_depth);
226                    let from = format!("{page_up}{hashed_rel}");
227                    let to = format!("{up}_shared/{shared_rel}");
228                    if content.contains(&from) {
229                        content = content.replace(&from, &to);
230                        changed = true;
231                    }
232                }
233                if changed {
234                    let _ = std::fs::write(&file, content);
235                }
236            }
237            if let Ok(rel) = file.strip_prefix(&loc_dir) {
238                let rel_str = rel.to_string_lossy().replace('\\', "/");
239                if replacements.iter().any(|(from, _)| from == &rel_str) {
240                    let _ = std::fs::remove_file(&file);
241                }
242            }
243        }
244    }
245
246    Ok(())
