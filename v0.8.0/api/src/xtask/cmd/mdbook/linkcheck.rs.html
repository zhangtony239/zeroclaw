1//! Internal link integrity check for the built book.
2//!
3//! mdBook does not validate that a page's `href` resolves to a real file, so a
4//! relative link that points nowhere (commonly from an `{{#include}}` that
5//! drags a page's own relative links into a different directory) renders as a
6//! dead link and ships silently. This walks the generated HTML for one locale,
7//! resolves every internal `href` against the file it appears in, and fails the
8//! build if any target is missing.
9//!
10//! Only same-site relative links are checked. External (`http(s)://`,
11//! `mailto:`), in-page anchors (`#...`), and the generated rustdoc `api/` tree
12//! are skipped — the api tree is copied in `assemble()` after this runs, and is
13//! rustdoc's own output, not authored.
14
15use std::collections::BTreeSet;
16use std::path::{Path, PathBuf};
17
18/// Check internal links in the built HTML for `tag`'s first locale.
19///
20/// Run after `build_locales` (HTML present) and before `assemble` (which adds
21/// the `api/` tree). Checking one locale is sufficient: every locale is built
22/// from the same Markdown source, so a broken authored link breaks identically
23/// in all of them.
24pub fn check_internal_links(root: &Path, tag: &str) -> anyhow::Result<()> {
25    let locale = crate::util::locale_entries()
26        .into_iter()
27        .next()
28        .map(|e| e.code)
29        .unwrap_or_else(|| "en".to_string());
30    let base = crate::util::book_dir(root)
31        .join("book")
32        .join(tag)
33        .join(&locale);
34    if !base.is_dir() {
35        return Ok(());
36    }
37
38    let mut broken: Vec<(PathBuf, String)> = Vec::new();
39    for html in walk_html(&base) {
40        let content = match std::fs::read_to_string(&html) {
41            Ok(c) => c,
42            Err(_) => continue,
43        };
44        let dir = html.parent().unwrap_or(&base);
45        for href in extract_hrefs(&content) {
46            if !is_checkable(&href) {
47                continue;
48            }
49            // Strip the in-page anchor; we only verify the target file exists.
50            let path_part = href.split('#').next().unwrap_or(&href);
51            if path_part.is_empty() {
52                continue;
53            }
54            let target = normalize(&dir.join(path_part));
55            // A link ending in `/` (or to a dir) resolves to its index.html.
56            let resolved = if path_part.ends_with('/') {
57                target.join("index.html")
58            } else {
59                target
60            };
61            if !resolved.exists() {
62                let rel = html.strip_prefix(&base).unwrap_or(&html).to_path_buf();
63                broken.push((rel, href));
64            }
65        }
66    }
67
68    if broken.is_empty() {
69        println!("==> Internal link check passed");
70        return Ok(());
71    }
72    let mut msg = String::from("internal link check failed; dead links found:\n");
73    let mut seen = BTreeSet::new();
74    for (page, href) in &broken {
75        let line = format!("  {} -> {}", page.display(), href);
76        if seen.insert(line.clone()) {
77            msg.push_str(&line);
78            msg.push('\n');
79        }
80    }
81    anyhow::bail!(msg)
82}

84fn walk_html(dir: &Path) -> Vec<PathBuf> {
85    let mut out = Vec::new();
86    let mut stack = vec![dir.to_path_buf()];
87    while let Some(p) = stack.pop() {
88        if let Ok(entries) = std::fs::read_dir(&p) {
89            for e in entries.flatten() {
90                let path = e.path();
91                if path.is_dir() {
92                    stack.push(path);
93                } else if path.extension().is_some_and(|x| x == "html") {
94                    out.push(path);
95                }
96            }
97        }
98    }
99    out
100}

102/// Pull `href="..."` values out of rendered HTML. Naive but sufficient for
103/// mdBook output, which quotes every href with double quotes.
104fn extract_hrefs(html: &str) -> Vec<String> {
105    let mut out = Vec::new();
106    let needle = "href=\"";
107    let mut rest = html;
108    while let Some(i) = rest.find(needle) {
109        rest = &rest[i + needle.len()..];
110        if let Some(end) = rest.find('"') {
111            out.push(rest[..end].to_string());
112            rest = &rest[end + 1..];
113        } else {
114            break;
115        }
116    }
117    out
118}

120/// Whether an href is a same-site relative link we should verify.
121fn is_checkable(href: &str) -> bool {
122    if href.is_empty() || href.starts_with('#') {
123        return false;
124    }
125    // Absolute URLs, protocol-relative, scheme links: not ours to check.
126    if href.contains("://") || href.starts_with("//") || href.starts_with("mailto:") {
127        return false;
128    }
129    // Site-absolute paths (e.g. `/_shared/...`) and the rustdoc api tree are
130    // added/owned outside the authored Markdown.
131    if href.starts_with('/') || href.starts_with("api/") || href.contains("/api/") {
132        return false;
133    }
134    true
135}

137/// Resolve `.`/`..` segments without touching the filesystem (the target may
138/// not exist, which is exactly what we are testing for).
139fn normalize(path: &Path) -> PathBuf {
140    let mut out = PathBuf::new();
141    for comp in path.components() {
142        match comp {
143            std::path::Component::ParentDir => {
144                out.pop();
145            }
146            std::path::Component::CurDir => {}
147            other => out.push(other.as_os_str()),
148        }
149    }
150    out
151}

153#[cfg(test)]
154mod tests {
155    use super::*;
156
157    #[test]
158    fn extract_hrefs_pulls_quoted_links() {
159        let html = r#"<a href="./config.html">x</a> <a href="../zerocode/overview.html#sec">y</a>"#;
160        assert_eq!(
161            extract_hrefs(html),
162            vec!["./config.html", "../zerocode/overview.html#sec"]
163        );
164    }
165
166    #[test]
167    fn is_checkable_skips_external_anchor_and_site_absolute() {
168        assert!(is_checkable("./config.html"));
169        assert!(is_checkable("../zerocode/running.html"));
170        assert!(!is_checkable("https://example.com"));
171        assert!(!is_checkable("#section"));
172        assert!(!is_checkable("/_shared/theme/custom.css"));
173        assert!(!is_checkable("mailto:x@y.z"));
174        assert!(!is_checkable("api/zeroclaw/index.html"));
175    }
176
177    #[test]
178    fn normalize_resolves_parent_segments() {
179        let p = normalize(Path::new(
180            "/book/en/getting-started/../zerocode/overview.html",
181        ));
182        assert_eq!(p, PathBuf::from("/book/en/zerocode/overview.html"));
183    }
