1//! Generate the zerocode keybinding snippets for the mdBook from the canonical
2//! default keymap (`apps/zerocode/src/keymap/actions.rs`). The registry is the
3//! single source of truth; the docs render whatever chords it carries so a
4//! rebind never drifts the rendered tables.
5//!
6//! Outputs (gitignored, derived — `{{#include}}`d at build time):
7//!   docs/book/src/_snippets/zerocode-pane-nav-keys.md
8//!   docs/book/src/_snippets/zerocode-quickstart-pane-keys.md
9
10use anyhow::Result;
11use std::fmt::Write as _;
12use std::path::Path;
13
14use zerocode::keymap::{Chord, GlobalAction, QuickstartTabAction, RebindableActions};
15
16const SNIPPET_DIR: &str = "docs/book/src/_snippets";
17
18const HEADER: &str = "<!--\n  GENERATED by `cargo xtask mdbook` from the default keymap in\n  apps/zerocode/src/keymap/actions.rs. Derived + gitignored: do not edit,\n  do not commit. Regenerated on every docs build.\n-->\n\n";
19
20pub fn run(root: &Path) -> Result<()> {
21    let dir = root.join(SNIPPET_DIR);
22    std::fs::write(
23        dir.join("zerocode-pane-nav-keys.md"),
24        render_table(pane_nav_rows()),
25    )?;
26    std::fs::write(
27        dir.join("zerocode-quickstart-pane-keys.md"),
28        render_table(quickstart_rows()),
29    )?;
30    println!("==> Generated zerocode keybinding snippets from default keymap");
31    Ok(())
32}
33
34fn render_chords(chords: &[Chord]) -> String {
35    let mut seen: Vec<String> = Vec::new();
36    for c in chords {
37        let d = c.display();
38        if !seen.contains(&d) {
39            seen.push(d);
40        }
41    }
42    seen.join(" / ")
43}
44
45fn quickstart_rows() -> Vec<(String, &'static str)> {
46    QuickstartTabAction::all()
47        .iter()
48        .map(|a| (render_chords(&a.defaults()), a.human_label()))
49        .collect()
50}
51
52fn pane_nav_rows() -> Vec<(String, &'static str)> {
53    GlobalAction::all()
54        .iter()
55        .filter(|a| matches!(a, GlobalAction::PaneNavLeft | GlobalAction::PaneNavRight))
56        .map(|a| (render_chords(&a.defaults()), a.human_label()))
57        .collect()
58}
59
60fn render_table(rows: Vec<(String, &'static str)>) -> String {
61    let mut out = String::from(HEADER);
62    out.push_str("| Keys | Action |\n|---|---|\n");
63    for (keys, label) in rows {
64        let _ = writeln!(out, "| `{keys}` | {label} |");
65    }
66    out
67}
