32/// Append-only logger for failed translation attempts — records the exact source string,
33/// raw model response, and error so failure patterns can be inspected after the run.
34struct FailureLog {
35    file: Mutex<std::fs::File>,
36}
37
38impl FailureLog {
39    fn open(path: &std::path::Path) -> anyhow::Result<Self> {
40        let file = std::fs::OpenOptions::new()
41            .create(true)
42            .append(true)
43            .open(path)?;
44        Ok(Self {
45            file: Mutex::new(file),
46        })
47    }
48
49    fn record(&self, chunk: usize, source: &str, response: &str, err: &anyhow::Error) {
50        let mut f = self.file.lock().expect("failure log mutex poisoned");
51        let _ = writeln!(
52            f,
53            "==== chunk {chunk} — {}\n-- error: {err}\n-- source: {source:?}\n-- response: {response:?}\n",
54            chrono_now()
55        );
56    }
57}

59fn chrono_now() -> String {
60    use std::time::{SystemTime, UNIX_EPOCH};
61    let secs = SystemTime::now()
62        .duration_since(UNIX_EPOCH)
63        .map(|d| d.as_secs())
64        .unwrap_or(0);
65    format!("epoch={secs}")
66}

68/// A parsed .po entry, carrying line positions so we can rewrite in place.
69struct Entry {

71    msgstr_line: usize,
72    /// 0-based line index of the `#, fuzzy` flag line, if present

76    /// Decoded msgstr text
77    msgstr: String,
78}

80/// Decode a run of po quoted-string lines into a plain Rust String.
81/// Each line looks like `"some text\n"` — strip outer quotes, unescape.
82fn decode_po_string(lines: &[String]) -> String {
83    let mut out = String::new();
84    for line in lines {
85        let inner = line.trim();
86        if inner.starts_with('"') && inner.ends_with('"') && inner.len() >= 2 {
87            let s = &inner[1..inner.len() - 1];
88            let mut chars = s.chars().peekable();
89            while let Some(c) = chars.next() {
90                if c == '\\' {
91                    match chars.next() {
92                        Some('n') => out.push('\n'),
93                        Some('t') => out.push('\t'),
94                        Some('\\') => out.push('\\'),
95                        Some('"') => out.push('"'),
96                        Some(other) => {
97                            out.push('\\');
98                            out.push(other);
99                        }
100                        None => out.push('\\'),
101                    }
102                } else {
103                    out.push(c);
104                }
105            }
106        }
107    }
108    out
109}

111/// Outcome of checking a model response against its source string.
112enum LeakCheck {
113    Clean,
114    Recovered(String),
115    Unrecoverable,
116}
117
118/// Detect a prompt-leak response and attempt to recover the real translation.
119///
120/// When a model leaks its instructions it translates them into the target language and
121/// often appends the actual translation at the end. The leak is structural: the response
122/// is far longer than any plausible translation of `source`, or starts with a bullet list.
123fn check_for_leak(source: &str, response: &str) -> LeakCheck {
124    let leak_threshold = source.len().saturating_mul(4).max(120);
125    let looks_like_bullets = response.trim_start().starts_with("- ") && response.contains("\\n- ");
126    let too_long = response.len() > leak_threshold;
127    if !too_long && !looks_like_bullets {
128        return LeakCheck::Clean;
129    }
130    // Try to recover: prefer the last paragraph after a blank line, else everything
131    // after the final terminal punctuation ('. ' or '.').
132    let candidate = response
133        .trim()
134        .rsplit("\n\n")
135        .find(|s| !s.trim().is_empty())
136        .map(str::to_string)
137        .or_else(|| {
138            response
139                .trim()
140                .rsplit(". ")
141                .next()
142                .map(|s| s.trim_end_matches('.').trim().to_string())
143        });
144    match candidate {
145        Some(c) if !c.is_empty() && c.len() <= leak_threshold => LeakCheck::Recovered(c),
146        _ => LeakCheck::Unrecoverable,
147    }
148}

150/// Encode a plain string into a single-line po `msgstr "..."` value.
151fn encode_po_string(s: &str) -> String {
152    let mut out = String::new();
153    for c in s.chars() {
154        match c {
155            '"' => out.push_str("\\\""),
156            '\\' => out.push_str("\\\\"),
157            '\n' => out.push_str("\\n"),
158            '\t' => out.push_str("\\t"),
159            other => out.push(other),
160        }
161    }
162    out
163}

165fn commit_entry(
166    entries: &mut Vec<Entry>,
167    fuzzy_line: Option<usize>,
168    msgstr_line_idx: Option<usize>,
169    msgid_lines: &[String],
170    msgstr_lines: &[String],
171) {
172    let Some(ms_line) = msgstr_line_idx else {
173        return;
174    };
175    let msgid = decode_po_string(msgid_lines);
176    let msgstr = decode_po_string(msgstr_lines);
177    if msgid.is_empty() {
178        return; // header entry
179    }
180    entries.push(Entry {
181        msgstr_line: ms_line,
182        fuzzy_line,
183        msgid,
184        msgstr,
185    });
186}

188fn parse_po(lines: &[String]) -> Vec<Entry> {
189    let mut entries = Vec::new();
190    let mut fuzzy_line: Option<usize> = None;
191    let mut in_msgid = false;
192    let mut in_msgstr = false;
193    let mut msgid_lines: Vec<String> = Vec::new();
194    let mut msgstr_lines: Vec<String> = Vec::new();
195    let mut msgstr_line_idx: Option<usize> = None;
196
197    for (idx, line) in lines.iter().enumerate() {
198        let trimmed = line.trim_end();
199
200        if trimmed.starts_with("#,") && trimmed.contains("fuzzy") {
201            commit_entry(
202                &mut entries,
203                fuzzy_line,
204                msgstr_line_idx,
205                &msgid_lines,
206                &msgstr_lines,
207            );
208            fuzzy_line = Some(idx);
209            in_msgid = false;
210            in_msgstr = false;
211            msgid_lines.clear();
212            msgstr_lines.clear();
213            msgstr_line_idx = None;
214            continue;
215        }
216
217        if let Some(rest) = trimmed.strip_prefix("msgid ") {
218            if msgstr_line_idx.is_some() {
219                commit_entry(
220                    &mut entries,
221                    fuzzy_line,
222                    msgstr_line_idx,
223                    &msgid_lines,
224                    &msgstr_lines,
225                );
226                fuzzy_line = None;
227                msgid_lines.clear();
228                msgstr_lines.clear();
229                msgstr_line_idx = None;
230            }
231            in_msgid = true;
232            in_msgstr = false;
233            msgid_lines.clear();
234            msgid_lines.push(rest.to_string());
235            continue;
236        }
237
238        if let Some(rest) = trimmed.strip_prefix("msgstr ") {
239            in_msgid = false;
240            in_msgstr = true;
241            msgstr_lines.clear();
242            msgstr_line_idx = Some(idx);
243            msgstr_lines.push(rest.to_string());
244            continue;
245        }
246
247        if trimmed.starts_with('"') {
248            if in_msgid {
249                msgid_lines.push(trimmed.to_string());
250            }
251            if in_msgstr {
252                msgstr_lines.push(trimmed.to_string());
253            }
254            continue;
255        }
256
257        if trimmed.is_empty() || trimmed.starts_with('#') {
258            in_msgid = false;
259            in_msgstr = false;
260        }
261    }
262    commit_entry(
263        &mut entries,
264        fuzzy_line,
265        msgstr_line_idx,
266        &msgid_lines,
267        &msgstr_lines,
268    );
269    entries
270}

272fn write_po(
273    lines: &[String],
274    raw: &str,
275    translations: &HashMap<usize, String>,
276    translated_entries: &[&Entry],
277    to_accept: &[&Entry],
278    path: &std::path::Path,
279) -> anyhow::Result<()> {
280    // Remove fuzzy flags for entries we translated + entries accepted as-is
281    let fuzzy_lines_to_remove: std::collections::HashSet<usize> = translated_entries
282        .iter()
283        .filter(|e| e.fuzzy_line.is_some() && translations.contains_key(&e.msgstr_line))
284        .chain(to_accept.iter())
285        .filter_map(|e| e.fuzzy_line)
286        .collect();
287
288    let mut output_lines: Vec<String> = Vec::with_capacity(lines.len());
289    let mut i = 0;
290    while i < lines.len() {
291        if fuzzy_lines_to_remove.contains(&i) {
292            i += 1;
293            continue;
294        }
295        if let Some(translated) = translations.get(&i) {
296            output_lines.push(format!("msgstr \"{}\"", encode_po_string(translated)));
297            i += 1;
298            while i < lines.len() && lines[i].trim_start().starts_with('"') {
299                i += 1;
300            }
301            continue;
302        }
303        output_lines.push(lines[i].clone());
304        i += 1;
305    }
306
307    let mut out = output_lines.join("\n");
308    if raw.ends_with('\n') {
309        out.push('\n');
310    }
311    std::fs::write(path, out)?;
312    Ok(())
313}

315/// Strip wrapping characters the model added that weren't present in the source.
316///
317/// Handles the common failure modes observed in logs: a whole translation wrapped in
318/// backticks, corner brackets (`「」`/`『』`), straight or curly quotes, or the JSON field
319/// leak `t="..."`. Applies each rule only when the wrapper is symmetric AND absent from
320/// the source, so legitimate source-side wrapping is preserved.
321/// Outcome of a translate_batch call. On failure, `raw_response` carries the full model
322/// response (empty if the failure was before we got one — e.g. network) for logging.
323struct BatchFailure {
324    err: anyhow::Error,
325    raw_response: String,
326}
327
328type BatchResult = Result<Vec<String>, BatchFailure>;
329
330fn fail(err: anyhow::Error, raw_response: impl Into<String>) -> BatchFailure {
331    BatchFailure {
332        err,
333        raw_response: raw_response.into(),
334    }
335}

337/// Translate each source string via the shared runtime provider. The provider
338/// stack handles endpoint, auth, and wire protocol per family — this tool
339/// builds no HTTP. One request per source string keeps the per-entry mapping
340/// unambiguous (the .po model is one msgid -> one msgstr).
341async fn translate_batch(
342    provider: &dyn zeroclaw_api::model_provider::ModelProvider,
343    model: &str,
344    locale: &str,
345    batch: &[&str],
346) -> BatchResult {
347    let system = format!(
348        "You translate English technical documentation strings to {locale}.\n\
349         - Preserve backticks, bold (**text**), inline code, URLs, and escape sequences where \
350           they appear in the source, character-for-character.\n\
351         - Do not translate: brand and project names, command names, CLI flags, file paths, \
352           environment variables, code literals, function/type names.\n\
353         - If the input is already in {locale}, a code literal, a URL, or a single identifier, \
354           return it unchanged.\n\
355         - Use established software-localization terminology in {locale} rather than literal \
356           morpheme-by-morpheme translation.\n\
357         - Return ONLY the translated string, no quotes, no preamble, no explanation."
358    );
359
360    let mut out = Vec::with_capacity(batch.len());
361    for source in batch {
362        let content = provider
363            .chat_with_system(Some(&system), source, model, None)
364            .await
365            .map_err(|e| fail(e, String::new()))?;
366        out.push(content.trim().to_string());
367    }
368    Ok(out)
369}

371#[tokio::main]
372async fn main() -> anyhow::Result<()> {
373    let args = Args::parse();
374
375    if args.po.extension().and_then(|e| e.to_str()) != Some("po") {
376        anyhow::bail!("--po path must have a .po extension: {}", args.po.display());
377    }
378    if !args.po.exists() {
379        anyhow::bail!("--po path does not exist: {}", args.po.display());
380    }
381
382    let (provider, model) =
383        xtask::util::build_model_provider(&args.model_provider, args.config_dir.as_deref())?;
384
385    let raw = std::fs::read_to_string(&args.po)?;
386    let lines: Vec<String> = raw.lines().map(str::to_owned).collect();
387
388    let entries = parse_po(&lines);
389
390    let mut translations: HashMap<usize, String> = HashMap::new();
391
392    // Repair entries where msgid ends with \n but msgstr doesn't — corrupted by
393    // interrupted runs. Pre-populate into translations so write_po fixes them inline.
394    let mut repaired = 0;
395    for entry in &entries {
396        if !entry.msgstr.is_empty() && entry.msgid.ends_with('\n') && !entry.msgstr.ends_with('\n')
397        {
398            translations.insert(entry.msgstr_line, format!("{}\n", entry.msgstr));
399            repaired += 1;
400        }
401    }
402    if repaired > 0 {
403        println!("==> Repairing {repaired} entries missing trailing \\n");
404    }
405
406    // Repair entries where the model previously leaked its instructions instead of translating.
407    // Recover the real translation from the response tail when possible, otherwise clear to ""
408    // so the entry gets re-translated on this run.
409    let mut leak_recovered = 0;
410    let mut leak_blanked = 0;
411    let mut lines: Vec<String> = lines;
412    for entry in &entries {
413        if entry.msgstr.is_empty() {
414            continue;
415        }
416        match check_for_leak(&entry.msgid, &entry.msgstr) {
417            LeakCheck::Clean => {}
418            LeakCheck::Recovered(r) => {
419                lines[entry.msgstr_line] = format!("msgstr \"{}\"", encode_po_string(&r));
420                leak_recovered += 1;
421            }
422            LeakCheck::Unrecoverable => {
423                lines[entry.msgstr_line] = "msgstr \"\"".to_string();
424                leak_blanked += 1;
425            }
426        }
427    }
428    if leak_recovered + leak_blanked > 0 {
429        println!(
430            "==> Leak repair: {leak_recovered} recovered, {leak_blanked} cleared for re-translation"
431        );
432    }
433    // Re-parse with repaired lines
434    let entries = if leak_recovered + leak_blanked > 0 {
435        parse_po(&lines)
436    } else {
437        entries
438    };
439
440    // Entries with empty msgstr need AI translation.
441    // Fuzzy entries already have a translation — accept it as-is, just drop the flag.
442    // --force retranslates everything regardless.
443    let to_translate: Vec<&Entry> = entries
444        .iter()
445        .filter(|e| args.force || e.msgstr.is_empty())
446        .collect();
447
448    let to_accept: Vec<&Entry> = entries
449        .iter()
450        .filter(|e| !args.force && e.fuzzy_line.is_some() && !e.msgstr.is_empty())
451        .collect();
452
453    if to_translate.is_empty() && to_accept.is_empty() && repaired == 0 {
454        println!("Nothing to translate.");
455        return Ok(());
456    }
457
458    println!(
459        "==> {} to translate, {} fuzzy accepted as-is, model_provider={}, model={}",
460        to_translate.len(),
461        to_accept.len(),
462        args.model_provider,
463        model,
464    );
465
466    let total = to_translate.len();
467    let total_chunks = total.div_ceil(args.batch).max(1);
468
469    let log_path = args
470        .log_failures
471        .clone()
472        .unwrap_or_else(|| args.po.with_extension("failures.log"));
473    let failure_log = FailureLog::open(&log_path)?;
474    println!("==> Logging failures to {}", log_path.display());
475
476    for (chunk_idx, chunk) in to_translate.chunks(args.batch).enumerate() {
477        let msgids: Vec<&str> = chunk.iter().map(|e| e.msgid.as_str()).collect();
478        println!(
479            "==> Chunk {}/{total_chunks} ({} entries)",
480            chunk_idx + 1,
481            chunk.len()
482        );
483
484        match translate_batch(provider.as_ref(), &model, &args.locale, &msgids).await {
485            Ok(translated) => {
486                for (entry, text) in chunk.iter().zip(translated.iter()) {
487                    // If msgid ends with \n, msgstr must too — gettext requires it.
488                    let text = if entry.msgid.ends_with('\n') && !text.ends_with('\n') {
489                        format!("{text}\n")
490                    } else {
491                        text.clone()
492                    };
493                    translations.insert(entry.msgstr_line, text);
494                }
495                write_po(
496                    &lines,
497                    &raw,
498                    &translations,
499                    &to_translate,
500                    &to_accept,
501                    &args.po,
502                )?;
503            }
504            Err(f) => {
505                let source_joined = msgids.join(" | ");
506                eprintln!(
507                    "  warning: chunk {} failed: {}\n    source: {:?}\n    response: {:?}",
508                    chunk_idx + 1,
509                    f.err,
510                    source_joined,
511                    f.raw_response
512                );
513                failure_log.record(chunk_idx + 1, &source_joined, &f.raw_response, &f.err);
514            }
515        }
516    }
517
518    // Final write — handles to_accept fuzzy removals even when to_translate is empty
519    write_po(
520        &lines,
521        &raw,
522        &translations,
523        &to_translate,
524        &to_accept,
525        &args.po,
526    )?;
527    println!(
528        "==> Done: {}/{total} entries translated.",
529        translations.len()
530    );
531    Ok(())
532}
