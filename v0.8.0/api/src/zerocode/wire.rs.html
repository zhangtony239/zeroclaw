1//! Hand-maintained mirrors for every type that crosses the JSON-RPC
2//! wire between `zerocode` and the ZeroClaw daemon.
3//!
4//! These mirrors exist so `apps/zerocode/Cargo.toml` carries zero
5//! workspace dependencies in `[dependencies]`. The TUI talks JSON-RPC
6//! to whatever daemon is at the configured address; the wire shape is
7//! the contract, not a shared Rust type.
8//!
9//! Drift between these mirrors and the canonical workspace types is
10//! caught by `apps/zerocode/tests/wire_drift.rs`, which pulls the
11//! canonical types via `[dev-dependencies]` and asserts JSON-byte
12//! equality after a serialize / deserialize / re-serialize cycle.
13//!
14//! Some mirrors here are unused by the running TUI today — they
15//! exist to lock the wire contract for every type the daemon emits
16//! so that adding a new use-site in the TUI doesn't have to re-derive
17//! the shape from scratch and risk drift.
18#![allow(dead_code)]
19

210/// Schema field-kind tag mirroring `zeroclaw_config::traits::PropKind`.
211/// Carries the canonical eight variants — adding one in the schema
212/// must mirror here too; `wire_drift::prop_kind_variants_round_trip`
213/// fails when they diverge.
214#[derive(Debug, Clone, Copy, Serialize, Deserialize, PartialEq, Eq)]
215#[serde(rename_all = "snake_case")]
216pub enum PropKind {
217    String,
218    Bool,
219    Integer,
220    Float,
221    Enum,
222    StringArray,
223    ObjectArray,
224    Object,
225}
226
227impl PropKind {
228    /// Wire name string, matching the canonical
229    /// `zeroclaw_config::traits::PropKind::wire_name`. Used by the
230    /// config explorer to render type hints.
231    pub fn wire_name(self) -> &'static str {
232        match self {
233            Self::String => "string",
234            Self::Bool => "bool",
235            Self::Integer => "integer",
236            Self::Float => "float",
237            Self::Enum => "enum",
238            Self::StringArray => "string_array",
239            Self::ObjectArray => "object_array",
240            Self::Object => "object",
241        }
242    }
243}

245/// Schema-defined config tab grouping. Mirrors
246/// `zeroclaw_config::traits::ConfigTab`. `Default` is `None` — the
247/// "flat list, no tab bar" state.
248#[derive(Debug, Clone, Copy, Serialize, Deserialize, PartialEq, Eq, Hash, Default)]
249pub enum ConfigTab {
250    #[default]
251    None,
252    Connection,
253    Advanced,
254    Model,
255    Behavior,
256    General,
257    Channels,
258    Providers,
259    Bundles,
260    Cron,
261    Tuning,
262    Workspace,
263    Memory,
264    PeerGroups,
265    Personality,
266    Settings,
267    Servers,
268    Limits,
269    Costs,
270    Skills,
271    Aliases,
272}
273
274impl ConfigTab {
275    pub fn label(self) -> &'static str {
276        match self {
277            Self::None => "",
278            Self::Connection => "Connection",
279            Self::Advanced => "Advanced",
280            Self::Model => "Model",
281            Self::Behavior => "Behavior",
282            Self::General => "General",
283            Self::Channels => "Channels",
284            Self::Providers => "Providers",
285            Self::Bundles => "Bundles",
286            Self::Cron => "Cron",
287            Self::Tuning => "Tuning",
288            Self::Workspace => "Workspace",
289            Self::Memory => "Memory",
290            Self::PeerGroups => "Peer Groups",
291            Self::Personality => "Personality",
292            Self::Settings => "Settings",
293            Self::Servers => "Servers",
294            Self::Limits => "Limits",
295            Self::Costs => "Costs",
296            Self::Skills => "Skills",
297            Self::Aliases => "Aliases",
298        }
299    }
300
301    pub fn is_none(&self) -> bool {
302        matches!(self, Self::None)
303    }
304}

306impl std::fmt::Display for ConfigTab {
307    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
308        f.write_str(self.label())
309    }
310}
