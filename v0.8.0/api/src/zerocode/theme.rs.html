1//! ZeroClaw TUI colour palette and style helpers.
2//!
3//! Shared between the onboarding UI (lib target) and the main chat TUI (binary
4//! target). Not every helper is used by both targets.

24/// "Inherit shell" — uses the terminal's own default colours. Every
25/// role is `Color::Reset`, and the app-level backdrop skips painting
26/// when `background` is `Reset`, so a user's tuned terminal palette
27/// shows through untouched.
28const TERMINAL: Theme = Theme {
29    title: Color::Reset,
30    heading: Color::Reset,
31    body: Color::Reset,
32    dim: Color::Reset,
33    accent: Color::Reset,
34    warn: Color::Reset,
35    selection_bg: Color::Reset,
36    tool: Color::Reset,
37    background: Color::Reset,
38};

40// The named preset palettes are generated at build time from
41// `web/src/contexts/themes.json`, the single source of truth shared with the
42// React dashboard and mdBook docs. See `build.rs` for the var→role mapping.
43// `TERMINAL` is authored here because it is the inherit-shell sentinel, not a
44// real palette.
45include!(concat!(env!("OUT_DIR"), "/theme_presets.rs"));
46
47pub(crate) const DEFAULT_THEME_NAME: &str = "icy_blue";
48
49/// The authored inherit-shell sentinel. Real palettes come from
50/// `GENERATED_THEMES`.
51const AUTHORED_THEMES: &[(&str, Theme)] = &[("terminal", TERMINAL)];
52
53/// Every named preset: the authored pair followed by the generated registry
54/// themes. The single iteration point both lookup helpers walk.
55fn all_themes() -> impl Iterator<Item = &'static (&'static str, Theme)> {
56    AUTHORED_THEMES.iter().chain(GENERATED_THEMES.iter())
57}
58
59pub(crate) fn theme_by_name(name: &str) -> Option<Theme> {
60    all_themes().find_map(|(n, t)| (*n == name).then_some(*t))
61}
62
63pub(crate) fn theme_names() -> impl Iterator<Item = &'static str> {
64    all_themes().map(|(n, _)| *n)
65}

67static ACTIVE: LazyLock<RwLock<Theme>> = LazyLock::new(|| RwLock::new(default_theme()));
68
69/// Per-agent theme overrides, keyed by agent alias. A process-global registry
70/// mirroring `ACTIVE`: the Config pane writes here on assign/clear (live, no
71/// restart), and the app loop reads it each frame to tint the Code/Chat pane
72/// for the focused agent. Lazily created so the static stays const-initialised.
73static AGENT_OVERRIDES: RwLock<Option<std::collections::HashMap<String, Theme>>> =
74    RwLock::new(None);
75
76/// Replace the whole agent-override registry (loaded once at startup).
77pub(crate) fn set_agent_overrides(map: std::collections::HashMap<String, Theme>) {
78    if let Ok(mut guard) = AGENT_OVERRIDES.write() {
79        *guard = Some(map);
80    }
81}
82
83/// Insert or replace one agent's override (live assign from the Config pane).
84pub(crate) fn set_agent_override(alias: &str, theme: Theme) {
85    if let Ok(mut guard) = AGENT_OVERRIDES.write() {
86        guard
87            .get_or_insert_with(std::collections::HashMap::new)
88            .insert(alias.to_string(), theme);
89    }
90}
91
92/// Remove one agent's override (live clear from the Config pane).
93pub(crate) fn clear_agent_override(alias: &str) {
94    if let Ok(mut guard) = AGENT_OVERRIDES.write()
95        && let Some(map) = guard.as_mut()
96    {
97        map.remove(alias);
98    }
99}
100
101/// The override palette for `alias`, if any. Read each frame by the app loop.
102pub(crate) fn agent_override(alias: &str) -> Option<Theme> {
103    AGENT_OVERRIDES
104        .read()
105        .ok()
106        .and_then(|g| g.as_ref().and_then(|m| m.get(alias).copied()))
107}
108
109pub(crate) fn set_active(theme: Theme) {
110    if let Ok(mut guard) = ACTIVE.write() {
111        *guard = theme;
112    }
113}

115pub(crate) fn active() -> Theme {
116    let raw = active_raw();
117    Theme {
118        title: crate::color_depth::downgrade(raw.title),
119        heading: crate::color_depth::downgrade(raw.heading),
120        body: crate::color_depth::downgrade(raw.body),
121        dim: crate::color_depth::downgrade(raw.dim),
122        accent: crate::color_depth::downgrade(raw.accent),
123        warn: crate::color_depth::downgrade(raw.warn),
124        selection_bg: crate::color_depth::downgrade(raw.selection_bg),
125        tool: crate::color_depth::downgrade(raw.tool),
126        background: crate::color_depth::downgrade(raw.background),
127    }
128}
129
130/// The stored palette without colour-depth downgrade. Used to snapshot and
131/// restore the base theme around a per-frame override swap: `set_active` stores
132/// raw RGB, so save/restore must round-trip the raw value, not the downgraded
133/// one `active()` returns.
134pub(crate) fn active_raw() -> Theme {
135    ACTIVE
136        .read()
137        .map(|g| *g)
138        .unwrap_or_else(|_| default_theme())
139}

166/// Full-screen backdrop style painting the theme background. Returns
167/// `None` when the theme inherits the terminal (`background == Reset`),
168/// so the caller can skip the backdrop entirely.
169pub(crate) fn backdrop_style() -> Option<Style> {
170    let bg = active().background;
171    if bg == Color::Reset {
172        None
173    } else {
174        Some(Style::default().bg(bg))
175    }
176}

298/// A bordered content panel with a themed border and an optional themed
299/// title. The single source of truth for pane chrome so borders never
300/// drift back to the terminal default.
301pub(crate) fn panel_block(title: &str) -> ratatui::widgets::Block<'static> {
302    let mut block = ratatui::widgets::Block::default()
303        .borders(ratatui::widgets::Borders::ALL)
304        .border_style(dim_style());
305    if !title.is_empty() {
306        block = block.title(ratatui::text::Span::styled(
307            title.to_string(),
308            title_style(),
309        ));
310    }
311    block
312}
313
314/// A modal/overlay panel: themed accent border, bold accent title, and a
315/// solid theme-background fill so the modal interior never shows through
316/// to the terminal default after a `Clear`.
317pub(crate) fn modal_block(title: &str) -> ratatui::widgets::Block<'static> {
318    let mut block = ratatui::widgets::Block::default()
319        .borders(ratatui::widgets::Borders::ALL)
320        .border_style(accent_style())
321        .style(fill_style());
322    if !title.is_empty() {
323        block = block.title(ratatui::text::Span::styled(
324            title.to_string(),
325            accent_style(),
326        ));
327    }
328    block
329}
330
331/// Solid panel fill: theme body foreground on the theme background. Used
332/// to back modals so their interior matches the active palette instead of
333/// the terminal default. Falls back to body-only when the theme inherits
334/// the terminal (`background == Reset`).
335pub(crate) fn fill_style() -> Style {
336    let t = active();
337    let s = Style::default().fg(t.body);
338    if t.background == Color::Reset {
339        s
340    } else {
341        s.bg(t.background)
342    }
343}

349    #[test]
350    fn icy_blue_rgb_unchanged() {
351        let t = theme_by_name("icy_blue").expect("icy_blue registered");
352        assert_eq!(t.title, Color::Rgb(100, 200, 255));
353        assert_eq!(t.heading, Color::Rgb(140, 230, 255));
354        assert_eq!(t.body, Color::Rgb(220, 240, 255));
355        assert_eq!(t.dim, Color::Rgb(80, 130, 170));
356        assert_eq!(t.accent, Color::Rgb(255, 100, 80));
357        assert_eq!(t.warn, Color::Rgb(255, 220, 80));
358        assert_eq!(t.selection_bg, Color::Rgb(30, 60, 100));
359        assert_eq!(t.tool, Color::Rgb(180, 140, 255));
360    }

372    #[test]
373    fn set_active_swaps_palette() {
374        // `active()` routes through the colour-depth downgrade; assert on the
375        // stored palette via the registry lookup so the test is independent of
376        // the terminal depth detected in the test environment.
377        set_active(theme_by_name("nord_dark").unwrap());
378        assert_eq!(
379            theme_by_name("nord_dark").unwrap().title,
380            Color::Rgb(136, 192, 208)
381        );
382        set_active(theme_by_name("icy_blue").unwrap());
383        assert_eq!(
384            theme_by_name("icy_blue").unwrap().title,
385            Color::Rgb(100, 200, 255)
386        );
387    }

411    #[test]
412    fn theme_names_are_snake_case() {
413        let ok = |s: &str| {
414            !s.is_empty()
415                && s.chars()
416                    .all(|c| c.is_ascii_lowercase() || c.is_ascii_digit() || c == '_')
417                && !s.starts_with('_')
418                && !s.ends_with('_')
419        };
420        for name in theme_names() {
421            assert!(ok(name), "theme name '{name}' is not snake_case");
422        }
423        assert!(ok(DEFAULT_THEME_NAME), "default theme name not snake_case");
424    }
