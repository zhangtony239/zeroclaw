3/// A single help entry: one or more keys that trigger the same action.
4///
5/// The renderer joins keys with " / " so you don't have to format manually.
6/// An entry with all-empty keys/action renders as a blank spacer row.
7///
8/// Keys are owned `String`s so callers can pass live chord displays resolved
9/// from the keymap (`Action::Variant.resolved()[..].display()`) instead of
10/// hardcoded literals — the help always reflects the actual bindings, including
11/// user overrides.
12#[derive(Debug, Clone, Default)]
13pub struct HelpEntry {
14    /// Keys that trigger this action, e.g. ["↑", "k"]. Rendered labels,
15    /// owned so registry-derived chord labels (`Chord::display`) can be
16    /// used alongside static literals.
17    pub keys: Vec<String>,
18    /// Human-readable description of the action.
19    pub action: String,
20}
21
22impl HelpEntry {
23    pub fn new<K: Into<String>>(
24        keys: impl IntoIterator<Item = K>,
25        action: impl Into<String>,
26    ) -> Self {
27        Self {
28            keys: keys.into_iter().map(Into::into).collect(),
29            action: action.into(),
30        }
31    }
32
33    /// Convenience: single key.
34    pub fn key(key: impl Into<String>, action: impl Into<String>) -> Self {
35        Self {
36            keys: vec![key.into()],
37            action: action.into(),
38        }
39    }
40
41    /// Blank spacer row.
42    pub fn spacer() -> Self {
43        Self {
44            keys: vec![],
45            action: String::new(),
46        }
47    }
48
49    /// Keyless description row (no key column, just text).
50    pub fn desc(action: impl Into<String>) -> Self {
51        Self {
52            keys: vec![],
53            action: action.into(),
54        }
55    }
56
57    /// Format keys as "↑ / k" etc.
58    pub fn key_str(&self) -> String {
59        self.keys.join(" / ")
60    }
61}

63/// A node in the help context tree.
64///
65/// The help system cascades: Pane → Tab → Widget (or Screen → Tab → Widget
66/// for the config pane). Each level produces one `HelpNode`. The modal renders
67/// them depth-first:
68///
69///   [title]
70///   [description, soft-wrapped]
71///   key   action
72///   key   action
73///   ── dim separator ──
74///   [child title]
75///   ...
76///
77/// Any field may be empty/None — the renderer skips it cleanly.
78#[derive(Debug, Clone, Default)]
79pub struct HelpNode {
80    /// Short label shown as a dim section header (e.g. "Tab", "Widget"). None = no header.
81    pub title: Option<String>,
82    /// Prose description shown above the keybindings, soft-wrapped to modal width.
83    pub description: Option<String>,
84    /// Keybinding entries for this level.
85    pub entries: Vec<HelpEntry>,
86    /// Child nodes (tab-level, widget-level, etc.).
87    pub children: Vec<HelpNode>,
88}
89
90impl HelpNode {
91    /// Leaf node with just keybindings.
92    pub fn entries(entries: Vec<HelpEntry>) -> Self {
93        Self {
94            entries,
95            ..Default::default()
96        }
97    }
98
99    /// Consume self and append a child node, returning the modified node.
100    pub fn with_child(mut self, child: HelpNode) -> Self {
101        self.children.push(child);
102        self
103    }
104}
105
106/// Implement this on any struct that can contribute to the help modal.
107pub trait HelpContext {
108    fn help_context(&self) -> HelpNode;
109}

119/// A one-row context-window usage bar.
120///
121/// Renders left-aligned into whatever `Rect` you hand it.
122/// Returns `None` from `widget()` when there is nothing to show.
123pub struct CtxBar {
124    pub input_tokens: Option<u64>,
125    pub max_tokens: Option<u64>,
126}
127
128impl CtxBar {
129    pub fn new(input_tokens: Option<u64>, max_tokens: Option<u64>) -> Self {
130        Self {
131            input_tokens,
132            max_tokens,
133        }
134    }
135
136    /// `true` when there is something worth rendering.
137    pub fn has_content(&self) -> bool {
138        self.input_tokens.is_some() || self.max_tokens.is_some()
139    }
140
141    /// Build a `Paragraph` widget, or `None` if there is nothing to show.
142    pub fn widget(&self) -> Option<Paragraph<'static>> {
143        let (text, pct_opt) = match (self.input_tokens, self.max_tokens) {
144            (Some(used), Some(max)) if max > 0 => {
145                let pct = (used as f64 / max as f64 * 100.0).min(100.0);
146                let bar_width: usize = 16;
147                let filled = ((pct / 100.0) * bar_width as f64).round() as usize;
148                let empty = bar_width.saturating_sub(filled);
149                let bar = format!(
150                    "[{}{}]",
151                    "\u{2588}".repeat(filled),
152                    "\u{2591}".repeat(empty)
153                );
154                let label = format!(
155                    " ctx: {:>7} / {:>7}  {}  {:.0}%",
156                    fmt_tokens(used),
157                    fmt_tokens(max),
158                    bar,
159                    pct,
160                );
161                (label, Some(pct))
162            }
163            (Some(used), None) => {
164                let label = format!(" ctx: {} tokens", fmt_tokens(used));
165                (label, None)
166            }
167            _ => return None,
168        };
169
170        let color = match pct_opt {
171            Some(p) if p >= 90.0 => Color::Red,
172            Some(p) if p >= 75.0 => Color::Yellow,
173            _ => Color::DarkGray,
174        };
175
176        Some(Paragraph::new(Line::from(Span::styled(
177            text,
178            Style::default().fg(color),
179        ))))
180    }
181}
182
183fn fmt_tokens(n: u64) -> String {
184    let s = n.to_string();
185    let mut out = String::with_capacity(s.len() + s.len() / 3);
186    for (i, ch) in s.chars().rev().enumerate() {
187        if i > 0 && i % 3 == 0 {
188            out.push(',');
189        }
190        out.push(ch);
191    }
192    out.chars().rev().collect()
193}

199/// How long an info message stays on the bar before it auto-clears. Named so
200/// the timeout is not a bare literal at the clear site.
201pub const INFO_BAR_TTL: Duration = Duration::from_secs(10);
202
203/// Severity of an info-bar message. Drives the colour; never matched on as a
204/// string.
205#[derive(Debug, Clone, Copy, PartialEq, Eq)]
206pub enum InfoKind {
207    /// Neutral operational note (e.g. "Fetching models for anthropic…").
208    Info,
209    /// A completed action worth confirming (e.g. "Model switched to …").
210    Note,
211    /// A failure the user should see (e.g. an RPC error).
212    Error,
213}
214
215/// A single user-facing message shown on the info bar. Owned by the app layer
216/// as `Option<InfoMessage>`; `None` means the bar is hidden. `set_at` drives the
217/// [`INFO_BAR_TTL`] auto-clear in the app tick loop.
218#[derive(Debug, Clone)]
219pub struct InfoMessage {
220    pub kind: InfoKind,
221    pub text: String,
222    pub set_at: Instant,
223}
224
225impl InfoMessage {
226    pub fn new(kind: InfoKind, text: impl Into<String>) -> Self {
227        Self {
228            kind,
229            text: text.into(),
230            set_at: Instant::now(),
231        }
232    }
233
234    pub fn info(text: impl Into<String>) -> Self {
235        Self::new(InfoKind::Info, text)
236    }
237
238    pub fn note(text: impl Into<String>) -> Self {
239        Self::new(InfoKind::Note, text)
240    }
241
242    pub fn error(text: impl Into<String>) -> Self {
243        Self::new(InfoKind::Error, text)
244    }
245
246    /// `true` once the message has been visible for at least [`INFO_BAR_TTL`].
247    pub fn is_expired(&self) -> bool {
248        self.set_at.elapsed() >= INFO_BAR_TTL
249    }
250}

252/// A one-row, single-line info bar. Renders the current message truncated to the
253/// available width; stores the full text untruncated so a wider window shows
254/// more without any state change.
255pub struct InfoBar<'a> {
256    message: Option<&'a InfoMessage>,
257}
258
259impl<'a> InfoBar<'a> {
260    pub fn new(message: Option<&'a InfoMessage>) -> Self {
261        Self { message }
262    }
263
264    pub fn has_content(&self) -> bool {
265        self.message.is_some()
266    }
267
268    /// Build the `Paragraph`, or `None` when there is no message. `width` is the
269    /// available column count; the text is truncated (with an ellipsis) to fit.
270    pub fn widget(&self, width: usize) -> Option<Paragraph<'static>> {
271        let msg = self.message?;
272        let palette = crate::theme::active();
273        let color = match msg.kind {
274            InfoKind::Info => palette.dim,
275            InfoKind::Note => palette.accent,
276            InfoKind::Error => palette.warn,
277        };
278        let text = truncate_to_width(&msg.text, width);
279        Some(Paragraph::new(Line::from(Span::styled(
280            text,
281            Style::default().fg(color),
282        ))))
283    }
284}

286/// Truncate `s` to at most `width` display columns, appending an ellipsis when
287/// it overflows. Approximates width by `char` count — adequate for the
288/// single-line status text the info bar carries.
289fn truncate_to_width(s: &str, width: usize) -> String {
290    if width == 0 {
291        return String::new();
292    }
293    if s.chars().count() <= width {
294        return s.to_string();
295    }
296    if width == 1 {
297        return "\u{2026}".to_string();
298    }
299    let keep = width - 1;
300    let mut out: String = s.chars().take(keep).collect();
301    out.push('\u{2026}');
302    out
303}

313/// A reusable centered modal list picker over a `Vec<String>`. Owns its items,
314/// cursor, and an optional title; renders a bordered, scrollable list with the
315/// highlighted row styled. The caller owns the state (see [`PickerState`]) and
316/// keys; this type is the renderer plus cursor-movement helpers so other
317/// surfaces can reuse it.
318pub struct PickerModal<'a> {
319    title: &'a str,
320    items: &'a [String],
321    cursor: usize,
322}
323
324impl<'a> PickerModal<'a> {
325    pub fn new(title: &'a str, items: &'a [String], cursor: usize) -> Self {
326        Self {
327            title,
328            items,
329            cursor,
330        }
331    }
332
333    /// Render the modal centered within `area`. No-op when there are no items.
334    pub fn render(&self, frame: &mut Frame, area: Rect) {
335        if self.items.is_empty() {
336            return;
337        }
338
339        // Width: longest item (clamped), plus borders/padding. Height: one row
340        // per item plus borders, clamped to the available area.
341        let longest = self
342            .items
343            .iter()
344            .map(|s| s.chars().count())
345            .max()
346            .unwrap_or(0)
347            .max(self.title.chars().count());
348        let inner_w = longest + 2; // 1 col padding each side
349        let box_w = (inner_w + 2).clamp(12, area.width as usize) as u16;
350        let box_h = (self.items.len() + 2).clamp(3, area.height as usize) as u16;
351
352        let x = area.x + area.width.saturating_sub(box_w) / 2;
353        let y = area.y + area.height.saturating_sub(box_h) / 2;
354        let modal_rect = Rect::new(x, y, box_w, box_h);
355
356        frame.render_widget(Clear, modal_rect);
357
358        let block = Block::default()
359            .borders(Borders::ALL)
360            .border_style(crate::theme::overlay_border_style())
361            .style(crate::theme::fill_style())
362            .title(Span::styled(
363                format!(" {} ", self.title),
364                crate::theme::heading_style(),
365            ));
366
367        let items: Vec<ListItem> = self
368            .items
369            .iter()
370            .enumerate()
371            .map(|(i, label)| {
372                let style = if i == self.cursor {
373                    crate::theme::selected_style()
374                } else {
375                    crate::theme::body_style()
376                };
377                ListItem::new(Span::styled(label.clone(), style))
378            })
379            .collect();
380
381        let mut list_state = ListState::default();
382        list_state.select(Some(self.cursor.min(self.items.len().saturating_sub(1))));
383
384        let list = List::new(items)
385            .block(block)
386            .highlight_style(crate::theme::selected_style());
387
388        frame.render_stateful_widget(list, modal_rect, &mut list_state);
389    }
390}

392/// Owned state for a [`PickerModal`]: the items and the current cursor. Caller
393/// holds this and drives it with the movement helpers; the widget borrows it for
394/// rendering. Reusable by any surface that needs a string picker.
395#[derive(Debug, Clone, Default)]
396pub struct PickerState {
397    pub items: Vec<String>,
398    pub cursor: usize,
399}
400
401impl PickerState {
402    /// Build a picker over `items`, pre-selecting `default` when present (else
403    /// the first row).
404    pub fn new(items: Vec<String>, default: Option<&str>) -> Self {
405        let cursor = default
406            .and_then(|d| items.iter().position(|i| i == d))
407            .unwrap_or(0);
408        Self { items, cursor }
409    }
410
411    pub fn is_empty(&self) -> bool {
412        self.items.is_empty()
413    }
414
415    pub fn move_up(&mut self) {
416        self.cursor = self.cursor.saturating_sub(1);
417    }
418
419    pub fn move_down(&mut self) {
420        if self.cursor + 1 < self.items.len() {
421            self.cursor += 1;
422        }
423    }
424
425    /// The currently highlighted value, if any.
426    pub fn selected(&self) -> Option<&str> {
427        self.items.get(self.cursor).map(String::as_str)
428    }
429}
