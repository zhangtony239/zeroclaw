1//! Terminal colour-depth detection and RGB down-conversion.
2//!
3//! ratatui emits `Color::Rgb` as 24-bit `\e[38;2;…m` SGR sequences
4//! unconditionally. Terminals that cap at 256 colours (older macOS
5//! Terminal.app, and tmux/screen advertising `screen-256color`) either
6//! ignore or mangle those sequences, producing the washed-out / garbled
7//! palette users hit over SSH+tmux. Detecting the supported depth once at
8//! startup and snapping every themed `Rgb` to the nearest xterm-256 (or
9//! ANSI-16) index keeps the palette legible everywhere instead of forcing
10//! the user to fall back to the uncoloured `terminal` theme.
11
12use std::sync::OnceLock;
13
14use ratatui::style::Color;
15
16#[derive(Debug, Clone, Copy, PartialEq, Eq)]
17pub(crate) enum ColorDepth {
18    /// 16 ANSI colours only (very old terminals, `TERM=…-16color`, `dumb`).
19    Ansi16,
20    /// xterm-256 palette.
21    Ansi256,
22    /// 24-bit RGB.
23    TrueColor,
24}
25
26static DEPTH: OnceLock<ColorDepth> = OnceLock::new();
27
28/// The terminal's colour depth, detected once from the environment on first
29/// use and memoised. Detection honours the `ZEROCODE_COLOR` override, then
30/// falls back to truecolor for every terminal except a `dumb`/`ansi`/`-16color`
31/// / empty `TERM`, which gets ANSI-16. Lazy initialisation means callers never
32/// need an explicit startup hook — the first themed colour to render triggers
33/// it.
34pub(crate) fn active() -> ColorDepth {
35    *DEPTH.get_or_init(|| {
36        detect_from_env(
37            std::env::var("ZEROCODE_COLOR").ok().as_deref(),
38            std::env::var("COLORTERM").ok().as_deref(),
39            std::env::var("TERM").ok().as_deref(),
40            std::env::var("TERM_PROGRAM").ok().as_deref(),
41            std::env::var("TMUX").is_ok(),
42        )
43    })
44}
45
46/// Pure detection core, split out for testing. `override_var` is the value
47/// of `ZEROCODE_COLOR` (`truecolor`/`24bit`, `256`, `16`/`ansi`), which
48/// short-circuits auto-detection when set to a recognised value.
49fn detect_from_env(
50    override_var: Option<&str>,
51    colorterm: Option<&str>,
52    term: Option<&str>,
53    term_program: Option<&str>,
54    in_tmux: bool,
55) -> ColorDepth {
56    if let Some(forced) = override_var.and_then(parse_override) {
57        return forced;
58    }
59
60    let term = term.unwrap_or("");
61
62    // Genuinely capability-free terminals: no TERM, or an explicit dumb/8-16
63    // colour terminfo. Only these get the lossy ANSI-16 path.
64    if term.is_empty() || term == "dumb" || term == "ansi" {
65        return ColorDepth::Ansi16;
66    }
67    if term.ends_with("-16color") {
68        return ColorDepth::Ansi16;
69    }
70
71    // macOS Terminal.app genuinely lacks 24-bit colour — it caps at xterm-256.
72    // Emitting truecolor there produces wrong colours, so cap it at 256. It
73    // identifies itself unambiguously via TERM_PROGRAM=Apple_Terminal (iTerm,
74    // WezTerm, kitty, Ghostty, etc. all do truecolor and fall through).
75    if matches!(term_program, Some(p) if p.eq_ignore_ascii_case("Apple_Terminal")) {
76        return ColorDepth::Ansi256;
77    }
78
79    // Remaining detection inputs do not gate the result; truecolor is the
80    // universal default below.
81    let _ = (colorterm, in_tmux);
82
83    // Default to truecolor. crossterm emits colours verbatim — a 24-bit
84    // `\e[38;2;R;G;Bm` SGR — and that sequence passes through terminal
85    // multiplexers (tmux/screen) uncorrupted even when their TERM advertises a
86    // low-colour terminfo. 256-indexed escapes do NOT: a `screen`/`tmux`
87    // terminfo down-translates `\e[38;5;Nm` to the nearest of the host's 16
88    // ANSI palette slots, collapsing every theme colour onto whatever those
89    // slots happen to be (often near-monochrome). Truecolor sidesteps that
90    // translation, so it is the most faithful universal output for every modern
91    // terminal. The `ZEROCODE_COLOR` override (handled above) forces 256/16 for
92    // the rare terminal that genuinely mishandles 24-bit SGR.
93    ColorDepth::TrueColor
94}
95
96fn parse_override(v: &str) -> Option<ColorDepth> {
97    let v = v.trim();
98    if v.eq_ignore_ascii_case("truecolor") || v.eq_ignore_ascii_case("24bit") || v == "24" {
99        Some(ColorDepth::TrueColor)
100    } else if v == "256" {
101        Some(ColorDepth::Ansi256)
102    } else if v == "16" || v.eq_ignore_ascii_case("ansi") {
103        Some(ColorDepth::Ansi16)
104    } else {
105        None
106    }
107}

109/// Snap a colour to what the active terminal can render. `Rgb` is passed
110/// through untouched on truecolor terminals, down-converted to the nearest
111/// xterm-256 index on 256-colour terminals, and to the nearest ANSI-16
112/// colour otherwise. Non-`Rgb` colours (named, indexed, `Reset`) are
113/// already renderable everywhere and pass through.
114pub(crate) fn downgrade(color: Color) -> Color {
115    downgrade_at(color, active())
116}
117
118fn downgrade_at(color: Color, depth: ColorDepth) -> Color {
119    let Color::Rgb(r, g, b) = color else {
120        return color;
121    };
122    match depth {
123        ColorDepth::TrueColor => color,
124        ColorDepth::Ansi256 => Color::Indexed(rgb_to_xterm256(r, g, b)),
125        ColorDepth::Ansi16 => rgb_to_ansi16(r, g, b),
126    }
127}
128
129/// Nearest xterm-256 index for an RGB triple. Considers both the 6×6×6
130/// colour cube (indices 16–231) and the 24-step grey ramp (232–255), and
131/// returns whichever is closer, so neutral theme colours map to crisp greys
132/// rather than muddy cube entries.
133fn rgb_to_xterm256(r: u8, g: u8, b: u8) -> u8 {
134    let cube_idx = |v: u8| -> u8 {
135        // xterm cube steps: 0, 95, 135, 175, 215, 255.
136        if v < 48 {
137            0
138        } else if v < 115 {
139            1
140        } else {
141            ((v as u16 - 35) / 40) as u8
142        }
143    };
144    let cube_val = |i: u8| -> u8 {
145        if i == 0 {
146            0
147        } else {
148            (55 + i as u16 * 40) as u8
149        }
150    };
151
152    let ri = cube_idx(r);
153    let gi = cube_idx(g);
154    let bi = cube_idx(b);
155    let cube_index = 16 + 36 * ri + 6 * gi + bi;
156    let cube_dist = dist(r, g, b, cube_val(ri), cube_val(gi), cube_val(bi));
157
158    // Grey ramp: 232 + n maps to 8 + n*10 for n in 0..24.
159    let grey_avg = ((r as u16 + g as u16 + b as u16) / 3) as u8;
160    let grey_n = if grey_avg < 8 {
161        0
162    } else if grey_avg > 238 {
163        23
164    } else {
165        ((grey_avg as u16 - 8) / 10) as u8
166    };
167    let grey_val = (8 + grey_n as u16 * 10) as u8;
168    let grey_dist = dist(r, g, b, grey_val, grey_val, grey_val);
169
170    if grey_dist < cube_dist {
171        232 + grey_n
172    } else {
173        cube_index
174    }
175}

177/// Nearest of the 16 ANSI colours, returned as a ratatui named `Color`.
178fn rgb_to_ansi16(r: u8, g: u8, b: u8) -> Color {
179    const PALETTE: [(u8, u8, u8, Color); 16] = [
180        (0, 0, 0, Color::Black),
181        (128, 0, 0, Color::Red),
182        (0, 128, 0, Color::Green),
183        (128, 128, 0, Color::Yellow),
184        (0, 0, 128, Color::Blue),
185        (128, 0, 128, Color::Magenta),
186        (0, 128, 128, Color::Cyan),
187        (192, 192, 192, Color::Gray),
188        (128, 128, 128, Color::DarkGray),
189        (255, 0, 0, Color::LightRed),
190        (0, 255, 0, Color::LightGreen),
191        (255, 255, 0, Color::LightYellow),
192        (0, 0, 255, Color::LightBlue),
193        (255, 0, 255, Color::LightMagenta),
194        (0, 255, 255, Color::LightCyan),
195        (255, 255, 255, Color::White),
196    ];
197    PALETTE
198        .iter()
199        .min_by_key(|(pr, pg, pb, _)| dist(r, g, b, *pr, *pg, *pb))
200        .map(|(_, _, _, c)| *c)
201        .unwrap_or(Color::Reset)
202}
203
204fn dist(r1: u8, g1: u8, b1: u8, r2: u8, g2: u8, b2: u8) -> u32 {
205    let dr = r1 as i32 - r2 as i32;
206    let dg = g1 as i32 - g2 as i32;
207    let db = b1 as i32 - b2 as i32;
208    (dr * dr + dg * dg + db * db) as u32
209}

215    #[test]
216    fn override_forces_depth() {
217        assert_eq!(
218            detect_from_env(
219                Some("16"),
220                Some("truecolor"),

222                None,
223                false

228            detect_from_env(Some("truecolor"), None, Some("dumb"), None, false),
229            ColorDepth::TrueColor

233            ColorDepth::Ansi256

237    #[test]
238    fn defaults_to_truecolor() {
239        // Truecolor is the universal default: it passes through multiplexers
240        // verbatim, whereas 256-indexed escapes get squashed to the host's 16
241        // ANSI slots by a screen/tmux terminfo. So every capable TERM — plain
242        // xterm, 256color, and a bare multiplexer TERM — emits truecolor.
243        for (term, in_tmux) in [
244            ("xterm", false),
245            ("xterm-256color", false),
246            ("screen", true),
247            ("tmux", true),
248            ("screen-256color", true),
249            ("xterm-kitty", false),
250        ] {
251            assert_eq!(
252                detect_from_env(None, None, Some(term), None, in_tmux),
253                ColorDepth::TrueColor,
254                "TERM={term} in_tmux={in_tmux} should default to truecolor"
255            );
256        }
257    }
