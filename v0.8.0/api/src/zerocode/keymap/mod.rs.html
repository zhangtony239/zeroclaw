1//! Keymap abstraction for zerocode.
2//!
3//! Each leaf action enum carries its own default bindings inline.
4//! Consumers call `ChatTabAction::from_chord(&key)` directly — no
5//! `Keymap` struct, no plumbed argument.
6//!
7//! On darwin, `Chord::matches` translates the `CTRL` modifier to
8//! `SUPER` so Linux's `Ctrl+K` and macOS's `⌘K` resolve identically.
9

19/// Uniform interface over every `keyactions!`-generated enum so generic
20/// code (the keybind surface) can walk variants, names, labels, and
21/// resolved chords without knowing the concrete enum.
22pub trait RebindableActions: Sized + Copy + 'static {
23    fn tag() -> &'static str;
24    fn all() -> &'static [Self];
25    fn key(&self) -> String;
26    fn human_label(&self) -> &'static str;
27    fn defaults(&self) -> Vec<Chord>;
28    fn resolved(&self) -> Vec<Chord>;
29}

31/// Bare chords reserved from user rebinding so structural controls
32/// (cancel/back, confirm, selection toggle) can't be stolen and
33/// soft-lock the TUI. The capture widget rejects these with the reason;
34/// presets validate against the same set.
35pub fn reserved_chords() -> &'static [(Chord, &'static str)] {
36    use crossterm::event::KeyCode;
37    use std::sync::OnceLock;
38    static CELL: OnceLock<Vec<(Chord, &'static str)>> = OnceLock::new();
39    CELL.get_or_init(|| {
40        vec![
41            (Chord::key(KeyCode::Esc), "reserved for cancel / back"),
42            (Chord::key(KeyCode::Enter), "reserved for confirm"),
43            (Chord::char(' '), "reserved for selection toggle"),
44        ]
45    })
46}
47
48/// Whether `chord` is a reserved bare control chord; returns the reason
49/// when it is, so the capture widget can explain the rejection.
50pub fn reserved_reason(chord: &Chord) -> Option<&'static str> {
51    reserved_chords()
52        .iter()
53        .find_map(|(c, reason)| (c == chord).then_some(*reason))
54}
55
56pub fn match_chord<A: Copy>(table: &[(Chord, A)], event: &KeyEvent) -> Option<A> {
57    table
58        .iter()
59        .find_map(|(c, a)| c.matches(event).then_some(*a))
60}

62/// Rendered, OS-aware key labels for an action's currently-resolved
63/// chords (e.g. `["Tab"]`, `["⌘K"]`). Help surfaces use this so the keys
64/// they advertise track the live keybinding registry instead of literals.
65pub fn action_key_labels<A: RebindableActions>(action: A) -> Vec<String> {
66    action.resolved().iter().map(Chord::display).collect()
67}

196    /// Every rebindable enum's TAG and serialized variant names must be
197    /// snake_case — the action-key wire form (`"<tag>.<variant>"`) is
198    /// only valid snake_case, and kebab-case is banned project-wide.
199    #[test]
200    fn tags_and_variant_names_are_snake_case() {
201        fn ok(s: &str) -> bool {
202            !s.is_empty()
203                && s.chars()
204                    .all(|c| c.is_ascii_lowercase() || c.is_ascii_digit() || c == '_')
205                && !s.starts_with('_')
206                && !s.ends_with('_')
207        }
208        fn check<A: RebindableActions>() {
209            assert!(ok(A::tag()), "tag '{}' is not snake_case", A::tag());
210            for v in A::all() {
211                let key = v.key();
212                let variant = key.split_once('.').map(|(_, v)| v).unwrap_or(&key);
213                assert!(ok(variant), "variant '{variant}' is not snake_case");
214            }
215        }
216        check::<GlobalAction>();
217        check::<ChatTabAction>();
218        check::<LogsTabAction>();
219        check::<DashboardTabAction>();
220        check::<ConfigTabAction>();
221        check::<QuickstartTabAction>();
222        check::<InputBarAction>();
223        check::<FileExplorerAction>();
224    }
