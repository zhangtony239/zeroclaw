1//! Key chord type — a `KeyCode` + modifier mask that knows how to
2//! match incoming events and render itself per-OS.

10/// A single keystroke pattern.
11///
12/// On darwin, `CONTROL` in a chord is silently translated to `SUPER`
13/// at match time so Linux's `Ctrl+K` and macOS's `⌘K` resolve to the
14/// same chord.
15#[derive(Debug, Clone, Eq, PartialEq, Hash)]
16pub struct Chord {
17    pub code: KeyCode,
18    pub modifiers: KeyModifiers,
19}
20
21impl Chord {
22    pub const fn key(code: KeyCode) -> Self {
23        Self {
24            code,
25            modifiers: KeyModifiers::NONE,
26        }
27    }
28
29    pub const fn char(c: char) -> Self {
30        Self::key(KeyCode::Char(c))
31    }
32
33    pub const fn with(code: KeyCode, modifiers: KeyModifiers) -> Self {
34        Self { code, modifiers }
35    }
36
37    pub const fn ctrl(c: char) -> Self {
38        Self::with(KeyCode::Char(c), KeyModifiers::CONTROL)
39    }
40
41    pub const fn shift(code: KeyCode) -> Self {
42        Self::with(code, KeyModifiers::SHIFT)
43    }
44
45    pub fn matches(&self, event: &KeyEvent) -> bool {
46        event.code == self.code
47            && normalise_mods(self.code, self.modifiers)
48                == normalise_mods(event.code, event.modifiers)
49    }
50
51    /// `Ctrl+K` on most platforms; `⌘K` on darwin.
52    pub fn display(&self) -> String {
53        let mut parts: Vec<&str> = Vec::new();
54        if self.modifiers.contains(KeyModifiers::CONTROL) {
55            parts.push(if cfg!(target_os = "macos") {
56                "⌘"
57            } else {
58                "Ctrl"
59            });
60        }
61        if self.modifiers.contains(KeyModifiers::ALT) {
62            parts.push(if cfg!(target_os = "macos") {
63                "⌥"
64            } else {
65                "Alt"
66            });
67        }
68        if self.modifiers.contains(KeyModifiers::SHIFT) {
69            parts.push("Shift");
70        }
71        let key = render_keycode(&self.code);
72        if parts.is_empty() {
73            key
74        } else if cfg!(target_os = "macos") {
75            format!("{}{}", parts.join(""), key)
76        } else {
77            format!("{}+{}", parts.join("+"), key)
78        }
79    }
80
81    /// OS-independent canonical wire form used for persistence:
82    /// lowercase, `+`-joined modifiers then key, e.g. `ctrl+k`,
83    /// `shift+up`, `ctrl+shift+down`, `f5`, `pageup`. Never uses the
84    /// darwin glyphs — a config written on macOS loads identically on
85    /// Linux. Round-trips with [`Chord::from_str`].
86    pub fn wire(&self) -> String {
87        let mut out = String::new();
88        // Modifier tokens walk the canonical registry so render and
89        // parse share one source of truth — no string-literal arms.
90        for (token, flag) in MOD_TOKENS {
91            if self.modifiers.contains(*flag) {
92                out.push_str(token);
93                out.push('+');
94            }
95        }
96        out.push_str(&keycode_wire(&self.code));
97        out
98    }
99}
100
101/// Canonical modifier token registry. Walked by both `wire()` (render)
102/// and `from_str` (parse), so the two directions can never drift.
103/// `super` is accepted on parse but never emitted on non-darwin; the
104/// Ctrl→Super normalisation stays purely at match time.
105const MOD_TOKENS: &[(&str, KeyModifiers)] = &[
106    ("ctrl", KeyModifiers::CONTROL),
107    ("alt", KeyModifiers::ALT),
108    ("shift", KeyModifiers::SHIFT),
109    ("super", KeyModifiers::SUPER),
110];
111
112/// Named (non-char, non-`F<n>`) key token registry. Walked by both
113/// `keycode_wire` and the key-token parse so they stay reversible.
114const KEY_TOKENS: &[(&str, KeyCode)] = &[
115    ("enter", KeyCode::Enter),
116    ("esc", KeyCode::Esc),
117    ("tab", KeyCode::Tab),
118    ("backtab", KeyCode::BackTab),
119    ("backspace", KeyCode::Backspace),
120    ("space", KeyCode::Char(' ')),
121    ("left", KeyCode::Left),
122    ("right", KeyCode::Right),
123    ("up", KeyCode::Up),
124    ("down", KeyCode::Down),
125    ("home", KeyCode::Home),
126    ("end", KeyCode::End),
127    ("pageup", KeyCode::PageUp),
128    ("pagedown", KeyCode::PageDown),
129    ("delete", KeyCode::Delete),
130    ("insert", KeyCode::Insert),
131];
132
133/// Render a `KeyCode` to its canonical wire token. Matching on the
134/// `KeyCode` enum (not on strings) is the legitimate direction; the
135/// reverse (`&str` -> `KeyCode`) walks `KEY_TOKENS`.
136fn keycode_wire(code: &KeyCode) -> String {
137    if let KeyCode::Char(c) = code {
138        if *c == ' ' {
139            return "space".to_string();
140        }
141        return c.to_string();
142    }
143    if let KeyCode::F(n) = code {
144        return format!("f{n}");
145    }
146    KEY_TOKENS
147        .iter()
148        .find_map(|(tok, kc)| (kc == code).then(|| (*tok).to_string()))
149        .unwrap_or_else(|| format!("{code:?}").to_lowercase())
150}
151
152/// Parse a single key token (no modifiers) into a `KeyCode`. Resolves
153/// named keys through `KEY_TOKENS`, single chars to `Char`, and `f<N>`
154/// to a function key — structurally, never via string-literal arms.
155fn parse_keycode(token: &str) -> Result<KeyCode, ChordParseError> {
156    let lower = token.to_lowercase();
157    if let Some((_, kc)) = KEY_TOKENS.iter().find(|(t, _)| *t == lower) {
158        return Ok(*kc);
159    }
160    if let Some(rest) = lower.strip_prefix('f')
161        && let Ok(n) = rest.parse::<u8>()
162    {
163        return Ok(KeyCode::F(n));
164    }
165    let mut chars = token.chars();
166    match (chars.next(), chars.next()) {
167        (Some(c), None) => Ok(KeyCode::Char(c)),
168        _ => Err(ChordParseError(token.to_string())),
169    }
170}

172/// Error for an unparseable chord wire-string.
173#[derive(Debug, Clone, PartialEq, Eq)]
174pub struct ChordParseError(String);
175
176impl fmt::Display for ChordParseError {
177    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
178        write!(f, "invalid chord '{}'", self.0)
179    }
180}
181
182impl std::error::Error for ChordParseError {}
183
184impl FromStr for Chord {
185    type Err = ChordParseError;
186
187    fn from_str(s: &str) -> Result<Self, Self::Err> {
188        let trimmed = s.trim();
189        if trimmed.is_empty() {
190            return Err(ChordParseError(s.to_string()));
191        }
192        // Single-char tokens bypass modifier splitting so `+` and `=`
193        // round-trip cleanly. Without this, `trimmed.split('+')` on
194        // `"+"` yields two empty segments and the parse fails.
195        if trimmed.chars().count() == 1 {
196            let code = parse_keycode(trimmed)?;
197            return Ok(Chord {
198                code,
199                modifiers: KeyModifiers::NONE,
200            });
201        }
202        let mut segments: Vec<&str> = trimmed.split('+').collect();
203        // Last segment is the key (case preserved so 'G' stays distinct
204        // from 'g'); everything before is a modifier (case-insensitive).
205        let key_token = segments
206            .pop()
207            .ok_or_else(|| ChordParseError(s.to_string()))?;
208        let mut modifiers = KeyModifiers::NONE;
209        for seg in segments {
210            let lower = seg.to_lowercase();
211            let flag = MOD_TOKENS
212                .iter()
213                .find_map(|(t, f)| (*t == lower).then_some(*f))
214                .ok_or_else(|| ChordParseError(s.to_string()))?;
215            modifiers.insert(flag);
216        }
217        let code = parse_keycode(key_token)?;
218        Ok(Chord { code, modifiers })
219    }
220}

222impl Serialize for Chord {
223    fn serialize<S: Serializer>(&self, ser: S) -> Result<S::Ok, S::Error> {
224        ser.serialize_str(&self.wire())
225    }
226}

228impl<'de> Deserialize<'de> for Chord {
229    fn deserialize<D: Deserializer<'de>>(de: D) -> Result<Self, D::Error> {
230        let s = String::deserialize(de)?;
231        Chord::from_str(&s).map_err(de::Error::custom)
232    }

235#[cfg(target_os = "macos")]
236fn normalise_mods(code: KeyCode, mut m: KeyModifiers) -> KeyModifiers {
237    if m.contains(KeyModifiers::CONTROL) {
238        m.remove(KeyModifiers::CONTROL);
239        m.insert(KeyModifiers::SUPER);
240    }
241    strip_redundant_shift(code, m)
242}
243
244#[cfg(not(target_os = "macos"))]
245fn normalise_mods(code: KeyCode, m: KeyModifiers) -> KeyModifiers {
246    strip_redundant_shift(code, m)
247}
248
249/// Drop the SHIFT bit for character keys. A shifted character (`?`,
250/// `G`, `:`) already encodes its shift in the glyph itself, but
251/// platforms disagree on whether SHIFT is *also* reported alongside
252/// it: Unix terminals strip it, the Windows console keeps it. Comparing
253/// it would make `?` (the default Help chord) only match on platforms
254/// that strip SHIFT, forcing Windows users to hand-bind `shift+?`.
255/// Modifier keys that genuinely change the keystroke (Ctrl/Alt/Super)
256/// are left untouched.
257fn strip_redundant_shift(code: KeyCode, mut m: KeyModifiers) -> KeyModifiers {
258    if matches!(code, KeyCode::Char(_)) {
259        m.remove(KeyModifiers::SHIFT);
260    }
261    m
262}

264#[allow(dead_code)]
265fn render_keycode(code: &KeyCode) -> String {
266    match code {
267        KeyCode::Char(c) => c.to_string(),
268        KeyCode::Enter => "Enter".into(),
269        KeyCode::Esc => "Esc".into(),
270        KeyCode::Tab => "Tab".into(),
271        KeyCode::BackTab => "Shift+Tab".into(),
272        KeyCode::Backspace => "Backspace".into(),
273        KeyCode::Left => "←".into(),
274        KeyCode::Right => "→".into(),
275        KeyCode::Up => "↑".into(),
276        KeyCode::Down => "↓".into(),
277        KeyCode::Home => "Home".into(),
278        KeyCode::End => "End".into(),
279        KeyCode::PageUp => "PgUp".into(),
280        KeyCode::PageDown => "PgDn".into(),
281        KeyCode::Delete => "Del".into(),
282        KeyCode::Insert => "Ins".into(),
283        KeyCode::F(n) => format!("F{n}"),
284        other => format!("{other:?}"),
285    }
