1//! Runtime keybinding overrides.
2//!
3//! A single process-global table, keyed by each action enum's `TAG`,
4//! mapping a variant's snake_case name to its overridden chords. Read
5//! only inside the generated `resolved_bindings()`, so every consumer's
6//! `from_chord` call site stays unchanged. Populated at startup from the
7//! local config and re-populated on preset pick / capture-modal save.
8
9use std::collections::HashMap;
10use std::sync::RwLock;
11
12use super::chord::Chord;
13
14/// `tag -> (variant_name -> chords)`. Sparse at every level: an absent
15/// tag means "use compile-time defaults for that whole enum"; an absent
16/// variant within a present tag keeps that variant's default chords.
17pub type OverrideTable = HashMap<String, HashMap<String, Vec<Chord>>>;
18
19static ACTIVE: RwLock<Option<OverrideTable>> = RwLock::new(None);
20
21/// Look up the override map for one enum's `tag`. Returns a clone so the
22/// lock is not held across the rebuild in `resolved_bindings`.
23pub fn lookup(tag: &str) -> Option<HashMap<String, Vec<Chord>>> {
24    ACTIVE
25        .read()
26        .ok()
27        .and_then(|guard| guard.as_ref().and_then(|t| t.get(tag).cloned()))
28}
29
30/// Replace the entire active override table (preset pick / config load).
31pub fn set_active(table: OverrideTable) {
32    if let Ok(mut guard) = ACTIVE.write() {
33        *guard = Some(table);
34    }
35}
36
37/// Insert or replace a single `tag.variant` row, leaving the rest of the
38/// active table intact (capture-modal save). Creates the table / tag
39/// bucket on demand.
40pub fn set_row(tag: &str, variant: &str, chords: Vec<Chord>) {
41    if let Ok(mut guard) = ACTIVE.write() {
42        let table = guard.get_or_insert_with(HashMap::new);
43        table
44            .entry(tag.to_string())
45            .or_default()
46            .insert(variant.to_string(), chords);
47    }
48}
49
50/// Reset to no overrides — test isolation only.
51#[cfg(test)]
52fn reset() {
53    if let Ok(mut guard) = ACTIVE.write() {
54        *guard = None;
55    }
56}
57
58#[cfg(test)]
59mod tests {
60    use super::*;
61    use crossterm::event::KeyCode;
62    use std::sync::Mutex;
63
64    // Both tests mutate the process-wide `ACTIVE` table; serialize them so
65    // parallel execution can't clobber one test's state from another.
66    static TEST_GUARD: Mutex<()> = Mutex::new(());
67
68    #[test]
69    fn set_and_lookup_round_trips() {
70        let _g = TEST_GUARD.lock().unwrap_or_else(|e| e.into_inner());
71        let mut table = OverrideTable::new();
72        let mut dash = HashMap::new();
73        dash.insert("refresh".to_string(), vec![Chord::key(KeyCode::F(5))]);
74        table.insert("dashboard".to_string(), dash);
75        set_active(table);
76
77        let got = lookup("dashboard").expect("tag present");
78        assert_eq!(
79            got.get("refresh").unwrap(),
80            &vec![Chord::key(KeyCode::F(5))]
81        );
82        assert!(lookup("chat").is_none());
83        reset();
84        assert!(lookup("dashboard").is_none());
85    }
86
87    #[test]
88    fn set_row_creates_on_demand() {
89        let _g = TEST_GUARD.lock().unwrap_or_else(|e| e.into_inner());
90        reset();
91        set_row("logs", "toggle_follow", vec![Chord::char('F')]);
92        let got = lookup("logs").expect("tag created");
93        assert_eq!(got.get("toggle_follow").unwrap(), &vec![Chord::char('F')]);
94        reset();
95    }
96}
