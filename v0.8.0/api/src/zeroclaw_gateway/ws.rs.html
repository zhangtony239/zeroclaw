1//! WebSocket agent chat handler.
2//!
3//! Connect: `ws://host:port/ws/chat?session_id=ID&name=My+Session`
4//!
5//! Protocol:
6//! ```text
7//! Server -> Client: {"type":"session_start","session_id":"...","name":"...","resumed":true,"message_count":42}
8//! Client -> Server: {"type":"message","content":"Hello"}
9//! Server -> Client: {"type":"chunk","content":"Hi! "}
10//! Server -> Client: {"type":"tool_call","name":"shell","args":{...}}
11//! Server -> Client: {"type":"tool_result","name":"shell","output":"..."}
12//! Server -> Client: {"type":"done","full_response":"..."}
13//! ```
14//!
15//! ## Tool approvals
16//!
17//! When supervised-mode tool calls hit the `ApprovalManager`, the server
18//! emits an `approval_request` and pauses the tool loop until the client
19//! responds. Mirrors the Telegram inline-keyboard / CLI Y/N/A pattern,
20//! over the WS frame transport.
21//!
22//! ```text
23//! Server -> Client: {
24//!     "type": "approval_request",
25//!     "request_id": "<uuid>",
26//!     "tool": "shell",
27//!     "arguments_summary": "command: git status",
28//!     "timeout_secs": 120
29//! }
30//! Client -> Server: {
31//!     "type": "approval_response",
32//!     "request_id": "<uuid>",
33//!     "decision": "approve" | "deny" | "always"
34//! }
35//! ```
36//!
37//! `approve` runs the tool once, `always` adds the tool to the session
38//! allowlist for the rest of the conversation, `deny` returns a structured
39//! error to the model. When no client is connected, or the client
40//! disconnects mid-prompt, the tool call is auto-denied after `timeout_secs`.
41//!
42//! ### `arguments_summary` security boundary
43//!
44//! `arguments_summary` is a human-readable string the runtime synthesises
45//! for the operator (e.g. `"command: git status"`, `"path: /etc/hosts"`).
46//! It is render-only; the operator's approve/deny choice attaches to the
47//! `request_id`, never to the summary string. The runtime must not echo
48//! any `#[secret]` or `#[derived_from_secret]` field (auth tokens, API
49//! keys, OAuth secrets) into the summary. The agent's tool loop runs
50//! tool args through `zeroclaw_runtime::approval::summarize_args` before
51//! the request reaches this transport; do not stringify raw args here.
52//!
53//! Query params:
54//! - `session_id` — resume or create a session (default: new UUID)
55//! - `name` — optional human-readable label for the session
56//! - `token` — bearer auth token (alternative to Authorization header)

127/// Extract a bearer token from WebSocket-compatible sources.
128///
129/// Precedence (first non-empty wins):
130/// 1. `Authorization: Bearer <token>` header
131/// 2. `Sec-WebSocket-Protocol: bearer.<token>` subprotocol
132/// 3. `?token=<token>` query parameter
133///
134/// Browsers cannot set custom headers on `new WebSocket(url)`, so the query
135/// parameter and subprotocol paths are required for browser-based clients.
136fn extract_ws_token<'a>(headers: &'a HeaderMap, query_token: Option<&'a str>) -> Option<&'a str> {
137    // 1. Authorization header
138    if let Some(t) = headers
139        .get(header::AUTHORIZATION)
140        .and_then(|v| v.to_str().ok())
141        .and_then(|auth| auth.strip_prefix("Bearer "))
142        && !t.is_empty()
143    {
144        return Some(t);
145    }
146
147    // 2. Sec-WebSocket-Protocol: bearer.<token>
148    if let Some(t) = headers
149        .get("sec-websocket-protocol")
150        .and_then(|v| v.to_str().ok())
151        .and_then(|protos| {
152            protos
153                .split(',')
154                .map(|p| p.trim())
155                .find_map(|p| p.strip_prefix(BEARER_SUBPROTO_PREFIX))
156        })
157        && !t.is_empty()
158    {
159        return Some(t);
160    }
161
162    // 3. ?token= query parameter
163    if let Some(t) = query_token
164        && !t.is_empty()
165    {
166        return Some(t);
167    }
168
169    None
170}

172/// GET /ws/chat — WebSocket upgrade for agent chat
173pub async fn handle_ws_chat(
174    State(state): State<AppState>,
175    Query(params): Query<WsQuery>,
176    headers: HeaderMap,
177    ws: WebSocketUpgrade,
178) -> impl IntoResponse {
179    // Auth: check header, subprotocol, then query param (precedence order)
180    if state.pairing.require_pairing() {
181        let token = extract_ws_token(&headers, params.token.as_deref()).unwrap_or("");
182        if !state.pairing.is_authenticated(token) {
183            return (
184                axum::http::StatusCode::UNAUTHORIZED,
185                "Unauthorized — provide Authorization header, Sec-WebSocket-Protocol bearer, or ?token= query param",
186            )
187                .into_response();
188        }
189    }
190
191    // Echo Sec-WebSocket-Protocol if the client requests our sub-protocol.
192    let ws = if headers
193        .get("sec-websocket-protocol")
194        .and_then(|v| v.to_str().ok())
195        .is_some_and(|protos| protos.split(',').any(|p| p.trim() == WS_PROTOCOL))
196    {
197        ws.protocols([WS_PROTOCOL])
198    } else {
199        ws
200    };
201
202    // Reject the upgrade up-front when the client didn't pick an agent.
203    // No default — every WS session is bound to an explicit agent.
204    let Some(agent_alias) = params.agent_alias.filter(|s| !s.trim().is_empty()) else {
205        return (
206            axum::http::StatusCode::BAD_REQUEST,
207            "Missing required `agent` query parameter — pass `?agent=<alias>` matching a configured [agents.<alias>] entry.",
208        )
209            .into_response();
210    };
211    {
212        let cfg = state.config.read();
213        if cfg.agent(&agent_alias).is_none() {
214            return (
215                axum::http::StatusCode::BAD_REQUEST,
216                format!(
217                    "Unknown agent `{agent_alias}` — no [agents.{agent_alias}] entry configured."
218                ),
219            )
220                .into_response();
221        }
222    }
223
224    let session_id = params.session_id;
225    let session_name = params.name;
226    let session_cwd = params.cwd.or(params.workspace_dir);
227    ws.on_upgrade(move |socket| {
228        handle_socket(
229            socket,
230            state,
231            agent_alias,
232            session_id,
233            session_name,
234            session_cwd,
235        )
236    })
237    .into_response()
238}

243async fn resolve_ws_memory_handle(
244    config: &zeroclaw_config::schema::Config,
245    agent_alias: &str,
246) -> anyhow::Result<Option<Arc<dyn zeroclaw_memory::Memory>>> {
247    if config.agent(agent_alias).is_some_and(|agent| {
248        matches!(
249            agent.memory.backend,
250            zeroclaw_config::multi_agent::MemoryBackendKind::None
251        )
252    }) {
253        return Ok(None);
254    }
255
256    let api_key = config
257        .resolved_model_provider_for_agent(agent_alias)
258        .and_then(|(_, _, cfg)| cfg.api_key.clone());
259    zeroclaw_memory::create_memory_for_agent(config, agent_alias, api_key.as_deref())
260        .await
261        .map(Some)
262}

264async fn handle_socket(
265    socket: WebSocket,
266    state: AppState,
267    agent_alias: String,
268    session_id: Option<String>,
269    session_name: Option<String>,
270    session_cwd: Option<String>,
271) {
272    let (mut sender, mut receiver) = socket.split();
273
274    // Resolve session ID: use provided or generate a new UUID
275    let session_id = session_id.unwrap_or_else(|| uuid::Uuid::new_v4().to_string());
276    let session_key = format!("{GW_SESSION_PREFIX}{session_id}");
277    // Match the sanitized form persisted by memory backend migrations.
278    let mut memory_session_id = zeroclaw_api::session_keys::sanitize_session_key(&session_id);
279
280    // Hydrate session metadata from persistence (if available). Agent
281    // construction is deferred until after the optional `connect` frame so the
282    // client can provide a per-session cwd for the security sandbox root.
283    let config = state.config.read().clone();
284    let ws_memory = match resolve_ws_memory_handle(&config, &agent_alias).await {
285        Ok(memory) => memory,
286        Err(e) => {
287            ::zeroclaw_log::record!(
288                WARN,
289                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note)
290                    .with_outcome(::zeroclaw_log::EventOutcome::Failure)
291                    .with_attrs(::serde_json::json!({
292                        "agent": &agent_alias,
293                        "error": format!("{e:#}"),
294                    })),
295                "WS per-agent memory resolution failed; consolidation disabled for connection"
296            );
297            None
298        }
299    };
300    let mut resumed = false;
301    let mut message_count: usize = 0;
302    let mut effective_name: Option<String> = None;
303    let mut stored_messages = Vec::new();
304    if let Some(ref backend) = state.session_backend {
305        let messages = backend.load(&session_key);
306        if !messages.is_empty() {
307            message_count = messages.len();
308            stored_messages = messages;
309            resumed = true;
310        }
311        // Set session name if provided (non-empty) on connect
312        if let Some(ref name) = session_name
313            && !name.is_empty()
314        {
315            let _ = backend.set_session_name(&session_key, name);
316            effective_name = Some(name.clone());
317        }
318        // If no name was provided via query param, load the stored name
319        if effective_name.is_none() {
320            effective_name = backend.get_session_name(&session_key).unwrap_or(None);
321        }
322        // Stamp the agent alias so future /api/sessions queries and
323        // per-agent filters can attribute this session to its agent.
324        let _ = backend.set_session_agent_alias(&session_key, &agent_alias);
325    }
326
327    // Send session_start message to client
328    let mut session_start = serde_json::json!({
329        "type": "session_start",
330        "session_id": session_id,
331        "resumed": resumed,
332        "message_count": message_count,
333    });
334    if let Some(ref name) = effective_name {
335        session_start["name"] = serde_json::Value::String(name.clone());
336    }
337    let _ = sender
338        .send(Message::Text(session_start.to_string().into()))
339        .await;
340
341    // ── Optional connect handshake ──────────────────────────────────
342    // The first message may be a `{"type":"connect",...}` frame carrying
343    // connection parameters.  If it is, we extract the params, send an
344    // ack, and proceed to the normal message loop.  If the first message
345    // is a regular `{"type":"message",...}` frame, we fall through and
346    // process it immediately (backward-compatible).
347    let mut first_msg_fallback: Option<String> = None;
348    let mut requested_cwd = session_cwd;
349
350    if let Some(first) = receiver.next().await {
351        match first {
352            Ok(Message::Text(text)) => {
353                if let Ok(cp) = serde_json::from_str::<ConnectParams>(&text) {
354                    if cp.msg_type == "connect" {
355                        ::zeroclaw_log::record!(DEBUG, ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note).with_attrs(::serde_json::json!({"session_id": cp.session_id, "device_name": cp.device_name, "capabilities": cp.capabilities, "cwd": cp.cwd})), "WebSocket connect params received");
356                        if let Some(sid) = &cp.session_id {
357                            memory_session_id =
358                                zeroclaw_api::session_keys::sanitize_session_key(sid);
359                            ::zeroclaw_log::record!(
360                                DEBUG,
361                                ::zeroclaw_log::Event::new(
362                                    module_path!(),
363                                    ::zeroclaw_log::Action::Note
364                                )
365                                .with_attrs(::serde_json::json!({"session_id": sid})),
366                                "WebSocket connect session override received"
367                            );
368                        }
369                        if cp.cwd.is_some() {
370                            requested_cwd = cp.cwd;
371                        }
372                        let ack = serde_json::json!({
373                            "type": "connected",
374                            "message": "Connection established"
375                        });
376                        let _ = sender.send(Message::Text(ack.to_string().into())).await;
377                    } else {
378                        // Not a connect message — fall through to normal processing
379                        first_msg_fallback = Some(text.to_string());
380                    }
381                } else {
382                    // Not parseable as ConnectParams — fall through
383                    first_msg_fallback = Some(text.to_string());
384                }
385            }
386            Ok(Message::Close(_)) | Err(_) => return,
387            _ => {}
388        }
389    }
390
391    let session_cwd = match resolve_session_cwd(requested_cwd.as_deref(), &config.data_dir) {
392        Ok(cwd) => cwd,
393        Err(e) => {
394            let err = serde_json::json!({
395                "type": "error",
396                "message": e.to_string(),
397                "code": "INVALID_CWD"
398            });
399            let _ = sender.send(Message::Text(err.to_string().into())).await;
400            return;
401        }
402    };
403
404    if let Some(err) = needs_onboarding_ws_error(&config) {
405        let _ = sender.send(Message::Text(err.to_string().into())).await;
406        return;
407    }
408
409    // Build a persistent Agent for this connection so history is maintained
410    // across turns. The session cwd becomes the security sandbox root; config
411    // workspace remains the daemon data directory. Routes through the
412    // backchannel constructor so this WS session shares its tool-approval
413    // path with the operator-driven dashboard. The agent_alias was
414    // validated up-front in handle_ws_chat against the configured agents.
415    let mut agent =
416        match zeroclaw_runtime::agent::Agent::from_config_with_session_cwd_and_mcp_backchannel(
417            &config,
418            &agent_alias,
419            Some(&session_cwd),
420            true,
421            false,
422        )
423        .await
424        {
425            Ok(a) => a,
426            Err(e) => {
427                ::zeroclaw_log::record!(
428                    ERROR,
429                    ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Fail)
430                        .with_outcome(::zeroclaw_log::EventOutcome::Failure)
431                        .with_attrs(::serde_json::json!({"error": format!("{}", e)})),
432                    "Agent initialization failed"
433                );
434                let err = serde_json::json!({
435                    "type": "error",
436                    "message": format!("Failed to initialise agent: {e}"),
437                    "code": "AGENT_INIT_FAILED"
438                });
439                let _ = sender.send(Message::Text(err.to_string().into())).await;
440                let _ = sender
441                    .send(Message::Close(Some(axum::extract::ws::CloseFrame {
442                        code: 1011,
443                        reason: axum::extract::ws::Utf8Bytes::from_static(
444                            "Agent initialization failed",
445                        ),
446                    })))
447                    .await;
448                return;
449            }
450        };
451    agent.set_memory_session_id(Some(memory_session_id));
452    if !stored_messages.is_empty() {
453        agent.seed_history(&stored_messages);
454    }
455
456    // ── Tool-approval back-channel ─────────────────────────────────
457    // Connection-level event channel that the WsApprovalChannel shares
458    // with the per-turn forward task: it pushes ApprovalRequest frames
459    // here when the agent's tool loop pauses for consent, and the
460    // forward task drains them out the same WebSocket as the regular
461    // streaming events. The pending map is shared with the receive loop
462    // so inbound `approval_response` frames can resolve the matching
463    // oneshot waiter.
464    let (approval_event_tx, mut approval_event_rx) =
465        tokio::sync::mpsc::channel::<zeroclaw_api::agent::TurnEvent>(8);
466    let pending_approvals: PendingApprovals = new_pending_approvals();
467    let approval_channel = Arc::new(WsApprovalChannel::new(
468        approval_event_tx.clone(),
469        pending_approvals.clone(),
470        Duration::from_secs(WS_APPROVAL_TIMEOUT_SECS),
471    ));
472    agent
473        .channel_handles()
474        .register_channel("ws", approval_channel.clone());
475
476    // Seed agent's channel handles with configured channels (telegram,
477    // etc.) so the dashboard agent can deliver to external channels.
478    // The agent creates its own fresh handles in
479    // from_config_with_session_cwd_and_mcp_backchannel, so they need
480    // to be populated here — separate from the gateway boot-time seeding.
481    let ch = agent.channel_handles();
482    let channel_names = zeroclaw_channels::orchestrator::register_channels_for_tools(
483        &config,
484        &ch.ask_user,
485        &Some(ch.reaction.clone()),
486        &ch.poll,
487        &ch.escalate,
488    );
489    if !channel_names.is_empty() {
490        ::zeroclaw_log::record!(
491            INFO,
492            ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note).with_attrs(
493                ::serde_json::json!({"channels": channel_names, "session": session_key})
494            ),
495            "Seeded {} channel(s) into dashboard agent session",
496        );
497    }
498
499    // Process the first message if it was not a connect frame
500    if let Some(ref text) = first_msg_fallback {
501        if let Ok(parsed) = serde_json::from_str::<serde_json::Value>(text) {
502            if parsed["type"].as_str() == Some("message") {
503                let content = parsed["content"].as_str().unwrap_or("").to_string();
504                if !content.is_empty() {
505                    let _session_guard = match state.session_queue.acquire(&session_key).await {
506                        Ok(guard) => guard,
507                        Err(e) => {
508                            let err = serde_json::json!({
509                                "type": "error",
510                                "message": e.to_string(),
511                                "code": session_queue_ws_error_code(&e)
512                            });
513                            let _ = sender.send(Message::Text(err.to_string().into())).await;
514                            return;
515                        }
516                    };
517                    process_chat_message(
518                        &state,
519                        &mut agent,
520                        &mut sender,
521                        &mut receiver,
522                        &mut approval_event_rx,
523                        &pending_approvals,
524                        &ws_memory,
525                        &content,
526                        &session_key,
527                    )
528                    .await;
529                }
530            } else {
531                let unknown_type = parsed["type"].as_str().unwrap_or("unknown");
532                let err = serde_json::json!({
533                    "type": "error",
534                    "message": format!(
535                        "Unsupported message type \"{unknown_type}\". Send {{\"type\":\"message\",\"content\":\"your text\"}}"
536                    )
537                });
538                let _ = sender.send(Message::Text(err.to_string().into())).await;
539            }
540        } else {
541            let err = serde_json::json!({
542                "type": "error",
543                "message": "Invalid JSON. Send {\"type\":\"message\",\"content\":\"your text\"}"
544            });
545            let _ = sender.send(Message::Text(err.to_string().into())).await;
546        }
547    }
548
549    // Subscribe to the shared broadcast channel so cron/heartbeat events
550    // are forwarded to this WebSocket client.
551    let mut broadcast_rx = state.event_tx.subscribe();
552
553    loop {
554        tokio::select! {
555            // ── Client message ────────────────────────────────────────
556            client_msg = receiver.next() => {
557                let Some(msg) = client_msg else { break };
558                let msg = match msg {
559                    Ok(Message::Text(text)) => text,
560                    Ok(Message::Close(_)) | Err(_) => break,
561                    _ => continue,
562                };
563
564                // Parse incoming message
565                let parsed: serde_json::Value = match serde_json::from_str(&msg) {
566                    Ok(v) => v,
567                    Err(e) => {
568                        let err = serde_json::json!({
569                            "type": "error",
570                            "message": format!("Invalid JSON: {}", e),
571                            "code": "INVALID_JSON"
572                        });
573                        let _ = sender.send(Message::Text(err.to_string().into())).await;
574                        continue;
575                    }
576                };
577
578                let msg_type = parsed["type"].as_str().unwrap_or("");
579
580                // ── Voice duplex event dispatch (gated by feature flag + runtime config) ──
581                #[cfg(feature = "gateway-voice-duplex")]
582                {
583                    // Multi-instance shape: presence in the map = enabled.
584                    let duplex_enabled = !state.config.read().channels.voice_duplex.is_empty();
585                    if duplex_enabled {
586                        if let Some(voice_event) = crate::voice_duplex::try_parse_voice_event(&msg) {
587                            if let Some(error_frame) = crate::voice_duplex::handle_voice_event(voice_event) {
588                                let _ = sender.send(Message::Text(error_frame.to_string().into())).await;
589                            }
590                            continue;
591                        }
592                    }
593                }
594
595                // ── approval_response (operator answered a tool prompt) ──
596                if msg_type == "approval_response" {
597                    let request_id = parsed["request_id"].as_str().unwrap_or("");
598                    let decision_str = parsed["decision"].as_str().unwrap_or("");
599                    let decision = match decision_str {
600                        "approve" => Some(ChannelApprovalResponse::Approve),
601                        "always" => Some(ChannelApprovalResponse::AlwaysApprove),
602                        "deny" => Some(ChannelApprovalResponse::Deny),
603                        _ => None,
604                    };
605                    if request_id.is_empty() || decision.is_none() {
606                        let err = serde_json::json!({
607                            "type": "error",
608                            "message": "approval_response requires request_id and decision in {approve,deny,always}",
609                            "code": "INVALID_APPROVAL_RESPONSE"
610                        });
611                        let _ = sender.send(Message::Text(err.to_string().into())).await;
612                        continue;
613                    }
614                    if let Some(tx) = pending_approvals.lock().remove(request_id) {
615                        let _ = tx.send(decision.expect("checked above"));
616                    } else {
617                        ::zeroclaw_log::record!(DEBUG, ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note).with_attrs(::serde_json::json!({"request_id": request_id})), "approval_response with no matching pending request");
618                    }
619                    continue;
620                }
621
622                if msg_type != "message" {
623                    let err = serde_json::json!({
624                        "type": "error",
625                        "message": format!(
626                            "Unsupported message type \"{msg_type}\". Send {{\"type\":\"message\",\"content\":\"your text\"}}"
627                        ),
628                        "code": "UNKNOWN_MESSAGE_TYPE"
629                    });
630                    let _ = sender.send(Message::Text(err.to_string().into())).await;
631                    continue;
632                }
633
634                let content = parsed["content"].as_str().unwrap_or("").to_string();
635                if content.is_empty() {
636                    let err = serde_json::json!({
637                        "type": "error",
638                        "message": "Message content cannot be empty",
639                        "code": "EMPTY_CONTENT"
640                    });
641                    let _ = sender.send(Message::Text(err.to_string().into())).await;
642                    continue;
643                }
644
645                // Acquire session lock to serialize concurrent turns
646                let _session_guard = match state.session_queue.acquire(&session_key).await {
647                    Ok(guard) => guard,
648                    Err(e) => {
649                        let err = serde_json::json!({
650                            "type": "error",
651                            "message": e.to_string(),
652                            "code": session_queue_ws_error_code(&e)
653                        });
654                        let _ = sender.send(Message::Text(err.to_string().into())).await;
655                        continue;
656                    }
657                };
658
659                process_chat_message(
660                    &state,
661                    &mut agent,
662                    &mut sender,
663                    &mut receiver,
664                    &mut approval_event_rx,
665                    &pending_approvals,
666                    &ws_memory,
667                    &content,
668                    &session_key,
669                )
670                .await;
671            }
672
673            // ── Broadcast event (cron/heartbeat results) ──────────────
674            event = broadcast_rx.recv() => {
675                if let Ok(event) = event
676                    && event_matches_session(&event, &session_id)
677                    && !is_observability_telemetry(&event)
678                {
679                    let _ = sender.send(Message::Text(event.to_string().into())).await;
680                }
681            }
682
683            // ── Approval request from the agent's tool loop ────────────
684            // The WsApprovalChannel emits these whenever a supervised tool
685            // call needs operator consent. Forwarded out the same socket
686            // as the regular streaming events; the matching response
687            // arrives via the `approval_response` arm above and resolves
688            // the channel's pending oneshot.
689            approval_event = approval_event_rx.recv() => {
690                let Some(event) = approval_event else { break };
691                let frame = match event {
692                    zeroclaw_api::agent::TurnEvent::ApprovalRequest {
693                        request_id,
694                        tool_name,
695                        arguments_summary,
696                        timeout_secs,
697                    } => serde_json::json!({
698                        "type": "approval_request",
699                        "request_id": request_id,
700                        "tool": tool_name,
701                        "arguments_summary": arguments_summary,
702                        "timeout_secs": timeout_secs,
703                    }),
704                    other => {
705                        ::zeroclaw_log::record!(WARN, ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note).with_outcome(::zeroclaw_log::EventOutcome::Unknown).with_attrs(::serde_json::json!({"kind": format!("{:?}", other)})), "non-ApprovalRequest event leaked into approval channel");
706                        continue;
707                    }
708                };
709                let _ = sender.send(Message::Text(frame.to_string().into())).await;
710            }
711        }
712    }
713}

747fn persist_conversation_messages(
748    backend: &dyn zeroclaw_infra::session_backend::SessionBackend,
749    session_key: &str,
750    messages: &[zeroclaw_providers::ConversationMessage],
751) {
752    // #7126: if the user deleted the session between the turn starting and
753    // the post-turn persistence, don't resurrect it. The `aborted` / `done`
754    // / `error` frames are still sent to the client; we just refuse to
755    // re-create the row that `DELETE /api/sessions/{id}` just wiped.
756    if !backend.session_exists(session_key) {
757        return;
758    }
759    for message in messages {
760        let zeroclaw_providers::ConversationMessage::Chat(message) = message else {
761            continue;
762        };
763        if message.role == "system" {
764            continue;
765        }
766        let _ = backend.append(session_key, message);
767    }
768}

794/// Returns true when a broadcast frame should be forwarded to the chat
795/// WebSocket subscribed to `session_id`.
796///
797/// Contract (mirrors `sse.rs::is_public_sse_event`): broadcast events must
798/// not include `session_id` unless they are intentionally scoped to that
799/// session. Frames without a `session_id` are therefore **global
800/// monitoring/observability events** — they belong on `/api/events`, not in
801/// per-session chat channels. The chat WebSocket only forwards a frame when
802/// it is either:
803///
804/// * explicitly scoped to this session via `session_id == session`, or
805/// * a global system event the chat UI is known to render (whitelisted in
806///   [`is_global_chat_event`]) — currently just `cron_result`.
807///
808/// Everything else (observability telemetry, log records, error broadcasts
809/// from unrelated subsystems, …) is dropped. Before #7151 this defaulted to
810/// `None => true`, which leaked `BroadcastObserver` telemetry — including a
811/// red `error` bubble — into every active chat user's view.
812fn event_matches_session(event: &serde_json::Value, session_id: &str) -> bool {
813    match event.get("session_id").and_then(|value| value.as_str()) {
814        Some(event_session_id) => event_session_id == session_id,
815        None => is_global_chat_event(event),
816    }
817}
818
819/// Whitelist of broadcast event `type` values that all chat WebSockets
820/// should receive even without a `session_id` scope.
821///
822/// Today this is just `cron_result` (the scheduler's automatic cron output
823/// and the manual `/api/cron/<id>/trigger` rebroadcast, both rendered by
824/// `AgentContext.tsx` as a markdown bubble). New entries must be backed by
825/// a matching `case` in the frontend message dispatcher — otherwise the
826/// frame is dead weight on the wire.
827fn is_global_chat_event(event: &serde_json::Value) -> bool {
828    matches!(
829        event.get("type").and_then(serde_json::Value::as_str),
830        Some("cron_result")
831    )
832}
833
834/// Defense-in-depth check for observability telemetry frames that leak onto
835/// the chat broadcast bus.
836///
837/// After #7151 the primary defense is [`event_matches_session`]'s inverted
838/// default — any frame without `session_id` is dropped unless explicitly
839/// whitelisted. This helper exists as a belt-and-braces guard for the case
840/// where a future emitter forgets `session_id` *and* its event type collides
841/// with a global-whitelisted one (e.g. someone adding `cron_result`-shaped
842/// telemetry). The discriminator is the `"source": "observability"` tag
843/// that `BroadcastObserver` (sse.rs) stamps on every emission.
844fn is_observability_telemetry(event: &serde_json::Value) -> bool {
845    event.get("source").and_then(serde_json::Value::as_str) == Some("observability")
846}

848/// Process a single chat message through the agent and send the response.
849///
850/// Uses [`Agent::turn_streamed`] so that intermediate text chunks, tool calls,
851/// and tool results are forwarded to the WebSocket client in real time.
852async fn process_chat_message(
853    state: &AppState,
854    agent: &mut zeroclaw_runtime::agent::Agent,
855    sender: &mut futures_util::stream::SplitSink<WebSocket, Message>,
856    receiver: &mut futures_util::stream::SplitStream<WebSocket>,
857    approval_event_rx: &mut tokio::sync::mpsc::Receiver<zeroclaw_api::agent::TurnEvent>,
858    pending_approvals: &PendingApprovals,
859    ws_memory: &Option<Arc<dyn zeroclaw_memory::Memory>>,
860    content: &str,
861    session_key: &str,
862) {
863    use futures_util::StreamExt as _;
864    use zeroclaw_runtime::agent::TurnEvent;
865
866    // Attribute telemetry, broadcasts, and cost to THIS agent's actual model
867    // (resolved per-turn), not the global default model or the first configured
868    // provider. Previously `provider_label` took the first `providers.models`
869    // entry and the model came from `model_label` (the global default), so every
870    // gateway_ws_turn / agent_start / cost record mislabelled the model.
871    let (turn_alias, turn_provider, turn_model) = agent.attribution_fields();
872    let provider_label = turn_provider.clone();
873    let model_label = turn_model.clone();
874
875    // Broadcast agent_start event
876    let _ = state.event_tx.send(serde_json::json!({
877        "type": "agent_start",
878        "model_provider": provider_label,
879        "model": model_label,
880    }));
881
882    // Set session state to running
883    let turn_id = uuid::Uuid::new_v4().to_string();
884    if let Some(ref backend) = state.session_backend {
885        let _ = backend.set_session_state(session_key, "running", Some(&turn_id));
886    }
887
888    // ── Cancellation token lifecycle ─────────────────────────────
889    // Create a token before the turn starts so the abort endpoint
890    // can cancel it. Remove it after the turn completes regardless
891    // of outcome (normal, error, or cancelled).
892    let cancel_token = tokio_util::sync::CancellationToken::new();
893    {
894        state
895            .cancel_tokens
896            .lock()
897            .expect("cancel_tokens lock poisoned")
898            .insert(session_key.to_string(), cancel_token.clone());
899    }
900
901    // Channel for streaming turn events from the agent.
902    let (event_tx, mut event_rx) = tokio::sync::mpsc::channel::<TurnEvent>(64);
903    let (steering_tx, mut steering_rx) = tokio::sync::mpsc::channel::<String>(32);
904
905    // Run the streamed turn concurrently: the agent produces events
906    // while we forward them to the WebSocket below.  We cannot move
907    // `agent` into a spawned task (it is `&mut`), so we use a join
908    // instead — `turn_streamed` writes to the channel and we drain it
909    // from the other branch.
910    let content_owned = content.to_string();
911    let session_key_owned = session_key.to_string();
912    let turn_fut = async {
913        use ::zeroclaw_log::Instrument as _;
914        let span = ::zeroclaw_log::info_span!(
915            target: "zeroclaw_log_internal_scope",
916            "zeroclaw_scope",
917            session_key = %session_key_owned,
918            agent_alias = %turn_alias,
919            model_provider = %turn_provider,
920            model = %turn_model,
921            channel = "wss",
922        );
923        zeroclaw_runtime::agent::loop_::scope_session_key(
924            Some(session_key_owned.clone()),
925            agent
926                .turn_streamed_with_steering_state(
927                    &content_owned,
928                    event_tx,
929                    Some(cancel_token.clone()),
930                    Some(&mut steering_rx),
931                )
932                .instrument(span),
933        )
934        .await
935    };
936
937    // Drive both futures concurrently: the agent turn produces events
938    // and we relay them over WebSocket. Track streamed chunks so we
939    // can reconstruct partial content on cancellation.
940    //
941    let mut accumulated_text = String::new();
942
943    // Aggregate token usage across all LLM calls in this turn.
944    // The agent emits TurnEvent::Usage once per LLM call when the provider
945    // surfaces usage; we sum to produce a single done-frame total.
946    let mut total_input_tokens: Option<u64> = None;
947    let mut total_output_tokens: Option<u64> = None;
948
949    // Routes the three concurrent streams that the running turn cares about:
950    //   1. inbound `approval_response` frames from the WebSocket client,
951    //   2. `TurnEvent::ApprovalRequest` events from `WsApprovalChannel`,
952    //   3. ordinary `TurnEvent`s from the agent loop.
953    // Without the multiplexed select, the loop draining only `event_rx`
954    // would block the approval back-channel for the whole turn, so a pending
955    // tool approval could neither be sent to the client nor answered before
956    // the timeout fired.
957    let forward_fut = async {
958        let mut cancel_drained = false;
959        loop {
960            tokio::select! {
961                biased;
962                // ── Cancellation arm ─────────────────────────────
963                // When `/abort` cancels the token, immediately drop every
964                // parked oneshot sender so any in-flight `request_approval`
965                // unblocks via the "sender dropped → deny" path in
966                // `WsApprovalChannel`. Without this, the approval future
967                // races only its own `timeout_secs` (default 120s) and
968                // ignores the cancel token, so the abort sits idle for up
969                // to two minutes before the tool loop even gets a chance
970                // to observe the cancellation.
971                _ = cancel_token.cancelled(), if !cancel_drained => {
972                    let drained: Vec<_> = pending_approvals.lock().drain().collect();
973                    drop(drained);
974                    cancel_drained = true;
975                    // Fall through; the agent loop will now wake from the
976                    // approval await, see the cancel token, and propagate
977                    // a ToolLoopCancelled error which closes event_rx and
978                    // breaks this loop on the `event_rx.recv()` arm below.
979                }
980                client_msg = receiver.next() => {
981                    // On client disconnect, `receiver.next()` returns `None`
982                    // (stream end) or `Err(_)` repeatedly. A bare `continue`
983                    // hot-loops the select; cancel the turn so `turn_fut`
984                    // resolves with `ToolLoopCancelled` and `tokio::join!`
985                    // below can return. See #6514.
986                    let text = match client_msg {
987                        Some(Ok(Message::Text(text))) => text,
988                        Some(Ok(Message::Close(_))) | Some(Err(_)) | None => {
989                            cancel_token.cancel();
990                            break;
991                        }
992                        _ => continue,
993                    };
994                    let Ok(parsed) = serde_json::from_str::<serde_json::Value>(&text) else {
995                        let err = serde_json::json!({
996                            "type": "error",
997                            "message": "Invalid JSON. Send {\"type\":\"message\",\"content\":\"your text\"}",
998                            "code": "INVALID_JSON"
999                        });
1000                        let _ = sender.send(Message::Text(err.to_string().into())).await;
1001                        continue;
1002                    };
1003                    match parsed["type"].as_str() {
1004                        Some("approval_response") => {
1005                            let request_id = parsed["request_id"].as_str().unwrap_or("");
1006                            let decision = match parsed["decision"].as_str().unwrap_or("") {
1007                                "approve" => Some(ChannelApprovalResponse::Approve),
1008                                "always" => Some(ChannelApprovalResponse::AlwaysApprove),
1009                                "deny" => Some(ChannelApprovalResponse::Deny),
1010                                _ => None,
1011                            };
1012                            if request_id.is_empty() || decision.is_none() {
1013                                continue;
1014                            }
1015                            if let Some(tx) = pending_approvals.lock().remove(request_id) {
1016                                let _ = tx.send(decision.expect("checked above"));
1017                            } else {
1018                                ::zeroclaw_log::record!(DEBUG, ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note).with_attrs(::serde_json::json!({"request_id": request_id})), "approval_response with no matching pending request (mid-turn)");
1019                            }
1020                        }
1021                        Some("message") => {
1022                            let content = parsed["content"].as_str().unwrap_or("").to_string();
1023                            if content.is_empty() {
1024                                let err = serde_json::json!({
1025                                    "type": "error",
1026                                    "message": "Message content cannot be empty",
1027                                    "code": "EMPTY_CONTENT"
1028                                });
1029                                let _ = sender.send(Message::Text(err.to_string().into())).await;
1030                                continue;
1031                            }
1032                            match steering_tx.try_send(content) {
1033                                Ok(()) => {}
1034                                Err(tokio::sync::mpsc::error::TrySendError::Full(_)) => {
1035                                    let err = serde_json::json!({
1036                                        "type": "error",
1037                                        "message": "Steering queue is full for the running turn",
1038                                        "code": "STEERING_QUEUE_FULL"
1039                                    });
1040                                    let _ = sender.send(Message::Text(err.to_string().into())).await;
1041                                }
1042                                Err(tokio::sync::mpsc::error::TrySendError::Closed(_)) => {
1043                                    let err = serde_json::json!({
1044                                        "type": "error",
1045                                        "message": "Running turn is no longer accepting steering messages",
1046                                        "code": "STEERING_CLOSED"
1047                                    });
1048                                    let _ = sender.send(Message::Text(err.to_string().into())).await;
1049                                }
1050                            }
1051                        }
1052                        _ => {}
1053                    }
1054                }
1055                approval = approval_event_rx.recv() => {
1056                    let Some(event) = approval else { continue };
1057                    if let TurnEvent::ApprovalRequest {
1058                        request_id,
1059                        tool_name,
1060                        arguments_summary,
1061                        timeout_secs,
1062                    } = event {
1063                        let frame = serde_json::json!({
1064                            "type": "approval_request",
1065                            "request_id": request_id,
1066                            "tool": tool_name,
1067                            "arguments_summary": arguments_summary,
1068                            "timeout_secs": timeout_secs,
1069                        });
1070                        let _ = sender.send(Message::Text(frame.to_string().into())).await;
1071                    }
1072                }
1073                    event_opt = event_rx.recv() => {
1074                    let Some(event) = event_opt else { break };
1075                    let ws_msg = match event {
1076                        TurnEvent::Usage {
1077                            input_tokens,
1078                            cached_input_tokens: _,
1079                            output_tokens,
1080                            cost_usd: _,
1081                        } => {
1082                            // `input_tokens` per TokenUsage contract is
1083                            // the *total* prompt size (uncached + cached).
1084                            // `cached_input_tokens` is a subset and must
1085                            // NOT be added — that would double-count
1086                            // cache reads.
1087                            if let Some(it) = input_tokens {
1088                                total_input_tokens = Some(total_input_tokens.unwrap_or(0) + it);
1089                            }
1090                            if let Some(ot) = output_tokens {
1091                                total_output_tokens = Some(total_output_tokens.unwrap_or(0) + ot);
1092                            }
1093                            continue;
1094                        }
1095                        TurnEvent::Chunk { ref delta } => {
1096                            accumulated_text.push_str(delta);
1097                            serde_json::json!({ "type": "chunk", "content": delta })
1098                        }
1099                        TurnEvent::Thinking { delta } => {
1100                            serde_json::json!({ "type": "thinking", "content": delta })
1101                        }
1102                        TurnEvent::ToolCall { id, name, args } => {
1103                            serde_json::json!({ "type": "tool_call", "id": id, "name": name, "args": args })
1104                        }
1105                        TurnEvent::ToolResult { id, name, output } => {
1106                            serde_json::json!({ "type": "tool_result", "id": id, "name": name, "output": output })
1107                        }
1108                        TurnEvent::ApprovalRequest {
1109                            request_id,
1110                            tool_name,
1111                            arguments_summary,
1112                            timeout_secs,
1113                        } => serde_json::json!({
1114                            "type": "approval_request",
1115                            "request_id": request_id,
1116                            "tool": tool_name,
1117                            "arguments_summary": arguments_summary,
1118                            "timeout_secs": timeout_secs,
1119                        }),
1120                    };
1121                    let _ = sender.send(Message::Text(ws_msg.to_string().into())).await;
1122                }
1123            }
1124        }
1125    };
1126
1127    let (result, ()) = tokio::join!(turn_fut, forward_fut);
1128
1129    // ── Remove cancel token (turn finished) ──────────────────────
1130    {
1131        state
1132            .cancel_tokens
1133            .lock()
1134            .expect("cancel_tokens lock poisoned")
1135            .remove(session_key);
1136    }
1137
1138    // Check if this turn was cancelled. `turn_streamed` propagates
1139    // `ToolLoopCancelled` through anyhow, so we detect it here.
1140    let was_cancelled = match &result {
1141        Err(e) => zeroclaw_runtime::agent::loop_::is_tool_loop_cancelled(&e.error),
1142        Ok(_) => false,
1143    };
1144
1145    if was_cancelled {
1146        if let Some(ref backend) = state.session_backend {
1147            // #7126: `DELETE /api/sessions/{id}` cancels the token and then
1148            // synchronously wipes the session row. The streaming task then
1149            // wakes up here with `was_cancelled = true`. If we blindly
1150            // append "[interrupted by user]" we resurrect both the
1151            // `sessions` row and the `session_metadata` row (via the
1152            // upsert inside `append`), and the next reconnect re-seeds the
1153            // resurrected history. Skip every write when the session no
1154            // longer exists — the `aborted` frame below still tells the
1155            // client the turn ended.
1156            let still_exists = backend.session_exists(session_key);
1157            if still_exists {
1158                match &result {
1159                    Err(error) if !error.new_messages.is_empty() => {
1160                        persist_conversation_messages(
1161                            backend.as_ref(),
1162                            session_key,
1163                            &error.new_messages,
1164                        );
1165                        if !has_assistant_chat_message(&error.new_messages) {
1166                            let truncated = if accumulated_text.is_empty() {
1167                                "[interrupted by user]".to_string()
1168                            } else {
1169                                format!("{accumulated_text}\n\n[interrupted by user]")
1170                            };
1171                            let assistant_msg =
1172                                zeroclaw_providers::ChatMessage::assistant(&truncated);
1173                            // Re-check before the raw append — the user can
1174                            // delete the session between the outer check and
1175                            // here; `persist_conversation_messages` already
1176                            // re-checks internally.
1177                            if backend.session_exists(session_key) {
1178                                let _ = backend.append(session_key, &assistant_msg);
1179                            }
1180                        }
1181                    }
1182                    _ => {
1183                        let truncated = if accumulated_text.is_empty() {
1184                            "[interrupted by user]".to_string()
1185                        } else {
1186                            format!("{accumulated_text}\n\n[interrupted by user]")
1187                        };
1188                        let assistant_msg = zeroclaw_providers::ChatMessage::assistant(&truncated);
1189                        if backend.session_exists(session_key) {
1190                            let _ = backend.append(session_key, &assistant_msg);
1191                        }
1192                    }
1193                }
1194            }
1195        }
1196
1197        // Inform the client the turn was aborted
1198        let aborted = serde_json::json!({ "type": "aborted" });
1199        let _ = sender.send(Message::Text(aborted.to_string().into())).await;
1200
1201        // Set session state to idle — but only for sessions that still
1202        // exist (#7126). `set_session_state` UPDATEs `session_metadata`,
1203        // so on a deleted session it's a harmless no-op (0 rows updated)
1204        // for SQLite but we still guard for cheap consistency with the
1205        // append path above.
1206        if let Some(ref backend) = state.session_backend
1207            && backend.session_exists(session_key)
1208        {
1209            let _ = backend.set_session_state(session_key, "idle", None);
1210        }
1211
1212        // Broadcast agent_end event
1213        let _ = state.event_tx.send(serde_json::json!({
1214            "type": "agent_end",
1215            "model_provider": provider_label,
1216            "model": model_label,
1217        }));
1218
1219        // Trace the cancelled turn so the doctor / replay tool sees it
1220        // alongside successful turns. #6001 follow-through.
1221        ::zeroclaw_log::record!(
1222            INFO,
1223            ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Cancel)
1224                .with_outcome(::zeroclaw_log::EventOutcome::Failure)
1225                .with_attrs(::serde_json::json!({
1226                    "model_provider": provider_label,
1227                    "model": model_label,
1228                    "session_key": session_key,
1229                    "reason": "interrupted by user",
1230                    "cancelled": true,
1231                    "trace_id": turn_id,
1232                })),
1233            "gateway_ws_turn"
1234        );
1235
1236        return;
1237    }
1238
1239    match result {
1240        Ok(outcome) => {
1241            if let Some(ref backend) = state.session_backend {
1242                persist_conversation_messages(backend.as_ref(), session_key, &outcome.new_messages);
1243            }
1244
1245            // Fire-and-forget memory consolidation so facts from WS sessions
1246            // are extracted to long-term memory (Daily + Core categories).
1247            if state.auto_save {
1248                if let Some(mem) = ws_memory.clone() {
1249                    let model_provider = state.model_provider.clone();
1250                    let model = state.model.clone();
1251                    let temperature = state.temperature;
1252                    let user_msg = content.to_string();
1253                    let assistant_resp = outcome.response.clone();
1254                    zeroclaw_spawn::spawn!(async move {
1255                        if let Err(e) = zeroclaw_memory::consolidation::consolidate_turn(
1256                            model_provider.as_ref(),
1257                            &model,
1258                            temperature,
1259                            mem.as_ref(),
1260                            &user_msg,
1261                            &assistant_resp,
1262                        )
1263                        .await
1264                        {
1265                            ::zeroclaw_log::record!(
1266                                DEBUG,
1267                                ::zeroclaw_log::Event::new(
1268                                    module_path!(),
1269                                    ::zeroclaw_log::Action::Note
1270                                )
1271                                .with_attrs(::serde_json::json!({"error": format!("{}", e)})),
1272                                "WS memory consolidation skipped"
1273                            );
1274                        }
1275                    });
1276                } else {
1277                    ::zeroclaw_log::record!(
1278                        DEBUG,
1279                        ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note),
1280                        "WS memory consolidation skipped"
1281                    );
1282                }
1283            }
1284
1285            // Compute cost from accumulated tokens + configured pricing,
1286            // then write the cost record so /api/cost and costs.jsonl reflect
1287            // this turn. Done before the done frame so cost_usd can ride along.
1288            let total_tokens = match (total_input_tokens, total_output_tokens) {
1289                (Some(i), Some(o)) => Some(i.saturating_add(o)),
1290                (Some(i), None) => Some(i),
1291                (None, Some(o)) => Some(o),
1292                (None, None) => None,
1293            };
1294            let cost_usd = record_turn_cost(
1295                state,
1296                &provider_label,
1297                &model_label,
1298                total_input_tokens,
1299                total_output_tokens,
1300                None,
1301            );
1302
1303            let done = serde_json::json!({
1304                "type": "done",
1305                "full_response": outcome.response,
1306                "input_tokens": total_input_tokens,
1307                "output_tokens": total_output_tokens,
1308                "tokens_used": total_tokens,
1309                "cost_usd": cost_usd,
1310                "model": model_label,
1311                "provider": provider_label,
1312            });
1313            let _ = sender.send(Message::Text(done.to_string().into())).await;
1314
1315            // Set session state to idle
1316            if let Some(ref backend) = state.session_backend {
1317                let _ = backend.set_session_state(session_key, "idle", None);
1318            }
1319
1320            // Broadcast agent_end event
1321            let _ = state.event_tx.send(serde_json::json!({
1322                "type": "agent_end",
1323                "model_provider": provider_label,
1324                "model": model_label,
1325            }));
1326
1327            // Append a runtime-trace.jsonl record so a `zeroclaw doctor`
1328            // sweep sees gateway WS turns alongside channel and CLI turns.
1329            // Closes the gateway-side trace gap from #6001.
1330            ::zeroclaw_log::record!(
1331                INFO,
1332                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Complete)
1333                    .with_outcome(::zeroclaw_log::EventOutcome::Success)
1334                    .with_attrs(::serde_json::json!({
1335                        "model_provider": provider_label,
1336                        "model": model_label,
1337                        "session_key": session_key,
1338                        "input_tokens": total_input_tokens,
1339                        "output_tokens": total_output_tokens,
1340                        "tokens_used": total_tokens,
1341                        "cost_usd": cost_usd,
1342                        "trace_id": turn_id,
1343                    })),
1344                "gateway_ws_turn"
1345            );
1346        }
1347        Err(e) => {
1348            if let Some(ref backend) = state.session_backend
1349                && !e.new_messages.is_empty()
1350            {
1351                persist_conversation_messages(backend.as_ref(), session_key, &e.new_messages);
1352            }
1353
1354            // Set session state to error
1355            if let Some(ref backend) = state.session_backend {
1356                let _ = backend.set_session_state(session_key, "error", Some(&turn_id));
1357            }
1358
1359            ::zeroclaw_log::record!(
1360                ERROR,
1361                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Fail)
1362                    .with_outcome(::zeroclaw_log::EventOutcome::Failure)
1363                    .with_attrs(::serde_json::json!({"error": format!("{}", e.error)})),
1364                "Agent turn failed"
1365            );
1366            let sanitized = zeroclaw_providers::sanitize_api_error(&e.error.to_string());
1367            let error_code = if sanitized.to_lowercase().contains("api key")
1368                || sanitized.to_lowercase().contains("authentication")
1369                || sanitized.to_lowercase().contains("unauthorized")
1370            {
1371                "AUTH_ERROR"
1372            } else if sanitized.to_lowercase().contains("model_provider")
1373                || sanitized.to_lowercase().contains("model")
1374            {
1375                "PROVIDER_ERROR"
1376            } else {
1377                "AGENT_ERROR"
1378            };
1379            let err = serde_json::json!({
1380                "type": "error",
1381                "message": sanitized,
1382                "code": error_code,
1383            });
1384            let _ = sender.send(Message::Text(err.to_string().into())).await;
1385
1386            // Broadcast error event
1387            let _ = state.event_tx.send(serde_json::json!({
1388                "type": "error",
1389                "component": "ws_chat",
1390                "message": sanitized,
1391            }));
1392
1393            // Trace the failed turn so the doctor / replay tool sees the
1394            // failure mode and the turn_id can be cross-referenced with
1395            // costs.jsonl. #6001 follow-through.
1396            ::zeroclaw_log::record!(
1397                WARN,
1398                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Fail)
1399                    .with_outcome(::zeroclaw_log::EventOutcome::Failure)
1400                    .with_attrs(::serde_json::json!({
1401                        "model_provider": provider_label,
1402                        "model": model_label,
1403                        "session_key": session_key,
1404                        "error": sanitized,
1405                        "error_code": error_code,
1406                        "trace_id": turn_id,
1407                    })),
1408                "gateway_ws_turn"
1409            );
1410        }
1411    }

1414/// Record token usage for the just-completed turn against the gateway's
1415/// cost tracker, returning the computed cost in USD (or `None` when no
1416/// tracker is configured or no usage was reported).
1417fn record_turn_cost(
1418    state: &AppState,
1419    provider_name: &str,
1420    model: &str,
1421    input_tokens: Option<u64>,
1422    output_tokens: Option<u64>,
1423    cached_input_tokens: Option<u64>,
1424) -> Option<f64> {
1425    let tracker = state.cost_tracker.as_ref()?;
1426    if input_tokens.is_none() && output_tokens.is_none() {
1427        return None;
1428    }
1429    let input = input_tokens.unwrap_or(0);
1430    let output = output_tokens.unwrap_or(0);
1431    let cached_input = cached_input_tokens.unwrap_or(0);
1432    if input == 0 && output == 0 {
1433        return None;
1434    }
1435    // V3 per-provider pricing lookup. Mirrors how the channels
1436    // orchestrator and the gateway lib.rs cost-tracking scope build
1437    // their `ModelProviderPricing`: walk every
1438    // `[providers.models.<type>.<alias>]` and key the per-profile
1439    // pricing map by `<type>.<alias>`. The streaming and non-streaming
1440    // paths derive identical costs because both bottom out in the same
1441    // `<type>.<alias>` key shape.
1442    let config = state.config.read();
1443    let pricing_map = config
1444        .providers
1445        .models
1446        .iter_entries()
1447        .filter(|(_, _, base)| !base.pricing.is_empty())
1448        .map(|(type_k, alias_k, base)| (format!("{type_k}.{alias_k}"), base.pricing.clone()))
1449        .collect::<std::collections::HashMap<String, std::collections::HashMap<String, f64>>>();
1450    drop(config);
1451    let model_pricing = pricing_map.get(provider_name);
1452    let try_lookup = |key: &str| -> (f64, f64, f64) {
1453        let Some(map) = model_pricing else {
1454            return (0.0, 0.0, 0.0);
1455        };
1456        let in_rate = map
1457            .get(&format!("{key}.input"))
1458            .copied()
1459            .or_else(|| map.get(key).copied())
1460            .unwrap_or(0.0);
1461        let out_rate = map
1462            .get(&format!("{key}.output"))
1463            .copied()
1464            .or_else(|| map.get(key).copied())
1465            .unwrap_or(0.0);
1466        let cached_rate = map
1467            .get(&format!("{key}.cached_input"))
1468            .copied()
1469            .unwrap_or(0.0);
1470        (in_rate, out_rate, cached_rate)
1471    };
1472    let (input_rate, output_rate, cached_rate) = match try_lookup(model) {
1473        (0.0, 0.0, 0.0) => model
1474            .rsplit_once('/')
1475            .map(|(_, suffix)| try_lookup(suffix))
1476            .unwrap_or((0.0, 0.0, 0.0)),
1477        rates => rates,
1478    };
1479    let usage = zeroclaw_runtime::cost::types::TokenUsage::new(
1480        model,
1481        input,
1482        output,
1483        cached_input,
1484        input_rate,
1485        output_rate,
1486        cached_rate,
1487    );
1488    let cost_usd = usage.cost_usd;
1489    if let Err(error) = tracker.record_usage(usage) {
1490        ::zeroclaw_log::record!(WARN, ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note).with_outcome(::zeroclaw_log::EventOutcome::Unknown).with_attrs(::serde_json::json!({"provider": provider_name, "model": model, "error": format!("{}", error)})), "Failed to record gateway turn cost");
1491    }
1492    Some(cost_usd)
1493}

1814    // must (a) classify stream-end / close / error frames as "client gone"

1816    // can return — a bare `continue` hot-loops the select forever.

1820        Continue,
1821        ProcessText,
1822    }

1824    fn classify_client_msg(
1825        msg: Option<Result<axum::extract::ws::Message, &'static str>>,
1826    ) -> DisconnectAction {
1827        use axum::extract::ws::Message;
1828        match msg {
1829            Some(Ok(Message::Text(_))) => DisconnectAction::ProcessText,
1830            Some(Ok(Message::Close(_))) | Some(Err(_)) | None => DisconnectAction::Break,
1831            _ => DisconnectAction::Continue,
1832        }
1833    }

1888    // ── #7126 regression ──────────────────────────────────────────────
1889    //
1890    // A `SessionBackend` mock that pretends the session has been deleted
1891    // (`session_exists` → false). `persist_conversation_messages` must
1892    // not call `append` against it — otherwise the SQLite backend's
1893    // `INSERT INTO sessions` + the metadata-upsert resurrect both rows
1894    // for a session the user explicitly wiped via
1895    // `DELETE /api/sessions/{id}` during a streaming turn, and the next
1896    // reconnect re-seeds the partial pre-clear history.
1897    //
1898    // Manual repro (no automated harness for the full streaming flow):
1899    //   1. start a long turn (e.g. ask the agent to count slowly).
1900    //   2. while the assistant is still streaming, click "Clear all".
1901    //   3. wait for the WebSocket to reconnect.
1902    //   4. ask "what did we talk about?" — pre-fix, the agent recalls
1903    //      the partial pre-clear conversation; post-fix, it does not.
1904    struct DeletedSessionBackend {
1905        append_calls: std::sync::Mutex<Vec<String>>,
1906    }
1907
1908    impl zeroclaw_infra::session_backend::SessionBackend for DeletedSessionBackend {
1909        fn load(&self, _session_key: &str) -> Vec<zeroclaw_providers::ChatMessage> {
1910            Vec::new()
1911        }
1912        fn append(
1913            &self,
1914            session_key: &str,
1915            message: &zeroclaw_providers::ChatMessage,
1916        ) -> std::io::Result<()> {
1917            self.append_calls.lock().unwrap().push(format!(
1918                "{}:{}:{}",
1919                session_key, message.role, message.content
1920            ));
1921            Ok(())
1922        }
1923        fn remove_last(&self, _session_key: &str) -> std::io::Result<bool> {
1924            Ok(false)
1925        }
1926        fn list_sessions(&self) -> Vec<String> {
1927            Vec::new()
1928        }
1929        fn session_exists(&self, _session_key: &str) -> bool {
1930            // The user deleted the session between cancel and append.
1931            false
1932        }
1933    }
